## **🎯 OBJECTIVE**

**Goal**: Transform ONCE following TRUE Web4 component pattern - each class is its own component with own model

**Web4 Principles (from Web4Articles/ONCE/0.3.1.1)**:
1. ✅ **ONCE is a kernel** - loads components from IORs, NOT a server itself
2. ✅ **Each class = Component** - own model, empty constructor, init(scenario)
3. ✅ **One file one type** - strictly enforced
4. ✅ **Scenario-driven composition** - components compose via scenarios
5. ✅ **Component instances belong to class** - NOT in model
6. ✅ **Path Authority** - derive paths once, store in model
7. ✅ **JavaBean style** - private data field, public getters

**Current Anti-Patterns (0.3.21.1)**:
- ❌ ONCE directly starts ServerHierarchyManager (not a kernel)
- ❌ All state mixed in ONCEModel (router state, TLS state, etc.)
- ❌ ServerHierarchyManager is not a component
- ❌ No clear separation between ONCE kernel and server capabilities

**Target Architecture (0.3.21.2)**:
- ✅ ONCE = kernel that loads components
- ✅ HTTPSServer = component with HTTPSServerModel
- ✅ Router = component with RouterModel
- ✅ Route = component with RouteModel
- ✅ LetsEncryptUpdater = component with LetsEncryptModel
- ✅ ServiceWorker = component with ServiceWorkerModel
- ✅ PWAManifest = component with PWAManifestModel
- ✅ Each component: empty constructor, init(scenario), toScenario()

### **My Analysis**

**Key Learnings from Web4Articles/ONCE/0.3.1.1**:

1. **ONCE = Kernel, not Server**
   ```typescript
   // ONCE loads components from IORs
   await once.loadComponent(httpServerIOR, httpServerScenario);
   await once.loadComponent(wsServerIOR, wsServerScenario);
   ```

2. **Component Pattern**
   ```typescript
   class DefaultHttpServer {
     private data: HttpServerModel;  // ✅ Own model
     private router: Router;         // ✅ Component instance (not in model)
     
     constructor() {
       this.data = { uuid: '', state: 'stopped', ... };
     }
     
     init(scenario: Scenario): this {
       if (scenario.model) {
         Object.assign(this.data, scenario.model);
       }
       return this;
     }
   }
   ```

3. **Model vs Instances**
   - Model = serializable data only (`this.data`)
   - Instances = behavior (`this.router`, `this.certUpdater`)

4. **One file one type** - strictly enforced everywhere

**Required Changes**:
1. ✅ Create separate models: `RouterModel`, `RouteModel`, `HTTPSServerModel`, etc.
2. ✅ Each class as own component (will splinter out later)
3. ✅ LetsEncryptUpdater: `init(scenario)` not `init(domain, email, certsDir)`
4. ✅ Component instances belong to class, not model
5. ✅ ONCE loads these as components, not instantiates directly

import * as https from 'https';
import * as fs from 'fs';
import { HTTPSServerModel } from '../layer3/HTTPSServerModel.interface.js';
import { Scenario } from '../../../../../Scenario/0.3.0.2/dist/ts/Scenario.js';
import { IOR } from '../../../../../IOR/0.3.0.3/dist/index.js';
import { DefaultRouter } from './DefaultRouter.js';
import { DefaultLetsEncrypt } from './DefaultLetsEncrypt.js';

/**
 * HTTPS Server Component
 * ✅ Own model, empty constructor, scenario init
 * ✅ Component instances (router, certUpdater) belong here, not in model
 */
export class DefaultHTTPSServer {
  private data!: HTTPSServerModel;  // ✅ Definite assignment
  
  // ✅ Component instances (behavior, not data)
  private server?: https.Server;
  private router?: DefaultRouter;
  private certUpdater?: DefaultLetsEncrypt;
  
  /**
   * Empty constructor (Web4 pattern)
   */
  constructor() {
    // Empty - init in init()
  }
  
  /**
   * Initialize from scenario (Web4 pattern)
   */
  init(scenario: Scenario<HTTPSServerModel>): this {
    if (!this.data) {
      // Create minimal model
      this.data = {
        uuid: crypto.randomUUID(),
        name: 'HTTPS Server',
        state: 'stopped',
        port: 443,
        host: '0.0.0.0',
        certPath: '',
        keyPath: '',
        connections: 0,
        requestsServed: 0,
        createdAt: new Date().toISOString(),
        updatedAt: new Date().toISOString()
      };
    }
    
    // ✅ Inject scenario data
    if (scenario.model) {
      Object.assign(this.data, scenario.model);
    }
    
    return this;
  }
  
  /**
   * Start server (async operation)
   */
  async start(): Promise<void> {
    this.data.state = 'starting';
    
    // Load TLS certificates
    const options = {
      key: fs.readFileSync(this.data.certPath),
      cert: fs.readFileSync(this.data.keyPath)
    };
    
    // Create HTTPS server
    this.server = https.createServer(options, (req, res) => {
      this.handleRequest(req, res);
    });
    
    return new Promise((resolve, reject) => {
      this.server!.listen(this.data.port, () => {
        this.data.state = 'running';
        this.data.startedAt = new Date().toISOString();
        console.log(`🔒 HTTPS server running on port ${this.data.port}`);
        resolve();
      });
      
      this.server!.on('error', reject);
    });
  }
  
  /**
   * Set router component (composition)
   */
  setRouter(router: DefaultRouter): this {
    this.router = router;
    this.data.routerIOR = {
      uuid: router.getUUID(),
      component: 'Router',
      version: '0.3.21.2'
    };
    return this;
  }
  
  /**
   * Set certificate updater component (composition)
   */
  setCertUpdater(updater: DefaultLetsEncrypt): this {
    this.certUpdater = updater;
    this.data.certUpdaterIOR = {
      uuid: updater.getUUID(),
      component: 'LetsEncrypt',
      version: '0.3.21.2'
    };
    return this;
  }
  
  /**
   * Hot reload certificates (called by LetsEncrypt)
   */
  reloadCertificates(): void {
    if (!this.server) return;
    
    const newOptions = {
      key: fs.readFileSync(this.data.certPath),
      cert: fs.readFileSync(this.data.keyPath)
    };
    
    // ✅ Zero downtime reload
    this.server.setSecureContext(newOptions);
    console.log('🔄 TLS certificates reloaded');
  }
  
  /**
   * Handle HTTP request (delegate to router)
   */
  private handleRequest(req: any, res: any): void {
    this.data.requestsServed++;
    
    if (this.router) {
      this.router.route(req, res);
    } else {
      res.writeHead(500);
      res.end('Router not initialized');
    }
  }
  
  /**
   * Save as scenario (Web4 pattern)
   */
  toScenario(): Scenario<HTTPSServerModel> {
    return new Scenario().init({
      ior: {
        uuid: this.data.uuid,
        component: 'HTTPSServer',
        version: '0.3.21.2'
      },
      owner: '', // Generated by ONCE kernel
      model: { ...this.data }
    });
  }
  
  // ✅ JavaBean getters
  getUUID(): string {
    return this.data.uuid;
  }
  
  getPort(): number {
    return this.data.port;
  }
  
  getState(): string {
    return this.data.state;
  }
}

    this.routes.push(route);
    this.data.routeIORs.push({
      uuid: route.getUUID(),
      component: 'Route',
      version: '0.3.21.2'
    });
    this.data.routesRegistered++;
    
    return this;
  }
  
  /**
   * Route HTTP request to matching route
   */
  route(req: any, res: any): void {
    this.data.requestsRouted++;
    
    const url = new URL(req.url || '/', `http://${req.headers.host}`);
    const method = req.method || 'GET';
    
    for (const route of this.routes) {
      const match = route.match(url.pathname, method);
      if (match.matched) {
        route.handle(req, res, match.params || {});
        return;
      }
    }
    
    // No route matched
    res.writeHead(404);
    res.end('Not Found');
  }
  
  /**
   * Save as scenario (Web4 pattern)
   */
  toScenario(): Scenario<RouterModel> {
    return new Scenario().init({
      ior: {
        uuid: this.data.uuid,
        component: 'Router',
        version: '0.3.21.2'
      },
      owner: '',
      model: { ...this.data }
    });
  }
  
  // ✅ JavaBean getters

    return this.data.uuid;
  }

import * as fs from 'fs';
import * as path from 'path';
import * as mime from 'mime-types';
import { RouteModel, HttpMethod } from '../layer3/RouteModel.interface.js';
import { Scenario } from '../../../../../Scenario/0.3.0.2/dist/ts/Scenario.js';

/**
 * Route Component
 * ✅ Own model, empty constructor, scenario init
 */
export class DefaultRoute {
  private data!: RouteModel;  // ✅ Definite assignment
  private eamdBasePath?: string;  // Injected by router
  
  /**
   * Empty constructor (Web4 pattern)
   */
  constructor() {
    // Empty - init in init()
  }
  
  /**
   * Initialize from scenario (Web4 pattern)
   */
  init(scenario: Scenario<RouteModel>): this {
    if (!this.data) {
      this.data = {
        uuid: crypto.randomUUID(),
        path: '',
        method: 'GET',
        target: '',
        hitCount: 0,
        createdAt: new Date().toISOString(),
        updatedAt: new Date().toISOString()
      };
    }
    
    // ✅ Inject scenario data
    if (scenario.model) {
      Object.assign(this.data, scenario.model);
    }
    
    return this;
  }
  
  /**
   * Inject EAMD base path (Path Authority from router)
   */
  setEAMDBasePath(basePath: string): this {
    this.eamdBasePath = basePath;
    return this;
  }
  
  /**
   * Match request path and method
   */
  match(pathname: string, method: string): { matched: boolean; params?: Record<string, string> } {
    if (method !== this.data.method) {
      return { matched: false };
    }
    
    // Simple exact match for now (TODO: param extraction)
    if (pathname === this.data.path) {
      return { matched: true, params: {} };
    }
    
    return { matched: false };
  }
  
  /**
   * Handle request - serve file from EAMD.ucp
   */
  handle(req: any, res: any, params: Record<string, string>): void {
    this.data.hitCount++;
    this.data.lastHit = new Date().toISOString();
    
    if (!this.eamdBasePath) {
      res.writeHead(500);
      res.end('EAMD base path not set');
      return;
    }
    
    // ✅ Path Authority: construct full path
    const fullPath = path.join(this.eamdBasePath, this.data.target);
    
    // Security: prevent directory traversal
    if (!fullPath.startsWith(this.eamdBasePath)) {
      res.writeHead(403);
      res.end('Forbidden');
      return;
    }
    
    if (!fs.existsSync(fullPath)) {
      res.writeHead(404);
      res.end('Not Found');
      return;
    }
    
    const content = fs.readFileSync(fullPath);
    const contentType = mime.lookup(fullPath) || 'application/octet-stream';
    
    res.writeHead(200, { 'Content-Type': contentType });
    res.end(content);
  }
  
  /**
   * Save as scenario (Web4 pattern)
   */
  toScenario(): Scenario<RouteModel> {
    return new Scenario().init({
      ior: {
        uuid: this.data.uuid,
        component: 'Route',
        version: '0.3.21.2'
      },
      owner: '',
      model: { ...this.data }
    });
  }
  
  // ✅ JavaBean getters
  getUUID(): string {
    return this.data.uuid;
  }
  
  getPath(): string {
    return this.data.path;
  }
  
  getMethod(): HttpMethod {
    return this.data.method;
  }
}

        updatedAt: new Date().toISOString()
      };
    }
    
    // ✅ Inject scenario data
    if (scenario.model) {
      Object.assign(this.data, scenario.model);
    }
    
    return this;
  }
  
  /**
   * Set HTTPS server (composition)
   */
  setHTTPSServer(server: DefaultHTTPSServer): this {
    this.httpsServer = server;
    return this;
  }
  
  /**
   * Start auto-renewal
   */
  startAutoRenewal(): void {
    this.data.autoRenewalActive = true;
    console.log('🔐 Starting Let\'s Encrypt auto-renewal');
    
    // Initial check
    this.checkAndRenew();
    
    // Schedule periodic checks
    this.intervalId = setInterval(() => {
      this.checkAndRenew();
    }, this.data.checkIntervalMs);
  }
  
  /**
   * Stop auto-renewal
   */
  stopAutoRenewal(): void {
    if (this.intervalId) {
      clearInterval(this.intervalId);
      this.intervalId = undefined;
      this.data.autoRenewalActive = false;
    }
  }
  
  /**
   * Check and renew certificate if needed
   */
  private async checkAndRenew(): Promise<void> {
    this.data.lastCheckAt = new Date().toISOString();
    this.data.nextCheckAt = new Date(Date.now() + this.data.checkIntervalMs).toISOString();
    
    try {
      const certPath = path.join(this.data.certsDir, 'fullchain.pem');
      
      if (!fs.existsSync(certPath)) {
        console.log('📜 No certificate found, requesting new one...');
        await this.requestNewCertificate();
        return;
      }
      
      const daysUntilExpiry = this.getDaysUntilExpiry(certPath);
      this.data.daysUntilExpiry = daysUntilExpiry;
      console.log(`📅 Certificate expires in ${daysUntilExpiry} days`);
      
      if (daysUntilExpiry <= this.data.renewThresholdDays) {
        console.log('🔄 Renewing certificate...');
        await this.renewCertificate();
        
        // ✅ Hot reload (zero downtime)
        if (this.httpsServer) {
          this.httpsServer.reloadCertificates();
        }
      }
    } catch (error) {
      console.error('❌ Certificate check failed:', error);
    }
  }
  
  /**
   * Get days until certificate expiry
   */
  private getDaysUntilExpiry(certPath: string): number {
    const cert = fs.readFileSync(certPath, 'utf8');
    const output = execSync(
      `openssl x509 -enddate -noout`,
      { input: cert, encoding: 'utf8' }
    );
    
    const match = output.match(/notAfter=(.+)/);
    if (!match) throw new Error('Cannot parse expiry date');
    
    const expiryDate = new Date(match[1]);
    const diffMs = expiryDate.getTime() - Date.now();
    return Math.floor(diffMs / (1000 * 60 * 60 * 24));
  }
  
  /**
   * Request new certificate from Let's Encrypt
   */
  private async requestNewCertificate(): Promise<void> {
    const cmd = `certbot certonly --standalone \
      --non-interactive \
      --agree-tos \
      --email ${this.data.email} \
      --domains ${this.data.domain} \
      --cert-path ${this.data.certsDir}`;
    
    execSync(cmd, { stdio: 'inherit' });
    console.log('✅ New certificate obtained');
  }
  
  /**
   * Renew existing certificate
   */
  private async renewCertificate(): Promise<void> {
    const cmd = `certbot renew --cert-name ${this.data.domain}`;
    execSync(cmd, { stdio: 'inherit' });
    console.log('✅ Certificate renewed');
  }
  
  /**
   * Save as scenario (Web4 pattern)
   */
  toScenario(): Scenario<LetsEncryptModel> {
    return new Scenario().init({
      ior: {
        uuid: this.data.uuid,
        component: 'LetsEncrypt',
        version: '0.3.21.2'
      },
      owner: '',
      model: { ...this.data }
    });
  }
  
  // ✅ JavaBean getters
  getUUID(): string {
    return this.data.uuid;
  }
  
  getDomain(): string {
    return this.data.domain;
  }
  
  isAutoRenewalActive(): boolean {
    return this.data.autoRenewalActive;
  }
}
