**Code Changes**: ✅ Fix hostname detection:
```typescript
// Location: ServerHierarchyManager constructor
constructor() {
  this.serverModel = {
    // ... initial model ...
  } as ONCEServerModel;
  
  // ✅ ADD: Call detection immediately
  this.detectAndSetEnvironment();
}

// ✅ FIX: detectAndSetEnvironment() method
private detectAndSetEnvironment(): void {
  // Use os.hostname() directly (Layer 1 will improve this in Iteration 2)
  const fullHostname = os.hostname(); // e.g., "McDonges-3.fritz.box"
  
  if (fullHostname && fullHostname !== 'localhost') {
    this.serverModel.host = fullHostname;
    
    // Extract short hostname and domain
    if (fullHostname.includes('.')) {
      const parts = fullHostname.split('.');
      this.serverModel.hostname = parts[0]; // "McDonges-3"
      this.serverModel.domain = parts.slice(1).join('.'); // "fritz.box"
    } else {
      this.serverModel.hostname = fullHostname;
      this.serverModel.domain = 'local.once'; // Fallback for non-FQDN
    }
    
    // Detect IP address
    const interfaces = os.networkInterfaces();
    for (const name of Object.keys(interfaces)) {
      for (const iface of interfaces[name] || []) {
        if (iface.family === 'IPv4' && !iface.internal) {
          this.serverModel.ip = iface.address;
          break;
        }
      }
      if (this.serverModel.ip !== '127.0.0.1') break;
    }
  }
  // If detection fails, keep fallback values from constructor
}
```

// ✅ ADD: Proper server readiness check
const waitForServer = async (port: number, maxWait: number = 5000): Promise<void> => {
  const start = Date.now();
  while (Date.now() - start < maxWait) {
    try {
      const res = await fetch(`http://localhost:${port}/health`);
      if (res.ok) return;
    } catch (e) {
      // Server not ready yet
    }
    await new Promise(resolve => setTimeout(resolve, 100));
  }

};
