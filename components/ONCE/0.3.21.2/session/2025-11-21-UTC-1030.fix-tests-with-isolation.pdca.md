**Step 1: Create test setup script**

```typescript
import * as fs from 'fs';
import * as path from 'path';
import { fileURLToPath } from 'url';

export async function setup() {
  const testFile = fileURLToPath(import.meta.url);
  const componentRoot = path.resolve(testFile, '../../../..');
  const testDataDir = path.join(componentRoot, 'test/data');
  const testComponentDir = path.join(testDataDir, 'components/ONCE/0.3.21.2');
  
  // Clean test/data
  if (fs.existsSync(testDataDir)) {
    fs.rmSync(testDataDir, { recursive: true, force: true });
  }
  
  // Copy component (exclude test/ and node_modules/)
  fs.mkdirSync(testComponentDir, { recursive: true });
  await copyComponent(componentRoot, testComponentDir);
  
  // Create scenarios directory
  fs.mkdirSync(path.join(testDataDir, 'scenarios'), { recursive: true });
  
  console.log(`✅ Test isolation prepared: ${testDataDir}`);
}

async function copyComponent(src: string, dest: string) {
  const entries = fs.readdirSync(src, { withFileTypes: true });
  for (const entry of entries) {
    if (entry.name === 'test' || entry.name === 'node_modules') continue;
    
    const srcPath = path.join(src, entry.name);
    const destPath = path.join(dest, entry.name);
    
    if (entry.isDirectory()) {
      fs.mkdirSync(destPath, { recursive: true });
      await copyComponent(srcPath, destPath);
    } else {
      fs.copyFileSync(srcPath, destPath);
    }
  }
}
```
