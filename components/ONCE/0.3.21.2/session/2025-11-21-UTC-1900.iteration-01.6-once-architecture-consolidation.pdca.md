```typescript
/**
 * Compute full IOR string from IOR attributes
 * Format: ior:protocol://host:port/component/version/uuid
 */
export function computeIorString(ior: Scenario<any>['ior']): string {
  // ✅ Return cached if available
  if (ior.iorString) return ior.iorString;
  
  // ✅ Use defaults if network attributes missing
  const protocol = ior.protocol || 'https';
  const host = ior.host || 'localhost';
  const port = ior.port || (protocol.includes('https') ? 443 : 80);
  const path = ior.path || `/${ior.component}/${ior.version}/${ior.uuid}`;
  
  // ✅ Build IOR string
  return `ior:${protocol}://${host}:${port}${path}`;
}

/**
 * Parse IOR string into IOR attributes
 * Example: 'ior:https://host:8080/ONCE/0.3.21.2/abc-123' → IOR object
 */
export function parseIorString(iorString: string): Partial<Scenario<any>['ior']> {
  // Remove 'ior:' prefix
  const urlString = iorString.replace(/^ior:/, '');
  
  try {
    const url = new URL(urlString);
    const pathParts = url.pathname.split('/').filter(p => p);
    
    return {
      protocol: url.protocol.replace(':', ''),
      host: url.hostname,
      port: parseInt(url.port) || (url.protocol === 'https:' ? 443 : 80),
      path: url.pathname,
      component: pathParts[0] || '',
      version: pathParts[1] || '',
      uuid: pathParts[2] || '',
      iorString: iorString
    };
  } catch (error) {
    throw new Error(`Invalid IOR string: ${iorString}`);
  }
}

/**
 * Update IOR with network location from current context
 * Call this when creating scenarios in server environment
 */
export function enrichIorWithLocation(
  ior: Scenario<any>['ior'], 
  location: { host: string; port: number; protocol?: string }
): Scenario<any>['ior'] {
  return {
    ...ior,
    protocol: location.protocol || ior.protocol || 'https',
    host: location.host,
    port: location.port,
    path: `/${ior.component}/${ior.version}/${ior.uuid}`,
    iorString: undefined  // Will be recomputed
  };
}
```

**Web4 Protocol-Less Pattern**:

```typescript
// ✅ RIGHT: Just send scenarios (no protocol wrapper)
const scenario = await component.toScenario();
ws.send(JSON.stringify(scenario));  // ✅ Just scenario!

// ✅ RIGHT: Receive and replicate
```

- Web4 is **object-oriented**, not **message-oriented**
- Send **object state** (scenarios), not **protocol messages**
- Objects **replicate** with same UUID (clones), not **communicate** via messages
- Call **object methods**, not **handle messages**

```typescript
// src/ts/layer2/LoaderRegistry.ts (NEW)
/**
 * LoaderRegistry - Manages protocol loaders
 * Registered in ONCE kernel
 */

export class LoaderRegistry {
  private loaders: Map<string, Loader>;
  
  constructor() {
    this.loaders = new Map();
  }
  
  /**
   * Register protocol loader
   */
  registerLoader(loader: Loader): void {
    this.loaders.set(loader.protocol, loader);
    console.log(`✅ Loader registered: ${loader.protocol}`);
  }
  
  /**
   * Load object from IOR using appropriate loader(s)
   */
  async load(ior: IOR): Promise<Scenario<Model>> {
    // Parse IOR protocols
    const protocols = this.parseProtocols(ior);
    
    // Build loader chain (reverse order - innermost first)
    let scenario: Scenario<Model> | null = null;
    
    for (const protocol of protocols.reverse()) {
      const loader = this.loaders.get(protocol);
      
      if (!loader) {
        throw new Error(`No loader registered for protocol: ${protocol}`);
      }
      
      // Execute loader
      scenario = await loader.load(ior);
    }
    
    if (!scenario) {
      throw new Error(`Failed to load IOR: ${ior.url}`);
    }
    
    return scenario;
  }
  
  private parseProtocols(ior: IOR): string[] {
    // ior:https:ssl:udp://... → ['https', 'ssl', 'udp']
    const match = ior.url.match(/^ior:([^:]+(?::[^:]+)*):\/\//);
    
    if (!match) {
      return ['https'];  // Default
    }
    
    return match[1].split(':');
  }
}
```

#### **Method Invocation from IOR**

```typescript
// src/ts/layer2/DefaultONCE.ts (UPDATED)

export class DefaultONCE implements ONCE {
  private loaderRegistry: LoaderRegistry;
  
  constructor() {
    super();
    this.loaderRegistry = new LoaderRegistry();
    
    // Register default loaders
    this.loaderRegistry.registerLoader(new HTTPSLoader());
    this.loaderRegistry.registerLoader(new WSSLoader());
    this.loaderRegistry.registerLoader(new SSLLoader());
  }
  
  /**
   * Invoke method on object via IOR
   * Web4 Pattern: IOR → load object → parse method → invoke
   */
  async invokeMethodViaIOR(iorUrl: string): Promise<any> {
    // 1. Parse IOR
    const ior = this.parseIOR(iorUrl);
    
    // 2. Load object scenario via protocol loaders
    const scenario = await this.loaderRegistry.load(ior);
    
    // 3. Create object instance from scenario
    const component = await this.createComponentFromScenario(scenario);
    
    // 4. Parse method name from IOR query
    const methodName = this.parseMethodName(ior);
    
    // 5. Parse parameters from IOR query
    const params = this.parseParameters(ior);
    
    // 6. Update scenario.model with parameters (if any)
    if (params) {
      Object.assign(component.model, params);
    }
    
    // 7. Invoke method
    if (typeof (component as any)[methodName] === 'function') {
      return await (component as any)[methodName]();
    } else {
      throw new Error(`Method not found: ${methodName} on ${scenario.ior.component}`);
    }
  }
  
  /**
   * Parse IOR from URL string
   */
  private parseIOR(iorUrl: string): IOR {
    // ior:https://domain/Component/version/uuid?method&params
    const urlObj = new URL(iorUrl.replace(/^ior:/, 'http:'));  // Parse as URL
    
    const pathParts = urlObj.pathname.split('/').filter(p => p);
    
    return {
      protocol: 'ior',
      protocols: this.parseProtocols(iorUrl),
      url: iorUrl,
      domain: urlObj.hostname,
      component: pathParts[0] || '',
      version: pathParts[1] || '',
      uuid: pathParts[2] || '',
      query: urlObj.search
    };
  }
  
  /**
   * Parse method name from IOR query
   */
  private parseMethodName(ior: IOR): string {
    // ?method&param=value → 'method'
    const params = new URLSearchParams(ior.query);
    const firstKey = Array.from(params.keys())[0];
    
    // First parameter without value = method name
    if (firstKey && !params.get(firstKey)) {
      return firstKey;
    }
    
    return 'toScenario';  // Default method
  }
  
  /**
   * Parse parameters from IOR query
   */
  private parseParameters(ior: IOR): Record<string, any> | null {
    const params = new URLSearchParams(ior.query);
    const result: Record<string, any> = {};
    
    for (const [key, value] of params.entries()) {
      if (value) {  // Has value = parameter
        result[key] = value;
      }
    }
    
    return Object.keys(result).length > 0 ? result : null;
  }
  
  /**
   * Create component instance from scenario
   */
  private async createComponentFromScenario(scenario: Scenario<Model>): Promise<Component> {
    const ComponentClass = this.resolveComponent(scenario.ior.component);
    const instance = new ComponentClass();
    instance.init(scenario);
    return instance;
  }
}
```
