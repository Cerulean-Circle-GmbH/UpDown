/**
 * HTTP Method Enum
 * ✅ Static readonly for type safety
 * ✅ One file one type
 */
export class HttpMethod {
  static readonly GET = 'GET' as const;
  static readonly POST = 'POST' as const;
  static readonly PUT = 'PUT' as const;
  static readonly DELETE = 'DELETE' as const;
  static readonly PATCH = 'PATCH' as const;
  static readonly OPTIONS = 'OPTIONS' as const;
  static readonly HEAD = 'HEAD' as const;
  
  /**
   * All methods as array
   */
  static readonly ALL = [
    HttpMethod.GET,
    HttpMethod.POST,
    HttpMethod.PUT,
    HttpMethod.DELETE,
    HttpMethod.PATCH,
    HttpMethod.OPTIONS,
    HttpMethod.HEAD
  ] as const;
  
  /**
   * Type for method values
   */
  static type(): 'GET' | 'POST' | 'PUT' | 'DELETE' | 'PATCH' | 'OPTIONS' | 'HEAD' {
    return 'GET'; // Type helper
  }
  
  /**
   * Validate method string
   */
  static isValid(method: string): boolean {
    return HttpMethod.ALL.includes(method as any);
  }
}
