## 📝 KEY TEST CODE

### Test23_ImageDropE2E.ts Skeleton

```typescript
/**
 * Test23_ImageDropE2E - End-to-end image drop and persistence
 * 
 * Validates complete workflow:
 * 1. Drop image in browser
 * 2. Component + Unit + Artefact created
 * 3. Scenario persisted to server
 * 4. Scenario can be reloaded
 * 
 * @pdca 2025-12-11-UTC-1200.e2e-image-drop-persistence.pdca.md
 */

import { ONCETestCase } from './ONCETestCase.js';
import { chromium, Browser, Page } from 'playwright';
import * as fs from 'fs';
import * as path from 'path';

export class Test23_ImageDropE2E extends ONCETestCase {
  
  readonly name = 'Test23_ImageDropE2E';
  readonly description = 'E2E: Drop image → Unit + Artefact → Persist → Reload';
  
  private browser: Browser | null = null;
  private page: Page | null = null;
  private serverProcess: any = null;
  private testImagePath: string = '';
  private savedUuid: string = '';
  
  // ═══════════════════════════════════════════════════════════════
  // Test Lifecycle
  // ═══════════════════════════════════════════════════════════════
  
  async setup(): Promise<void> {
    // 1. Start ONCE server
    this.serverProcess = await this.startOnceServer();
    
    // 2. Launch browser
    this.browser = await chromium.launch({ headless: false });
    this.page = await this.browser.newPage();
    
    // 3. Create test image file
    this.testImagePath = await this.createTestImage();
  }
  
  async teardown(): Promise<void> {
    // Clean up browser
    if (this.page) await this.page.close();
    if (this.browser) await this.browser.close();
    
    // Clean up server
    if (this.serverProcess) await this.stopOnceServer();
    
    // Clean up test files
    if (this.testImagePath) fs.unlinkSync(this.testImagePath);
    
    // Clean up saved scenarios
    if (this.savedUuid) await this.cleanupScenarios(this.savedUuid);
  }
  
  // ═══════════════════════════════════════════════════════════════
  // Test Execution
  // ═══════════════════════════════════════════════════════════════
  
  async execute(): Promise<void> {
    await this.setup();
    
    try {
      // Phase 1: Navigate to once.html
      await this.testNavigateToOnce();
      
      // Phase 2: Drop image
      await this.testDropImage();
      
      // Phase 3: Verify component created
      await this.testVerifyComponent();
      
      // Phase 4: Verify Unit + Artefact
      await this.testVerifyScenario();
      
      // Phase 5: Persist to server
      await this.testPersistToServer();
      
      // Phase 6: Reload and verify
      await this.testReloadScenario();
      
      this.logSuccess('All E2E tests passed!');
      
    } finally {
      await this.teardown();
    }
  }
  
  // ═══════════════════════════════════════════════════════════════
  // Test Phases
  // ═══════════════════════════════════════════════════════════════
  
  private async testNavigateToOnce(): Promise<void> {
    this.logStep('E2E.1: Navigate to once.html');
    await this.page!.goto(`https://localhost:${this.httpsPort}/once.html`);
    await this.page!.waitForSelector('once-default-view');
    this.logSuccess('Navigated to once.html');
  }
  
  private async testDropImage(): Promise<void> {
    this.logStep('E2E.2: Drop image file');
    
    // Read test image as buffer
    const imageBuffer = fs.readFileSync(this.testImagePath);
    
    // Simulate drop via page.evaluate with DataTransfer
    await this.page!.evaluate(async (imageData: number[]) => {
      const view = document.querySelector('once-default-view');
      const uint8Array = new Uint8Array(imageData);
      const blob = new Blob([uint8Array], { type: 'image/png' });
      const file = new File([blob], 'test-drop.png', { type: 'image/png' });
      
      // Create DataTransfer with file
      const dataTransfer = new DataTransfer();
      dataTransfer.items.add(file);
      
      // Dispatch drop event
      const dropEvent = new DragEvent('drop', {
        bubbles: true,
        cancelable: true,
        dataTransfer
      });
      
      view!.dispatchEvent(dropEvent);
    }, Array.from(imageBuffer));
    
    // Wait for image view to appear
    await this.page!.waitForSelector('image-default-view', { timeout: 5000 });
    this.logSuccess('Image dropped and component created');
  }
  
  private async testVerifyComponent(): Promise<void> {
    this.logStep('E2E.3: Verify component');
    
    const hasImage = await this.page!.evaluate(() => {
      const imageView = document.querySelector('image-default-view');
      const img = imageView?.shadowRoot?.querySelector('img');
      return img !== null && img.src.startsWith('blob:');
    });
    
    this.assert(hasImage, 'Image element has blob: src');
    this.logSuccess('Component verified');
  }
  
  private async testVerifyScenario(): Promise<void> {
    this.logStep('E2E.4: Verify Unit + Artefact');
    
    const result = await this.page!.evaluate(async () => {
      const imageView = document.querySelector('image-default-view') as any;
      const component = imageView?.component;
      
      if (!component) return { hasComponent: false };
      
      const unit = component.unit;
      
      // ✅ Web4 Pattern: Use TypeRegistry instead of window globals
      const TypeRegistry = await import('/dist/ts/layer3/TypeRegistry.js');
      const DefaultArtefact = TypeRegistry.default.lookup('DefaultArtefact');
      
      const artefact = component.controller.relatedObjectLookupFirst(DefaultArtefact);
      
      return {
        hasComponent: true,
        hasUnit: unit !== null,
        hasArtefact: artefact !== null,
        artefactHasHash: artefact?.model?.contentHash?.length === 64,
        unitLinksArtefact: unit?.model?.artefactUuid === artefact?.model?.uuid
      };
    });
    
    this.assert(result.hasComponent, 'Component accessible');
    this.assert(result.hasUnit, 'Unit created');
    this.assert(result.hasArtefact, 'Artefact created');
    this.assert(result.artefactHasHash, 'Artefact has SHA-256 hash');
    this.assert(result.unitLinksArtefact, 'Unit links to Artefact');
    
    this.logSuccess('Unit + Artefact verified');
  }
  
  private async testPersistToServer(): Promise<void> {
    this.logStep('E2E.5: Persist to IndexedDB + sync to server');
    
    this.savedUuid = await this.page!.evaluate(async () => {
      const imageView = document.querySelector('image-default-view') as any;
      const component = imageView?.component;
      
      // ✅ Web4 Pattern: Use BrowserScenarioStorage
      // Ensure storage is set on component (BrowserOnce should do this)
      if (!component.storage) {
        const { BrowserScenarioStorage } = await import('/dist/ts/layer2/BrowserScenarioStorage.js');
        const storage = new BrowserScenarioStorage();
        await storage.init({ uuid: 'test-storage', projectRoot: '/' });
        component.storageSet(storage);
      }
      
      // Hibernate stores to IndexedDB and creates symlink indices
      await component.hibernate([
        'type/DefaultImage',
        'domain/local'
      ]);
      
      return component.unit?.model?.uuid || '';
    });
    
    // Verify in IndexedDB via browser evaluate
    const inIndexedDB = await this.page!.evaluate(async (uuid: string) => {
      const { BrowserScenarioStorage } = await import('/dist/ts/layer2/BrowserScenarioStorage.js');
      const storage = new BrowserScenarioStorage();
      await storage.init({ uuid: 'verify-storage', projectRoot: '/' });
      const scenario = await storage.scenarioLoad(uuid);
      return scenario !== null;
    }, this.savedUuid);
    
    this.assert(inIndexedDB, 'Scenario in IndexedDB');
    
    // Wait for sync to server (if server is available)
    await this.page!.waitForTimeout(1000);
    
    this.logSuccess(`Scenario persisted: ${this.savedUuid}`);
  }
  
  private async testReloadScenario(): Promise<void> {
    this.logStep('E2E.6: Reload from scenario');
    
    // Refresh page
    await this.page!.reload();
    await this.page!.waitForSelector('once-default-view');
    
    // Load scenario via TypeRegistry + BrowserScenarioStorage
    const reloaded = await this.page!.evaluate(async (uuid: string) => {
      // ✅ Web4 Pattern: Use TypeRegistry for component lookup
      const { TypeRegistry } = await import('/dist/ts/layer2/TypeRegistry.js');
      const DefaultImageClass = TypeRegistry.instance.classFromName('DefaultImage');
      
      if (!DefaultImageClass) {
        return { loaded: false, error: 'DefaultImage not in TypeRegistry' };
      }
      
      // Create component and set up storage
      const { BrowserScenarioStorage } = await import('/dist/ts/layer2/BrowserScenarioStorage.js');
      const storage = new BrowserScenarioStorage();
      await storage.init({ uuid: 'restore-storage', projectRoot: '/' });
      
      const imageComponent = new DefaultImageClass();
      imageComponent.storageSet(storage);
      
      // Restore from IndexedDB
      await imageComponent.restore(uuid);
      
      return {
        loaded: true,
        hasModel: imageComponent.model !== null,
        modelName: imageComponent.model?.name || '',
        hasUnit: imageComponent.unit !== null
      };
    }, this.savedUuid);
    
    this.assert(reloaded.loaded, 'Scenario loaded');
    this.assert(reloaded.hasModel, 'Model restored');
    this.assert(reloaded.hasUnit, 'Unit restored');
    
    this.logSuccess('Scenario reloaded successfully');
  }
  
  // ═══════════════════════════════════════════════════════════════
  // Helpers
  // ═══════════════════════════════════════════════════════════════
  
  private async createTestImage(): Promise<string> {
    // Create minimal 1x1 PNG
    const png = Buffer.from([
      0x89, 0x50, 0x4E, 0x47, 0x0D, 0x0A, 0x1A, 0x0A, // PNG signature
      0x00, 0x00, 0x00, 0x0D, 0x49, 0x48, 0x44, 0x52, // IHDR
      0x00, 0x00, 0x00, 0x01, 0x00, 0x00, 0x00, 0x01, // 1x1
      0x08, 0x02, 0x00, 0x00, 0x00, 0x90, 0x77, 0x53, 0xDE,
      0x00, 0x00, 0x00, 0x0C, 0x49, 0x44, 0x41, 0x54, // IDAT
      0x08, 0xD7, 0x63, 0xF8, 0x00, 0x00, 0x00, 0x01,
      0x00, 0x01, 0xFD, 0x16, 0x60, 0x11,
      0x00, 0x00, 0x00, 0x00, 0x49, 0x45, 0x4E, 0x44, // IEND
      0xAE, 0x42, 0x60, 0x82
    ]);
    
    const testPath = path.join(this.testDir, 'test-e2e-image.png');
    fs.writeFileSync(testPath, png);
    return testPath;
  }
  
  private async waitForFile(filePath: string, timeoutMs: number): Promise<void> {
    const start = Date.now();
    while (Date.now() - start < timeoutMs) {
      if (fs.existsSync(filePath)) return;
      await new Promise(r => setTimeout(r, 100));
    }
    throw new Error(`Timeout waiting for file: ${filePath}`);
  }
  
  private async cleanupScenarios(uuid: string): Promise<void> {
    // Delete unit and artefact scenario files
    // Implementation depends on actual paths
  }
}

export async function runTest(): Promise<void> {
  const test = new Test23_ImageDropE2E();
  await test.execute();
}
```

## ✅ ARCHITECTURE CLARIFICATIONS (Blockers Resolved)

### 1. Storage Implementation — RESOLVED

**Key Implementation: `BrowserScenarioStorage.ts`** (Layer 2)
- Implements `PersistenceManager` interface
- Uses **IndexedDB** with 4 object stores:
  - `scenarios-index`: Primary storage by UUID
  - `scenarios-type`: Index by component/version
  - `scenarios-domain`: Index by domain/component/version
  - `scenarios-capability`: Index by capability/value

**Component Integration:**
```typescript
// Components get storage via storageSet()
const storage = new BrowserScenarioStorage().init({...});
component.storageSet(storage);

// Then hibernate/restore work
await component.hibernate(['type/Image', 'domain/local']);
await component.restore(uuid);
```

### 3. TypeRegistry vs Global Namespace — RESOLVED

**WRONG**: Components on `window` global
```typescript
// ❌ Anti-pattern
const image = new (window as any).DefaultImage();
```

**CORRECT**: Components from TypeRegistry
```typescript
// ✅ Web4 Pattern - TypeRegistry lookup
const ImageClass = TypeRegistry.lookup('DefaultImage');
const image = new ImageClass();

// Or via IOR
const component = await IOR.create('ior:esm:/ONCE/0.3.22.1/DefaultImage');
```

The TypeRegistry:
- Stores all dynamically loaded component classes
- Provides lookup by name or interface
- Avoids global namespace pollution
- Supports lazy loading via IOR
