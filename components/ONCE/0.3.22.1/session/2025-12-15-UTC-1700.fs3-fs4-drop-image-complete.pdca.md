```typescript
// New interface for drop events
export interface FileDropDetail {
  file: File;
  mimetype: string;
  filename: string;
  size: number;
  targetView: UcpView<any>;
}

// New property for opt-out
@property({ type: String, reflect: true })
dropDisabled: string = 'false';

// Static CSS for drop visual feedback
static dropStyles: CSSResultGroup = css`
  :host(.drop-active) {
    outline: 2px dashed var(--color-primary, #1976d2);
    background-color: rgba(25, 118, 210, 0.08);
  }
`;

// Event handlers added in connectedCallback
dropHandlersSetup(): void {
  this.addEventListener('dragover', this.handleDragOver);
  this.addEventListener('dragleave', this.handleDragLeave);
  this.addEventListener('drop', this.handleDrop);
  this.addEventListener('dragenter', this.handleDragEnter);
}

// Drop processing dispatches 'file-drop' CustomEvent
fileDropProcess(file: File): void {
  this.dispatchEvent(new CustomEvent<FileDropDetail>('file-drop', {
    detail: { file, mimetype, filename, size, targetView: this },
    bubbles: true, composed: true
  }));
}
```

### Drop Flow

```
User drops file on any UcpView
         â†“
handleDrop() extracts File from DataTransfer
         â†“
fileDropProcess() dispatches 'file-drop' event
         â†“
Parent component/controller listens and handles
         â†“
FileSystem.componentFromFile() creates component
         â†“
Component view added to DOM
```

### How to Use

```typescript
// In a parent component or controller
this.addEventListener('file-drop', (e: CustomEvent<FileDropDetail>) => {
  const { file, mimetype } = e.detail;
  const handler = mimetypeRegistry.handlerLookup(mimetype);
  if (handler) {
    const component = new handler.componentClass();
    await component.initFromFile(file);
    this.add(component.view);
  }
});

// To disable drops on a specific view
<read-only-view dropDisabled="true"></read-only-view>
```

### DefaultImage.ts Key Features

```typescript
export class DefaultImage extends UcpComponent<ImageModel> {
  
  // Auto-registers on module load
  static registerMimetypeHandler(): void {
    const registry = DefaultMimetypeHandlerRegistry.instance;
    registry.handlerRegister({
      pattern: 'image/*',
      componentClass: DefaultImage,
      priority: 100,
      name: 'Image',
      description: 'Handles all image file types'
    });
  }
  
  // Initialize from dropped file
  async initFromFile(file: File): Promise<void> {
    // Create underlying File component
    this.fileInstance = new DefaultFile();
    await this.fileInstance.initFromBlob(file, file.name, file.type);
    
    // Store in RelatedObjects (P24)
    this.controller.relatedObjectRegister(DefaultFile, this.fileInstance);
    
    // Compute dimensions
    const dimensions = await this.imageDimensionsCompute(file);
    
    // Update model
    this.model.width = dimensions.width;
    this.model.height = dimensions.height;
    this.model.objectUrl = this.fileInstance.objectUrl;
  }
}

// Auto-register at module load
DefaultImage.registerMimetypeHandler();
```

### Note on P7 Compliance

FileOrchestrator provides Layer 4 entry points for async operations. Currently delegates to Layer 2 async methods for backward compatibility. Full P7 compliance (Layer 2 fully synchronous) would require:
- Layer 2 methods return void with event callbacks
- FileOrchestrator handles all browser async APIs
- Event-based notification for operation completion
