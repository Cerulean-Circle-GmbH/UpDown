## Context: Unit Architecture Understanding

After re-reading the Unit README and related interfaces, I now understand:

### What is a Unit?

**Unit = Instance of a component with its scenario data**

A Unit is the **atomic executable element** within Web4 components. It represents:
- A specific instance of a component
- Links to its content (via Artefact)
- Links to metadata (via File)
- All aliases/references to access it

### Unit Architecture Diagram

```
┌─────────────────────────────────────────────────────────────────┐
│                     Scenario<UnitModel>                         │
├─────────────────────────────────────────────────────────────────┤
│ uuid: "unit-uuid-v4"                                            │
│ componentType: "File"                                           │
│ componentIor: "ior:esm:/File/0.3.22.1/unit-uuid/..."           │
│                                                                 │
│ ┌─────────────────┐    ┌─────────────────────────────────────┐ │
│ │ artefactUuid ───┼────► Artefact (content by hash)          │ │
│ │                 │    │ contentHash: "sha256:abc..."         │ │
│ │                 │    │ size: 12345                          │ │
│ │                 │    │ mimetype: "image/png"                │ │
│ └─────────────────┘    │ content: Blob                        │ │
│                        └─────────────────────────────────────┘ │
│ ┌─────────────────┐    ┌─────────────────────────────────────┐ │
│ │ fileUuid ───────┼────► File (metadata)                     │ │
│ │                 │    │ path: "/EAMD.ucp/components/..."     │ │
│ │                 │    │ filename: "styles.css"               │ │
│ │                 │    │ mimetype: "text/css"                 │ │
│ └─────────────────┘    └─────────────────────────────────────┘ │
│                                                                 │
│ references: UnitReference[]                                     │
│ ├── { linkLocation: "ior:fs://project/components/...",         │
│ │     linkTarget: "ior:unit:{uuid}", syncStatus: SYNCED }      │
│ ├── { linkLocation: "ior:url:///EAMD.ucp/components/...",      │
│ │     linkTarget: "ior:unit:{uuid}", syncStatus: SYNCED }      │
│ └── { linkLocation: "ior:blob:abc123...",    ← BlobUrl!        │
│       linkTarget: "ior:unit:{uuid}", syncStatus: RUNTIME }     │
└─────────────────────────────────────────────────────────────────┘
```

### Key Insight: BlobUrls are UnitReferences

BlobUrls are **NOT** a separate concept — they're just another `UnitReference`:

```typescript
// BlobUrl as UnitReference
{
  linkLocation: "ior:blob:https://..." ,  // The blob URL
  linkTarget: "ior:unit:{uuid}",           // Points to the Unit
  syncStatus: SyncStatus.RUNTIME           // New status: exists only at runtime
}
```

This means:
- No need for separate BlobUrlCache
- Unit.references already tracks all aliases
- Blob URL lifecycle managed by Unit lifecycle

### Revised Architecture

```
StaticFileRoute serves file
        │
        ▼

        │
        ▼

└───────────────────────────────────────┘
        │
        ▼
┌───────────────────────────────────────┐
│ 4. Store in BrowserScenarioStorage    │
│    - Scenario<UnitModel>              │
│    - Scenario<FileModel>              │
│    - Scenario<ArtefactModel> + Blob   │
└───────────────────────────────────────┘
        │
        ▼
┌───────────────────────────────────────┐
│ 5. Browser retrieves via IOR          │
│    - Cache check → find Unit by path  │
│    - Load Artefact → get Blob         │

```

```typescript
// layer2/BrowserScenarioStorage.ts

// Add new object store in init()
const artefactStore = db.createObjectStore('artefacts', { keyPath: 'hash' });

// New methods:
async artefactSave(artefact: ArtefactModel, blob: Blob): Promise<void> {
  const tx = this.db.transaction('artefacts', 'readwrite');
  const store = tx.objectStore('artefacts');
  await store.put({ ...artefact, content: blob });
}

async artefactLoad(hash: string): Promise<{ model: ArtefactModel; content: Blob } | null> {
  const tx = this.db.transaction('artefacts', 'readonly');
  const store = tx.objectStore('artefacts');
  return await store.get(hash);
}

async artefactExists(hash: string): Promise<boolean> {
  const result = await this.artefactLoad(hash);
  return result !== null;
}
```

### I.9.6-I.9.8: Unit-based IOR Caching

```typescript
// layer4/IOR.ts

private async cacheStore(result: T): Promise<void> {
  // ... existing checks ...
  
  // Create Unit scenario (links everything together)
  const unitUuid = crypto.randomUUID();
  const unit: Scenario<UnitModel> = {
    ior: { uuid: unitUuid, component: 'Unit', version: '0.3.22.1' },
    owner: 'browser-cache',
    model: {
      uuid: unitUuid,
      componentType: this.model.component || 'IOR',
      componentIor: this.toUrl(),
      artefactUuid: null,  // Set if content is Blob
      fileUuid: null,      // Set if this is a file
      createdAt: Date.now(),
      modifiedAt: Date.now(),
      storagePath: null,
      indexPath: null,
      references: [
        {
          linkLocation: `ior:url:${this.toUrl()}`,
          linkTarget: `ior:unit:${unitUuid}`,
          syncStatus: SyncStatus.SYNCED
        }
      ]
    }
  };
  
  await storage.scenarioSave(unitUuid, unit, []);
  console.log(`💾 [IOR] Cached Unit: ${unitUuid}`);
}

// Lookup by path (UnitReference.linkLocation)
private async cacheCheck(): Promise<T | undefined> {
  const storage = await this.browserStorageGet();
  if (!storage) return undefined;
  
  // Search for Unit with matching linkLocation
  const unit = await storage.unitFindByReference(`ior:url:${this.toUrl()}`);
  if (unit) {
    console.log(`💾 [IOR] Cache hit via reference: ${this.toUrl()}`);
    // Load the actual data from artefact or model
    return this.unitDataLoad(unit);
  }
  
  return undefined;
}

// Create blob: URL and register as reference
blobUrlCreate(blob: Blob): string {
  const url = URL.createObjectURL(blob);
  
  // Add to current unit's references
  if (this.currentUnit) {
    this.currentUnit.model.references.push({
      linkLocation: `ior:blob:${url}`,
      linkTarget: `ior:unit:${this.currentUnit.model.uuid}`,
      syncStatus: SyncStatus.RUNTIME
    });
  }
  
  return url;
}
```

### I.9.9-I.9.13: StaticFileRoute Unit Creation

```typescript
// layer2/StaticFileRoute.ts

private async unitCreateForFile(
  urlPath: string, 
  filePath: string, 
  content: Buffer, 
  mimeType: string
): Promise<Scenario<UnitModel>> {
  
  // 1. Create File scenario
  const fileUuid = crypto.randomUUID();
  const file: Scenario<FileModel> = {
    ior: { uuid: fileUuid, component: 'File', version: '0.3.22.1' },
    owner: 'ONCE/0.3.22.1',
    model: {
      uuid: fileUuid,
      path: filePath,
      filename: path.basename(filePath),
      mimetype: mimeType,
      size: content.length,
      contentHash: this.hashCompute(content),
      createdAt: Date.now(),
      modifiedAt: Date.now(),
      content: null  // Server-side: content in Artefact, not here
    }
  };
  
  // 2. Create Artefact scenario (content-addressable)
  const hash = this.hashCompute(content);
  const artefact: Scenario<ArtefactModel> = {
    ior: { uuid: hash, component: 'Artefact', version: '0.3.22.1' },
    owner: 'ONCE/0.3.22.1',
    model: {
      uuid: hash,  // Artefact UUID = content hash!
      contentHash: hash,
      algorithm: 'SHA-256',
      size: content.length,
      unitUuid: '',  // Set below
      createdAt: Date.now(),
      mimetype: mimeType
    }
  };
  
  // 3. Create Unit scenario (links everything)
  const unitUuid = crypto.randomUUID();
  const unit: Scenario<UnitModel> = {
    ior: { uuid: unitUuid, component: 'Unit', version: '0.3.22.1' },
    owner: 'ONCE/0.3.22.1',
    model: {
      uuid: unitUuid,
      componentType: 'File',
      componentIor: `ior:esm:/File/0.3.22.1/${fileUuid}`,
      artefactUuid: hash,
      fileUuid: fileUuid,
      createdAt: Date.now(),
      modifiedAt: Date.now(),
      storagePath: null,
      indexPath: null,
      references: [
        // Filesystem path
        {
          linkLocation: `ior:fs:${filePath}`,
          linkTarget: `ior:unit:${unitUuid}`,
          syncStatus: SyncStatus.SYNCED
        },
        // URL path (/EAMD.ucp/...)
        {
          linkLocation: `ior:url:${urlPath}`,
          linkTarget: `ior:unit:${unitUuid}`,
          syncStatus: SyncStatus.SYNCED
        }
      ]
    }
  };
  
  // Update artefact with unit reference
  artefact.model.unitUuid = unitUuid;
  
  return unit;
}
```
