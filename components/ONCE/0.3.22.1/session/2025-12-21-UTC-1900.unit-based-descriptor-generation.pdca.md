async saveScenario(uuid: string, scenario: Scenario, symlinkPaths: string[]): Promise<void> {
  // Create UUID folder structure (5 levels deep)
  const folderPath = this.generateUUIDFolderPath(uuid);  // → scenarios/index/a/b/c/d/e/
  await fs.mkdir(folderPath, { recursive: true });

  // Create scenario file path
  const scenarioPath = join(folderPath, `${uuid}.scenario.json`);

  // Update scenario model with storage data
  const scenarioWithStorage = {
    ...scenario,
    model: {
      ...scenario.model,
      indexPath: scenarioPath,
      updatedAt: new Date().toISOString()
    }
  };

  // Save scenario to index
  await fs.writeFile(scenarioPath, JSON.stringify(scenarioWithStorage, null, 2), 'utf-8');

  // Create symbolic links
  for (const linkPath of symlinkPaths) {
    await this.createSymbolicLink(scenarioPath, linkPath);
  }
}

async toScenario(): Promise<Scenario<UnitModel>> {
  const componentVersion = await this.getComponentVersion();
  const componentName = await this.getComponentName();
  
  const scenario: Scenario = {
    ior: {
      uuid: this.model.uuid,
      component: componentName,
      version: componentVersion
    },
    owner: ownerData,
    model: this.model
  };

  // Create named link in current directory
  const links: string[] = [];
  if (this.model.name) {
    const linkFilename = `${this.convertNameToFilename(this.model.name)}.unit`;
    const namedLink = `${currentDir}/${linkFilename}`;
    links.push(namedLink);
  }
  
  await this.storage.saveScenario(this.model.uuid, scenario, links);
  await this.createAutomaticLinks();  // ontology and M3 type links
  
  return scenario;
}

### Key Differences: Component vs Unit Storage

| Aspect | Component Descriptor | File Unit | Folder Unit |
|--------|---------------------|-----------|-------------|
| Primary Storage | `scenarios/index/{uuid[0-4]}/` | `scenarios/index/{uuid[0-4]}/` | `scenarios/index/{uuid[0-4]}/` |
| Symlink in Component | `{Name}.component.json` | `{filename}.unit` | `°folder.unit` |
| Type Mirror | `scenarios/type/{Comp}/{Ver}/` | `scenarios/type/.../filename.unit` | `scenarios/type/.../°folder.unit` |

**Required Changes** (DRY — leverage Unit/0.3.0.5):
1. Import `DefaultUnit` from `Unit/0.3.0.5`
2. For each file: call `unit.from(filePath)` → creates `{filename}.unit` + stores in `scenarios/index/`
3. For each folder: call `unit.from(folderPath)` → creates `°folder.unit` + stores in `scenarios/index/`
4. For type mirror: call `unit.linkInto(uuid, typePath)` → creates symlink in `scenarios/type/`
5. Add `unit` attribute to descriptor: `descriptorUnit.toScenario()`
6. Update descriptor symlink to `scenarios/index/`

**NO reimplementation needed** — Unit component already does everything!

#### UD.4.1: Walk Component Tree + Call `unit from`

```typescript
async componentDescriptorUpdate(): Promise<this> {
  // ... existing code ...
  
  // Import Unit component
  const { DefaultUnit } = await import(
    `${projectRoot}/components/Unit/0.3.0.5/dist/ts/layer2/DefaultUnit.js`
  );
  
  // Walk component tree
  const walk = async (dir: string) => {
    const entries = await fs.readdir(dir, { withFileTypes: true });
    
    // Create folder unit
    const folderUnit = new DefaultUnit();
    await folderUnit.from(dir);  // Creates °folder.unit automatically!
    
    for (const entry of entries) {
      const fullPath = path.join(dir, entry.name);
      if (entry.isDirectory() && !entry.name.startsWith('.')) {
        await walk(fullPath);
      } else if (entry.isFile() && this.isUnitableFile(entry.name)) {
        // Create file unit
        const fileUnit = new DefaultUnit();
        await fileUnit.from(fullPath);  // Creates {filename}.unit symlink!
      }
    }
  };
  
  await walk(componentRoot);
}
```
