### Sub-Iteration FFM.0.2: delete() + deleteLink() (1h)

**Focus**: Add Unit-aware `delete()` and `deleteLink()` to DefaultFile

**TRON Clarification**: Two distinct operations:
1. **deleteLink()** — Removes a symlink only, original remains
2. **delete()** — RARE! Deletes ORIGINAL + ALL links + Unit → P2P network broadcast

**Tasks**:
- [ ] Add `deleteLink(): Promise<void>` — removes symlink, updates Unit references
- [ ] Add `delete(): Promise<void>` — deletes original, ALL links, Unit (P2P broadcast later)
- [ ] Distinguish: is this file a link or the original?

**Implementation**:

```typescript
// In DefaultFile.ts

/**
 * Delete a LINK only (not the original).
 * Updates Unit references to remove this link path.
 * Safe operation — original file remains.
 */
async deleteLink(): Promise<void> {
  if (!this.model.isLink) {
    throw new Error('[DefaultFile] deleteLink() called on original file, not a link');
  }
  
  // 1. Update Unit references (remove this link path)
  const unit = await this.unitGet();
  if (unit) {
    unit.model.references = unit.model.references.filter(
      function matchesPath(ref: UnitReference): boolean {
        return ref.path !== this.fullPath;
      }.bind(this)
    );
    await unit.save();
  }
  
  // 2. Delete the symlink only
  await this.persistenceManager.unlink(this.fullPath);
}

/**
 * Delete the ORIGINAL file (RARE operation!).
 * Deletes ALL links pointing to this file.
 * Deletes the Unit.
 * In P2P network: triggers broadcast (handled by infrastructure later).
 */
async delete(): Promise<void> {
  if (this.model.isLink) {
    // If this is a link, just delete the link
    await this.deleteLink();
    return;
  }
  
  // 1. Get Unit to find ALL links
  const unit = await this.unitGet();
  if (unit) {
    // 2. Delete ALL symlinks pointing to this file
    for (const ref of unit.model.references) {
      if (ref.type === 'symlink' && ref.path !== this.fullPath) {
        await this.persistenceManager.unlink(ref.path);
      }
    }
    
    // 3. Delete the Unit itself
    await unit.delete();
  }
  
  // 4. Delete the original file
  await this.persistenceManager.delete(this.fullPath);
  
  // NOTE: P2P network broadcast will be handled by infrastructure (emergent)
}
```

**CHECK Phase**:
- [ ] `tsc --noEmit` passes

- [ ] `delete()` removes original + ALL links + Unit
- [ ] Error thrown if `deleteLink()` called on original
- [ ] No new linter errors

**Success Criteria**:
- ✅ DefaultFile has `deleteLink(): Promise<void>` (safe, link-only)
- ✅ DefaultFile has `delete(): Promise<void>` (RARE, deletes everything)

```typescript
// In DefaultFile.ts

async unitGet(): Promise<Reference<DefaultUnit>> {
  // Check if Unit already exists for this file's UUID
  const unitPath = this.unitPathCompute();
  if (await this.persistenceManager.exists(unitPath)) {
    const unit = new DefaultUnit();
    await unit.initFromScenario(unitPath);
    return unit;
  }
  return null;
}

async unitCreate(): Promise<DefaultUnit> {
  const unit = new DefaultUnit();
  await unit.init();
  await unit.from(this.model.path);  // Creates Unit from this file
  return unit;
}

async unitEnsure(): Promise<DefaultUnit> {
  // First touch creates, second rediscovers
  const existing = await this.unitGet();
  if (existing) return existing;
  return await this.unitCreate();
}

async initFromPath(path: string): Promise<this> {
  await this.init();
  this.model.path = path;
  this.model.filename = path.split('/').pop() || '';
  
  // Try to rediscover existing Unit
  const unit = await this.unitGet();
  if (unit) {
    // Load model from Unit scenario
    this.model = { ...this.model, ...unit.model };
  }
  
  return this;
}

private unitPathCompute(): string {
  // §/scenarios/type/{Component}/{version}/.../{filename}.unit
  return `${this.model.path}.unit`;
}
```

### Sub-Iteration FFM.1.2: Folder delete() + list() (1h)

**Focus**: Add recursive `delete()` and `list()` to DefaultFolder

**Tasks**:
- [ ] Add `delete(recursive?: boolean): Promise<void>` — deletes children + Units
- [ ] Add `list(): Collection<FileSystemNode>` — returns children

**Implementation**:

```typescript
// In DefaultFolder.ts

async delete(recursive: boolean = false): Promise<void> {
  if (recursive) {
    // Delete all children first (and their Units!)
    const children = await this.list();
    for (const child of children) {
      if (child instanceof DefaultFolder) {
        await child.delete(true);
      } else {
        await child.delete();
      }
    }
  }
  
  // Delete this folder's Unit
  const unit = await this.unitGet();
  if (unit) {
    await unit.delete();
  }
  
  // Delete the folder itself
  await this.persistenceManager.rmdir(this.model.path);
}

list(): Collection<FileSystemNode> {
  const entries = this.persistenceManager.readdir(this.model.path);
  const result: FileSystemNode[] = [];
  
  for (const entry of entries) {
    const fullPath = `${this.model.path}/${entry}`;
    const stats = this.persistenceManager.stat(fullPath);
    
    if (stats.isDirectory) {
      const folder = new DefaultFolder();
      folder.initSync({ model: { ...folder.modelDefault(), path: fullPath, folderName: entry } });
      result.push(folder);
    } else {
      const file = new DefaultFile();
      file.initSync({ model: { ...file.modelDefault(), path: fullPath, filename: entry } });
      result.push(file);
    }
  }
  
  return result;
}
```

### Sub-Iteration FFM.2.1: Unit-Aware pathMove for File (1h) — CRITICAL

**Focus**: Implement `pathMove()` that updates Units and symlinks

**TRON Rule**: "one of the most important file operations will be the Unit aware move, that changes paths in the symlinks and the units references in their model"

**Tasks**:
- [ ] Add `pathMove(newPath: string): Promise<void>` to DefaultFile
- [ ] Update Unit.model.references[].path
- [ ] Update symlinks in §/scenarios/type/

**Implementation**:

```typescript
// In DefaultFile.ts

async pathMove(newPath: string): Promise<void> {
  const oldPath = this.fullPath;
  
  // 1. Update file's model.path
  const newDir = newPath.substring(0, newPath.lastIndexOf('/'));
  const newFilename = newPath.split('/').pop() || '';
  this.model.path = newDir;
  this.model.filename = newFilename;
  this.model.modifiedAt = Date.now();
  
  // 2. Update Unit references (if Unit exists)
  const unit = await this.unitGet();
  if (unit) {
    for (const ref of unit.model.references) {
      if (ref.path === oldPath) {
        ref.path = newPath;
      }
    }
    await unit.save();
  }
  
  // 3. Update symlinks in §/scenarios/type/
  await this.symlinkPathsUpdate(oldPath, newPath);
  
  // 4. Actually move the file
  await this.persistenceManager.rename(oldPath, newPath);
  
  await this.save();
}

private async symlinkPathsUpdate(oldPath: string, newPath: string): Promise<void> {
  // Find and update all symlinks pointing to this file
  // in §/scenarios/type/{Component}/{version}/...
  const unit = await this.unitGet();
  if (!unit) return;
  
  for (const ref of unit.model.references) {
    if (ref.type === 'symlink') {
      // Update the symlink to point to new location
      const symlinkPath = ref.path;
      await this.persistenceManager.unlink(symlinkPath);
      await this.persistenceManager.symlink(newPath, symlinkPath);
    }
  }
}
```

```typescript
async pathMove(newPath: string): Promise<void> {
  const oldPath = this.fullPath;
  
  // 1. Update folder's model
  const newDir = newPath.substring(0, newPath.lastIndexOf('/'));
  const newFolderName = newPath.split('/').pop() || '';
  this.model.path = newDir;
  this.model.folderName = newFolderName;
  this.model.modifiedAt = Date.now();
  
  // 2. Recursively update all children's paths
  for (const child of this.childrenCache.values()) {
    const childOldPath = child instanceof DefaultFolder 
      ? child.fullPath 
      : (child as DefaultFile).fullPath;
    const childNewPath = childOldPath.replace(oldPath, newPath);
    await child.pathMove(childNewPath);
  }
  
  // 3. Update this folder's Unit
  const unit = await this.unitGet();
  if (unit) {
    for (const ref of unit.model.references) {
      if (ref.path === oldPath) {
        ref.path = newPath;
      }
    }
    await unit.save();
  }
  
  // 4. Update symlinks
  await this.symlinkPathsUpdate(oldPath, newPath);
  
  // 5. Actually rename the folder
  await this.persistenceManager.rename(oldPath, newPath);
  
  await this.save();
}
```

```typescript
// In Web4TSComponent

extractableComponentDetect(filename: string, parentVersion: string): Reference<{name: string, version: string}> {
  // Default{Name}.ts → {Name}/{parentVersion}
  const match = filename.match(/^Default([A-Z][a-zA-Z0-9]+)\.ts$/);
  if (!match) return null;
  
  return {
    name: match[1],  // e.g., "File", "Folder"
    version: parentVersion  // e.g., "0.3.22.1" (inherited!)
  };
}

// In DefaultUnit

async createDualSymlinks(filePath: string, parentComponent: string, parentVersion: string): Promise<void> {
  const filename = filePath.split('/').pop() || '';
  const extractable = this.web4ts.extractableComponentDetect(filename, parentVersion);
  
  // 1. Current symlink: §/scenarios/type/ONCE/0.3.22.1/.../DefaultFile.ts.unit
  const currentSymlink = `§/scenarios/type/${parentComponent}/${parentVersion}/${this.relativePath(filePath)}.unit`;
  await this.persistenceManager.symlink(this.indexPath, currentSymlink);
  
  // 2. Future symlink (if extractable): §/scenarios/type/File/0.3.22.1/.../DefaultFile.ts.unit
  if (extractable) {
    const futureSymlink = `§/scenarios/type/${extractable.name}/${extractable.version}/${this.relativePath(filePath)}.unit`;
    await this.persistenceManager.symlink(this.indexPath, futureSymlink);
  }
}
```
