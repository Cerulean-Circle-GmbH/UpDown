**Step 1: Add helper method:**
```typescript
/**
 * Register HTML route with trailing slash variant
 * DRY: Eliminates duplicate route registration
 */
private registerHTMLRoute(
    pattern: string, 
    provider: () => Promise<string>,
    priority: number = 50,
    includeTrailingSlash: boolean = true
): void {
    const route = new HTMLRoute();
    route.model.uuid = this.idProvider.create();
    route.setPattern(pattern, HttpMethod.GET);
    route.setProvider(provider);
    route.model.priority = priority;
    this.httpRouter.registerRoute(route);
    
    // Auto-register trailing slash variant
    if (includeTrailingSlash && !pattern.endsWith('/')) {
        const trailingRoute = new HTMLRoute();
        trailingRoute.model.uuid = this.idProvider.create();
        trailingRoute.setPattern(pattern + '/', HttpMethod.GET);
        trailingRoute.setProvider(provider);
        trailingRoute.model.priority = priority;
        this.httpRouter.registerRoute(trailingRoute);
    }
}
```
