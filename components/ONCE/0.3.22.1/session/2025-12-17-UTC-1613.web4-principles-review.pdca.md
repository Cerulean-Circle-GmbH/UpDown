### **Fix 2: HTTPSLoader.ts — P6 (Empty Constructor)**

**Action:** Empty constructor, move initialization to `modelDefault()`

**Before:**
```typescript
constructor() {
    const now = new Date().toISOString();
    this.model = {
        uuid: '',
        name: 'HTTPSLoader',
        // ... 15 lines of initialization
    };
}
```

**After:**
```typescript
public model!: LoaderModel;  // Initialized by init()

constructor() {
    // Empty - P6 compliant. Use init(scenario) for initialization.
}

/**
 * Default model for HTTPSLoader
 * Called by init() when no scenario provided
 */
protected modelDefault(): LoaderModel {
    const now = new Date().toISOString();
    return {
        uuid: '',  // Set by caller or IDProvider
        name: 'HTTPSLoader',
        protocol: 'https',
        iorComponent: 'HTTPSLoader',
        iorVersion: '',
        statistics: {
            totalOperations: 0,
            successCount: 0,
            errorCount: 0,
            lastOperationAt: '',
            lastErrorAt: '',
            createdAt: now,
            updatedAt: now
        }
    };
}

/**
 * Initialize with scenario (P6 pattern)
 */
public init(scenario?: Scenario<LoaderModel>): this {
    this.model = scenario?.model ?? this.modelDefault();
    return this;
}
```
