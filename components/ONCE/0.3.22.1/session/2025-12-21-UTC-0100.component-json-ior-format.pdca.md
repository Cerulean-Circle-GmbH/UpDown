### CD.2: Update parseUnitSymlink

**File**: `src/ts/layer2/UnitDiscoveryService.ts`

```typescript
/**
 * Parse a .unit symlink to extract unit IOR
 * Returns: "ior:unit:{uuid}" format
 */
private async parseUnitSymlink(
  symlinkPath: string,
  symlinkName: string
): Promise<string | null> {
  try {
    const target = await fs.promises.readlink(symlinkPath);
    const uuidMatch = target.match(/([0-9a-f-]{36})\.scenario\.json/);
    
    if (!uuidMatch) {
      return null;
    }
    
    const uuid = uuidMatch[1];
    return `ior:unit:${uuid}`;
  } catch {
    return null;
  }
}
```

### CD.3: Expand collectManifestUnits

**File**: `src/ts/layer2/UnitDiscoveryService.ts`

```typescript
private async collectManifestUnits(
  directory: string,
  units: ManifestUnits
): Promise<void> {
  const entries = await fs.promises.readdir(directory, { withFileTypes: true });
  
  for (const entry of entries) {
    const fullPath = path.join(directory, entry.name);
    
    if (entry.isDirectory()) {
      // Skip build/cache directories
      if (['node_modules', 'dist', '.git', 'coverage'].includes(entry.name)) {
        continue;
      }
      await this.collectManifestUnits(fullPath, units);
    } else if (entry.isSymbolicLink() && entry.name.endsWith('.unit')) {
      const iorRef = await this.parseUnitSymlink(fullPath, entry.name);
      if (iorRef) {
        this.categorizeUnit(entry.name, iorRef, units);
      }
    }
  }
}

private categorizeUnit(filename: string, iorRef: string, units: ManifestUnits): void {
  const originalFile = filename.replace('.unit', '');
  
  if (originalFile.endsWith('.css')) {
    units.css.push(iorRef);
  } else if (originalFile.endsWith('.html')) {
    units.templates.push(iorRef);
  } else if (originalFile.endsWith('.ts')) {
    if (filename.includes('Test')) {
      units.tests.push(iorRef);
    } else {
      units.typescript.push(iorRef);
    }
  } else if (originalFile === 'package.json' || originalFile === 'tsconfig.json') {
    units.configuration.push(iorRef);
  } else if (originalFile.endsWith('.sh')) {
    units.lifecycle.push(iorRef);
  }
}
```
