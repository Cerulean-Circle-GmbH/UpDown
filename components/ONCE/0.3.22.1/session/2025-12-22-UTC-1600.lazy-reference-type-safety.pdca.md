## Problem Statement

### User Feedback (Quoted)

> "the issue is not that the methods are resolved or not resolved. the issue is that the types used in the methods do not guarantee that its resolved."

> "both are no solutions. typescript will narrow the types on if. so we have to check the type of children[1] in these critical methods."

### The Real Issue: Type Safety via Type Narrowing

TypeScript **WILL narrow types** based on `if` checks. The solution is NOT `guardReady()` or `asResolved()` — it's **proper type guards within each method**.

```typescript
// LazyReference<T> = string | IOR<T> | T | null
for (const child of this.model.children) {
  // child: string | IOR<FileJs> | FileJs | null
  
  if (typeof child === 'string') {
    // child: string — IOR string, not yet loaded
    continue;
  }
  
  if (child === null) {
    // child: null — empty slot
    continue;
  }
  
  if ('isResolving' in child) {
    // child: IOR<FileJs> — resolution in progress
    continue;
  }
  
  // child: FileJs — TypeScript NARROWS here!
  console.log(child.model.uuid);  // ✅ Safe — no compile error
}
```

## Solution: Type Guards via TypeScript Narrowing

TypeScript narrows types with proper `if` checks. No `guardReady()` or `asResolved()` needed.

### Standard Type Guard Pattern

```typescript
/**
 * Type guard for resolved FileJs instance
 */
function isResolved(child: LazyReference<FileJs>): child is FileJs {
  return child !== null && typeof child !== 'string' && !('isResolving' in child);
}

// Usage in DefaultFolder methods:
for (const child of this.model.children) {
  if (!isResolved(child)) continue;
  
  // TypeScript NARROWS: child is FileJs here
  console.log(child.model.uuid);  // ✅ No compile error
}
```

### Inline Type Guard (No Helper Function)

For methods that only need to filter resolved instances:

```typescript
for (const child of this.model.children) {
  if (typeof child === 'string') continue;  // Skip IOR strings
  if (child === null) continue;              // Skip nulls
  if ('isResolving' in child) continue;      // Skip pending IOR objects
  
  // child: FileJs — TypeScript narrows automatically
  child.pathSet(newPath);  // ✅ Safe
}
```

## Three-Case Callback Pattern (Reference Implementation)

For mutation methods like `childAdd()`, we need behavior for ALL THREE cases.
The pattern uses **naming conventions** to define callbacks for each reference state.

### Naming Convention

| Reference State | Callback Method | When Called |
|-----------------|-----------------|-------------|
| `string` (IOR) | `onChildAddIOR(ior: string)` | Immediately on add |
| `Scenario<T>` (JSON) | `onChildAddScenario(scenario: Scenario<FileModel>)` | Immediately on add |
| `FileJs` (resolved) | `onChildAdd(child: FileJs)` | After resolution |

### Pattern: `childAdd()` as Dispatcher

```typescript
// DefaultFolder.ts

/**
 * Add a child to this folder
 * 
 * Accepts any LazyReference state:
 * - string: IOR reference (deferred resolution)
 * - Scenario: JSON scenario (needs instantiation)
 * - FileJs: Resolved instance (immediate processing)
 * 
 * Dispatches to appropriate callback based on reference state.
 */
childAdd(child: LazyReference<FileJs>): void {
  // Always add to model (regardless of state)
  this.model.children = [...this.model.children, child];
  this.model.modifiedAt = Date.now();
  
  // Case 1: IOR string reference
  if (typeof child === 'string') {
    this.onChildAddIOR(child);
    
    // Register resolution callback
    IOR.resolve<FileJs>(child).then((resolved) => {
      this.onChildAdd(resolved);
    });
    return;
  }
  
  // Case 2: null — skip
  if (child === null) return;
  
  // Case 3: Scenario JSON (has 'model' but no methods)
  if ('model' in child && !('parentSet' in child)) {
    this.onChildAddScenario(child as Scenario<FileModel>);
    
    // Instantiate and register callback
    IOR.instantiate<FileJs>(child).then((resolved) => {
      this.onChildAdd(resolved);
    });
    return;
  }
  
  // Case 4: IOR object (resolving)
  if ('isResolving' in child) {
    // Wait for resolution
    (child as IOR<FileJs>).then((resolved) => {
      this.onChildAdd(resolved);
    });
    return;
  }
  
  // Case 5: Resolved instance
  this.onChildAdd(child);
}

/**
 * Called when IOR string is added (before resolution)
 * 
 * Use for: logging, tracking pending references, UI updates
 */
protected onChildAddIOR(ior: string): void {
  // Optional: track pending IORs
  console.log(`Folder ${this.name}: pending child IOR added`);
}

/**
 * Called when Scenario JSON is added (before instantiation)
 * 
 * Use for: validation, migration, preprocessing
 */
protected onChildAddScenario(scenario: Scenario<FileModel>): void {
  // Optional: validate scenario structure
  console.log(`Folder ${this.name}: scenario added, instantiating...`);
}

/**
 * Called when resolved instance is added (THE MAIN IMPLEMENTATION)
 * 
 * This is where the real work happens:
 * - Set parent reference
 * - Update path
 * - Trigger re-render
 */
protected onChildAdd(child: FileJs): void {
  // Update child's parent reference
  child.parentSet(this);
  
  // Update child's path
  child.pathSet(this.childPath);
  
  // Trigger view update (model change already happened)
  console.log(`Folder ${this.name}: child ${child.name} fully added`);
}
```

### Apply Pattern to All Mutation Methods

| Method | Callbacks | Purpose |
|--------|-----------|---------|
| `childAdd()` | `onChildAddIOR`, `onChildAddScenario`, `onChildAdd` | Add child with deferred setup |
| `childRemove()` | `onChildRemoveIOR`, `onChildRemoveScenario`, `onChildRemove` | Remove with cleanup |
| `move()` | `onMoveIOR`, `onMoveScenario`, `onMove` | Move with path updates |
| `rename()` | `onRenameIOR`, `onRenameScenario`, `onRename` | Rename with path updates |

### Subclass Extension

Subclasses can override specific callbacks:

```typescript
// ImageFolder.ts — specialized folder for images
class ImageFolder extends DefaultFolder {
  
  /**
   * Override: Validate image files before adding
   */
  protected override onChildAddScenario(scenario: Scenario<FileModel>): void {
    super.onChildAddScenario(scenario);
    
    // Validate it's an image
    if (!scenario.model.mimeType?.startsWith('image/')) {
      throw new Error('ImageFolder only accepts image files');
    }
  }
  
  /**
   * Override: Generate thumbnail on add
   */
  protected override onChildAdd(child: FileJs): void {
    super.onChildAdd(child);
    
    // Generate thumbnail for resolved image
    if (child instanceof ImageFile) {
      child.generateThumbnail();
    }
  }
}
```

### Benefits

1. **Type Safety**: Each callback has correct types for its state
2. **Extensibility**: Subclasses override specific callbacks
3. **Separation of Concerns**: IOR handling vs instance handling cleanly separated
4. **Deferred Processing**: Heavy work (parentSet, pathSet) only on resolved instances
5. **Observability**: Easy to add logging/metrics per state

```typescript
import type { LazyReference } from './LazyReference.interface.js';
import type { IOR } from '../layer4/IOR.js';

/**
 * Type guard: Check if a LazyReference is a resolved instance
 * 
 * @param ref The lazy reference to check
 * @returns true if ref is a resolved instance T (not string, IOR, or null)
 */
export function isResolved<T>(ref: LazyReference<T>): ref is T {
  return ref !== null && typeof ref !== 'string' && !('isResolving' in ref);
}

/**
 * Type guard: Check if a LazyReference is an IOR string
 */
export function isIORString<T>(ref: LazyReference<T>): ref is string {
  return typeof ref === 'string' && ref.startsWith('ior:');
}

/**
 * Type guard: Check if a LazyReference is an IOR object (resolving)
 */
export function isIORObject<T>(ref: LazyReference<T>): ref is IOR<T> {
  return ref !== null && typeof ref === 'object' && 'isResolving' in ref;
}
```

```typescript
childRemove(child: LazyReference<FileJs>): boolean {
  // Find index based on type
  const index = this.model.children.findIndex(function matchChild(c) {
    if (typeof c === 'string') return c.includes(extractUuid(child));
    if (c === null) return false;
    if ('isResolving' in c) return false;
    return c.model.uuid === extractUuid(child);
  });
  
  if (index === -1) return false;
  
  // Remove from model
  const removed = this.model.children[index];
  this.model.children = this.model.children.filter(function notMatch(_, i) { 
    return i !== index; 
  });
  this.model.modifiedAt = Date.now();
  
  // Dispatch to appropriate callback
  if (typeof removed === 'string') {
    this.onChildRemoveIOR(removed);
  } else if (removed !== null && !('isResolving' in removed)) {
    this.onChildRemove(removed);
  }
  
  return true;
}
```
