## Problem Statement

### User Feedback (Quoted)

> "the issue is not that the methods are resolved or not resolved. the issue is that the types used in the methods do not guarantee that its resolved."

> "both are no solutions. typescript will narrow the types on if. so we have to check the type of children[1] in these critical methods."

### The Real Issue: Type Safety via Type Narrowing

TypeScript **WILL narrow types** based on `if` checks. The solution is NOT `guardReady()` or `asResolved()` — it's **proper type guards within each method**.

```typescript
// LazyReference<T> = string | IOR<T> | T | null
for (const child of this.model.children) {
  // child: string | IOR<FileJs> | FileJs | null
  
  if (typeof child === 'string') {
    // child: string — IOR string, not yet loaded
    continue;
  }
  
  if (child === null) {
    // child: null — empty slot
    continue;
  }
  
  if ('isResolving' in child) {
    // child: IOR<FileJs> — resolution in progress
    continue;
  }
  
  // child: FileJs — TypeScript NARROWS here!
  console.log(child.model.uuid);  // ✅ Safe — no compile error
}
```

## Solution: Type Guards via TypeScript Narrowing

TypeScript narrows types with proper `if` checks. No `guardReady()` or `asResolved()` needed.

### Standard Type Guard Pattern

```typescript
/**
 * Type guard for resolved FileJs instance
 */
function isResolved(child: LazyReference<FileJs>): child is FileJs {
  return child !== null && typeof child !== 'string' && !('isResolving' in child);
}

// Usage in DefaultFolder methods:
for (const child of this.model.children) {
  if (!isResolved(child)) continue;
  
  // TypeScript NARROWS: child is FileJs here
  console.log(child.model.uuid);  // ✅ No compile error
}
```

### Inline Type Guard (No Helper Function)

For methods that only need to filter resolved instances:

```typescript
for (const child of this.model.children) {
  if (typeof child === 'string') continue;  // Skip IOR strings
  if (child === null) continue;              // Skip nulls
  if ('isResolving' in child) continue;      // Skip pending IOR objects
  
  // child: FileJs — TypeScript narrows automatically
  child.pathSet(newPath);  // ✅ Safe
}
```

## Four-Case Callback Pattern (Reference Implementation)

For mutation methods like `childAdd()`, we handle FOUR cases.
**Key insight**: `onChildAdd()` is THE REAL implementation — all cases lead to it.

### Case Order (null FIRST)

| Order | Case | Immediate Action | Leads to `onChildAdd()`? |
|-------|------|------------------|--------------------------|
| **1** | `null` | Initialize array if needed | No (nothing to add) |
| **2** | `string` (IOR) | Add to array, resolve async | Yes (via callback) |
| **3** | `IOR<T>` object | Add to array, check resolved | Yes (immediate or callback) |
| **4** | `FileJs` instance | Add to array | Yes (immediately) |

### Pattern: `childAdd()` as Dispatcher

```typescript
// DefaultFolder.ts

/**
 * Add a child to this folder
 * 
 * Accepts LazyReference<FileJs> — handles all four states.
 * THE REAL IMPLEMENTATION is onChildAdd() — called only when resolved.
 */
childAdd(child: LazyReference<FileJs>): void {
  
  // ═══════════════════════════════════════════════════════════════
  // CASE 1: null — Initialize array if needed, nothing to add
  // ═══════════════════════════════════════════════════════════════
  if (child === null) {
    // Ensure children array exists (folder might be empty)
    if (this.model.children === null) {
      this.model.children = [];
    }
    return;  // Nothing to add
  }
  
  // ═══════════════════════════════════════════════════════════════
  // CASE 2: string (IOR) — Add to array, defer onChildAdd
  // ═══════════════════════════════════════════════════════════════
  if (typeof child === 'string') {
    // Add IOR string to collection
    this.model.children = [...this.model.children, child];
    this.model.modifiedAt = Date.now();
    
    // Resolve and call onChildAdd when ready
    IOR.resolve<FileJs>(child).then(this.onChildAdd.bind(this));
    return;
  }
  
  // ═══════════════════════════════════════════════════════════════
  // CASE 3: IOR<T> object — Check if already resolved
  // ═══════════════════════════════════════════════════════════════
  if ('isResolving' in child) {
    const ior = child as IOR<FileJs>;
    
    // Add IOR object to collection
    this.model.children = [...this.model.children, ior];
    this.model.modifiedAt = Date.now();
    
    // Check if already resolved
    if (ior.isResolved) {
      // Already resolved — call onChildAdd immediately
      this.onChildAdd(ior.value);
    } else {
      // Not resolved — pass onChildAdd as callback
      ior.then(this.onChildAdd.bind(this));
    }
    return;
  }
  
  // ═══════════════════════════════════════════════════════════════
  // CASE 4: FileJs instance — Add and call onChildAdd immediately
  // ═══════════════════════════════════════════════════════════════
  this.model.children = [...this.model.children, child];
  this.model.modifiedAt = Date.now();
  this.onChildAdd(child);
}

/**
 * THE REAL IMPLEMENTATION — Called when child is resolved
 * 
 * This is where the actual work happens:
 * - Set parent reference on child
 * - Update child's path
 * - Replace IOR/string in array with instance (ISR)
 * 
 * @param child The resolved FileJs instance
 */
protected onChildAdd(child: FileJs): void {
  // Set parent reference
  child.parentSet(this);
  
  // Update path to be under this folder
  child.pathSet(this.childPath);
  
  // ISR: Replace IOR/string in array with resolved instance
  const index = this.model.children.findIndex(function matchChild(c) {
    if (typeof c === 'string') return c.includes(child.model.uuid);
    if (c !== null && 'model' in c) return c.model.uuid === child.model.uuid;
    return false;
  });
  
  if (index !== -1) {
    // Replace with resolved instance (Self-Replacement)
    this.model.children[index] = child;
  }
}
```

### Apply Pattern to All Mutation Methods

Each mutation method has ONE real implementation callback (`onXxx`).
The dispatcher handles the 4 cases and defers to `onXxx` when resolved.

| Method | Dispatcher | Real Implementation | What `onXxx` Does |
|--------|------------|---------------------|-------------------|
| `childAdd()` | Handles 4 cases | `onChildAdd(child)` | Set parent, update path, ISR |
| `childRemove()` | Handles 4 cases | `onChildRemove(child)` | Clear parent, cleanup |
| `move()` | Handles 4 cases | `onMove(child)` | Update paths recursively |
| `rename()` | Handles 4 cases | `onRename(child)` | Update name + paths |
| `delete()` | Handles 4 cases | `onDelete(child)` | Recursive delete |
| `dispose()` | Handles 4 cases | `onDispose(child)` | Cleanup resources |

### Flow Diagram

```
dispatcher(child: LazyReference<T>)
     │
     ▼
┌─ null? ───────────────► Initialize array, return
│
├─ string? ─────────────► Add to array → IOR.resolve().then(onXxx)
│
├─ IOR object? ──┬──────► Add to array
│                │
│          resolved? ─yes─► onXxx(ior.value)
│                │
│                └──no──► ior.then(onXxx)
│
└─ T instance? ─────────► Add to array → onXxx(instance)
```

### Subclass Extension

Subclasses override `onXxx` to add behavior:

```typescript
// ImageFolder.ts — specialized folder for images
class ImageFolder extends DefaultFolder {
  
  /**
   * Override: Generate thumbnail after adding
   */
  protected override onChildAdd(child: FileJs): void {
    // ALWAYS call super first
    super.onChildAdd(child);
    
    // Add specialized behavior
    if (child instanceof ImageFile) {
      child.generateThumbnail();
    }
  }
  
  /**
   * Override: Cleanup thumbnail on remove
   */
  protected override onChildRemove(child: FileJs): void {
    // Cleanup specialized resources first
    if (child instanceof ImageFile) {
      child.deleteThumbnail();
    }
    
    // Then call super for standard cleanup
    super.onChildRemove(child);
  }
}
```

### Benefits

1. **Single Implementation**: `onXxx()` is THE only real implementation — no duplication
2. **Type Safety**: `onXxx(child: FileJs)` guarantees resolved instance
3. **Extensibility**: Subclasses override `onXxx()` for specialized behavior
4. **Deferred Processing**: Heavy work only happens when resolved
5. **ISR Integration**: Dispatcher handles IOR→instance replacement
6. **Null Safety**: `null` case initializes array, prevents NPE

```typescript
import type { LazyReference } from './LazyReference.interface.js';
import type { IOR } from '../layer4/IOR.js';

/**
 * Type guard: Check if a LazyReference is a resolved instance
 * 
 * @param ref The lazy reference to check
 * @returns true if ref is a resolved instance T (not string, IOR, or null)
 */
export function isResolved<T>(ref: LazyReference<T>): ref is T {
  return ref !== null && typeof ref !== 'string' && !('isResolving' in ref);
}

/**
 * Type guard: Check if a LazyReference is an IOR string
 */
export function isIORString<T>(ref: LazyReference<T>): ref is string {
  return typeof ref === 'string' && ref.startsWith('ior:');
}

/**
 * Type guard: Check if a LazyReference is an IOR object (resolving)
 */
export function isIORObject<T>(ref: LazyReference<T>): ref is IOR<T> {
  return ref !== null && typeof ref === 'object' && 'isResolving' in ref;
}
```

```typescript
// childRemove() — Line 198 (simplified, uses callbacks)
childRemove(child: LazyReference<FileJs>): boolean {
  // Find index based on type
  const index = this.model.children.findIndex(function matchChild(c) {
    if (typeof c === 'string') return c.includes(extractUuid(child));
    if (c === null) return false;
    if ('isResolving' in c) return false;
    return c.model.uuid === extractUuid(child);
  });
  
  if (index === -1) return false;
  
  // Remove from model
  const removed = this.model.children[index];
  this.model.children = this.model.children.filter(function notMatch(_, i) { 
    return i !== index; 
  });
  this.model.modifiedAt = Date.now();
  
  // Dispatch to appropriate callback
  if (typeof removed === 'string') {
    this.onChildRemoveIOR(removed);
  } else if (removed !== null && !('isResolving' in removed)) {
    this.onChildRemove(removed);
  }
  
  return true;
}
```
