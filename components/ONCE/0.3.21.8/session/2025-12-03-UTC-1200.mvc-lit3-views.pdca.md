**1.2 Add WebSocket Listener for Real-Time Updates**
```typescript
// demo-hub.html - Add WebSocket connection like 0.3.21.5
function connectWebSocket() {
    const ws = new WebSocket(`ws://${peerHost}`);
    
    ws.onmessage = (event) => {
        const message = JSON.parse(event.data);
        
        switch (message.type) {
            case 'server-registered':
                handleServerRegistered(message.data);
                break;
            case 'server-stopped':
                handleServerStopped(message.data);
                break;
        }
    };
}

function handleServerRegistered(serverModel) {
    // Add peer to kernel.model.peers
    const peer = {
        ior: { uuid: serverModel.uuid, version: onceVersion },
        model: { state: { state: 'RUNNING', capabilities: serverModel.capabilities } }
    };
    kernel.model.peers.push(peer);
    updatePeerDisplay();
}
```

**1.3 Fix peerStopAll to Cascade to Clients**
```typescript
// NodeJsOnce.ts - peerStopAll must notify all registered clients
async peerStopAll(): Promise<void> {
    // 1. Get all registered client servers from ServerHierarchyManager
    const registeredServers = this.serverHierarchyManager.getRegisteredServers();
    
    // 2. Send peerStop IOR to each client
    for (const server of registeredServers) {
        const port = server.capabilities.find(c => c.capability === 'httpPort')?.port;
        const clientIor = `http://localhost:${port}/ONCE/${this.model.version}/${server.uuid}/peerStop`;
        await fetch(clientIor, { method: 'POST' }).catch(() => {});
    }
    
    // 3. Stop self (primary)
    await this.peerStop();
}
```

### Phase 2: MVC Architecture with Lit 3 Web Components

**Architecture Vision:**
```
┌────────────────────────────────────────────────────────────┐
│                    Web4 MVC Architecture                    │
├────────────────────────────────────────────────────────────┤
│  Layer 5: Views (Lit 3 Web Components)                     │
│  ┌──────────┐ ┌─────────────┐ ┌────────────────┐          │
│  │ ItemView │ │ DefaultView │ │ DetailsView    │ (deferred)│
│  │ (list)   │ │ (page)      │ │ (programmer)   │          │
│  └────┬─────┘ └──────┬──────┘ └────────────────┘          │
│       │              │                                      │
│       └──────┬───────┘                                      │
│              ▼                                              │
│  ┌───────────────────┐                                      │
│  │   View interface  │                                      │
│  └─────────┬─────────┘                                      │
├────────────┼─────────────────────────────────────────────────┤
│  Layer 4: Controller (Actions)                              │
│  ┌─────────────────────────────────────────────────────────┐│
│  │ Methods without params → Actions → Buttons              ││
│  │ e.g., peerStart() → [▶ Start Server] button             ││
│  └─────────────────────────────────────────────────────────┘│
├────────────────────────────────────────────────────────────┤
│  Layer 3: Model (Scenarios)                                 │
│  ┌─────────────────────────────────────────────────────────┐│
│  │ Component Scenario (model state)                        ││
│  │ ├── ior: { uuid, component, version }                   ││
│  │ ├── model: { state, capabilities, peers }               ││
│  │ └── actions: ['peerStart', 'peerStop', 'peersDiscover'] ││
│  └─────────────────────────────────────────────────────────┘│
└────────────────────────────────────────────────────────────┘
```

**2.1 View Interface (layer3/View.interface.ts)**
```typescript
/**
 * Web4 View Interface
 * All views implement this for consistent rendering
 * 
 * @ior ior:esm:/ONCE/{version}/View
 */
export interface View<TModel = any> {
  /** Scenario driving this view */
  scenario: Scenario<TModel>;
  
  /** Render the view - returns Lit TemplateResult */
  render(): TemplateResult;
  
  /** Connect to model updates (reactive) */
  modelConnect(scenario: Scenario<TModel>): void;
  
  /** Get available actions (methods with no required params) */
  actionsDiscover(): Action[];
  
  /** Mobile-first: Touch events, drag-drop support */
  touchEnabled: boolean;
}
```

**2.2 ItemView - For Lists/Trees (layer5/ItemView.ts)**
```typescript
import { LitElement, html, css, TemplateResult } from 'lit';
import { customElement, property } from 'lit/decorators.js';

/**
 * ItemView - Compact view for lists and tree nodes
 * 
 * Features:
 * - Mobile touch-first
 * - Drag-and-drop via HTML5 DnD API
 * - Compact horizontal layout
 * - Swipe gestures for actions
 * 
 * @ior ior:esm:/ONCE/{version}/ItemView
 */
@customElement('once-item-view')
export class ItemView extends LitElement implements View {
  @property({ type: Object }) scenario: Scenario<any>;
  @property({ type: Boolean }) touchEnabled = true;
  
  static styles = css`
    :host {
      display: flex;
      align-items: center;
      padding: 12px 16px;
      border-bottom: 1px solid var(--border-color, #333);
      cursor: grab;
      touch-action: pan-y;
      user-select: none;
    }
    
    :host(:active) {
      cursor: grabbing;
      background: var(--highlight-bg, rgba(0,255,0,0.1));
    }
    
    .status-dot {
      width: 12px;
      height: 12px;
      border-radius: 50%;
      margin-right: 12px;
    }
    
    .status-dot.running { background: #0f0; }
    .status-dot.stopped { background: #666; }
    
    .item-info {
      flex: 1;
    }
    
    .item-title {
      font-weight: bold;
      color: var(--text-primary, #fff);
    }
    
    .item-subtitle {
      font-size: 0.85em;
      color: var(--text-secondary, #888);
    }
    
    .action-zone {
      display: flex;
      gap: 8px;
    }
  `;
  
  render(): TemplateResult {
    const model = this.scenario?.model || {};
    const state = model.state?.state || 'unknown';
    const uuid = this.scenario?.ior?.uuid?.substring(0, 8) || '???';
    const port = model.state?.capabilities?.find(c => c.capability === 'httpPort')?.port || 'N/A';
    
    return html`
      <div class="status-dot ${state.toLowerCase()}"></div>
      <div class="item-info">
        <div class="item-title">Peer ${uuid}</div>
        <div class="item-subtitle">Port ${port} · ${state}</div>
      </div>
      <div class="action-zone">
        ${this._renderActions()}
      </div>
    `;
  }
  
  private _renderActions(): TemplateResult {
    const actions = this.actionsDiscover();
    return html`
      ${actions.map(action => html`
        <button @click=${() => this._invokeAction(action)}>
          ${action.icon} ${action.label}
        </button>
      `)}
    `;
  }
  
  actionsDiscover(): Action[] {
    // Discover methods with no required params from scenario
    return [
      { method: 'peerStop', label: 'Stop', icon: '🛑', style: 'danger' }
    ];
  }
  
  modelConnect(scenario: Scenario<any>): void {
    this.scenario = scenario;
    this.requestUpdate();
  }
  
  private async _invokeAction(action: Action): Promise<void> {
    const ior = this.scenario?.ior;
    if (!ior) return;
    
    const url = `http://localhost:${this._getPort()}/ONCE/${ior.version}/${ior.uuid}/${action.method}`;
    await fetch(url, { method: 'POST' });
  }
}
```

**2.4 Action Interface (layer3/Action.interface.ts)**
```typescript
/**
 * Web4 Action - Method exposed as UI action
 * 
 * Actions are methods with NO REQUIRED PARAMETERS
 * that can be mapped directly to buttons.
 * 
 * @ior ior:esm:/ONCE/{version}/Action
 */
export interface Action {
  /** Method name to invoke via IOR */
  method: string;
  
  /** Human-readable label */
  label: string;
  
  /** Emoji or icon */
  icon: string;
  
  /** Visual style: primary, secondary, danger */
  style: 'primary' | 'secondary' | 'danger' | 'warning';
  
  /** Keyboard shortcut (optional) */
  shortcut?: string;
  
  /** Confirmation required before invoke */
  confirmRequired?: boolean;
  
  /** Confirmation message */
  confirmMessage?: string;
}
```

```html
<script type="module">
```
