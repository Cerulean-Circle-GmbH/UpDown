**1.2 Protocol-Less Scenario Live Updates (Web4 Principle 11)**

> ⚠️ **CRITICAL**: WebSocket messages like `server-registered` and `server-stopped` are **PROTOCOL MESSAGES** - this violates Web4 Principle 11!

**The Wrong Way (0.3.21.5 - Protocol Messages):**
```typescript
// ❌ WRONG: Protocol-based message types
ws.onmessage = (event) => {
    const message = JSON.parse(event.data);
    switch (message.type) {
        case 'server-registered':  // ❌ Protocol type!
            handleServerRegistered(message.data);
            break;
    }
};
```

**The Web4 Way (Scenario Replication):**
```typescript
// ✅ RIGHT: Receive scenarios, replicate objects
ws.onmessage = (event) => {
    const scenario = JSON.parse(event.data) as Scenario;
    
    // Scenario IS the peer - just replicate it!
    const existingPeer = kernel.model.peers.find(p => p.ior?.uuid === scenario.ior?.uuid);
    
    if (existingPeer) {
        // Update existing clone's state
        Object.assign(existingPeer, scenario);
    } else {
        // New peer - add clone to registry
        kernel.model.peers.push(scenario);
    }
    
    updatePeerDisplay();
};
```

**How It Works:**
1. Client server calls `toScenario()` to hibernate itself
2. Sends raw scenario over WebSocket (no protocol wrapper!)
3. Primary receives scenario, creates replica (clone) with same UUID
4. Browser receives scenario, updates local replica
5. **Objects communicate by state replication, not messages!**

**Reference:** [§/components/ONCE/0.3.21.2/session/2025-11-21-UTC-1900.iteration-01.6-once-architecture-consolidation.pdca.md#L1728](../../../0.3.21.2/session/2025-11-21-UTC-1900.iteration-01.6-once-architecture-consolidation.pdca.md#L1728) (Section 12: Protocol-Less Communication)
```

**1.3 Fix peerStopAll to Cascade to Clients**
```typescript
// NodeJsOnce.ts - peerStopAll must notify all registered clients
async peerStopAll(): Promise<void> {
    // 1. Get all registered client servers from ServerHierarchyManager
    const registeredServers = this.serverHierarchyManager.getRegisteredServers();
    
    // 2. Send peerStop IOR to each client
    for (const server of registeredServers) {
        const port = server.capabilities.find(c => c.capability === 'httpPort')?.port;
        const clientIor = `http://localhost:${port}/ONCE/${this.model.version}/${server.uuid}/peerStop`;
        await fetch(clientIor, { method: 'POST' }).catch(() => {});
    }
    
    // 3. Stop self (primary)
    await this.peerStop();
}

**2.1 View Interface (layer3/View.interface.ts)**
```typescript
/**
 * Web4 View Interface
 * All views implement this for consistent rendering
 * 
 * @ior ior:esm:/ONCE/{version}/View
 */
export interface View<TModel = any> {
  /** Scenario driving this view */
  scenario: Scenario<TModel>;
  
  /** Render the view - returns Lit TemplateResult */
  render(): TemplateResult;
  
  /** Connect to model updates (reactive) */
  modelConnect(scenario: Scenario<TModel>): void;
  
  /** Get available actions (methods with no required params) */
  actionsDiscover(): Action[];
  
  /** Mobile-first: Touch events, drag-drop support */
  touchEnabled: boolean;
}
```

**2.2.1 AbstractWebBean Base Class (layer5/views/AbstractWebBean.ts)**

Leverages Unit component to track and load external CSS/HTML files:

```typescript
import { LitElement, CSSResult, TemplateResult, html, unsafeCSS } from 'lit';
import { Unit } from '../../layer3/Unit.interface.js';

/**
 * AbstractWebBean - Base class for all Web4 Lit web components
 * 
 * Manages loading of external CSS and HTML template files.
 * Uses Unit component to track file references with UUIDs.
 * 
 * Web4 Principles:
 * - P19: One File One Type (CSS/HTML/TS separate)
 * - P6: Empty constructor, init via scenario
 * - P8: DRY - shared base class
 */
export abstract class AbstractWebBean extends LitElement {
  protected cssUnit: Unit | null = null;
  protected templateUnit: Unit | null = null;
  
  /** Load CSS from external file via Unit */
  protected async cssLoad(cssPath: string): Promise<CSSResult> {
    const cssContent = await this.unitLoad(cssPath);
    return unsafeCSS(cssContent);
  }
  
  /** Load HTML template from external file via Unit */
  protected async templateLoad(templatePath: string): Promise<string> {
    return this.unitLoad(templatePath);
  }
  
  /** Generic file loader via Unit component */
  private async unitLoad(filePath: string): Promise<string> {
    const response = await fetch(filePath);
    return response.text();
  }
  
  /** Abstract: Subclass must provide view name for file discovery */
  abstract get viewName(): string;
  
  /** Construct paths based on view name */
  protected get cssPath(): string {
    return `../../../view/css/${this.viewName}.css`;
  }
  
  protected get templatePath(): string {
    return `../../../view/webBeans/${this.viewName}.html`;
  }
}
```

**2.2.2 ItemView Implementation (layer5/views/ItemView.ts)**

```typescript
import { customElement, property } from 'lit/decorators.js';
import { html, TemplateResult } from 'lit';
import { AbstractWebBean } from './AbstractWebBean.js';
import { View } from '../../layer3/View.interface.js';
import { Action } from '../../layer3/Action.interface.js';
import { Scenario } from '../../layer3/Scenario.interface.js';

/**
 * ItemView - Compact list/tree item view
 * 
 * @ior ior:esm:/ONCE/{version}/ItemView
 */
@customElement('once-item-view')
export class ItemView extends AbstractWebBean implements View {
  @property({ type: Object }) scenario: Scenario<any> | null = null;
  @property({ type: Boolean }) touchEnabled = true;
  
  get viewName(): string { return 'item-view'; }
  
  // CSS loaded from view/css/item-view.css (via AbstractWebBean)
  // Template structure defined in view/webBeans/item-view.html
  
  render(): TemplateResult {
    // Template delegates to external HTML file pattern
    // Implementation uses lit-html for reactive binding
    return this.renderFromModel();
  }
  
  private renderFromModel(): TemplateResult {
    const model = this.scenario?.model || {};
    const state = model.state?.state || 'unknown';
    const uuid = this.scenario?.ior?.uuid?.substring(0, 8) || '???';
    const port = model.state?.capabilities?.find(
      (c: any) => c.capability === 'httpPort'
    )?.port || 'N/A';
    
    return html`
      <div class="status-dot ${state.toLowerCase()}"></div>
      <div class="item-info">
        <div class="item-title">Peer ${uuid}</div>
        <div class="item-subtitle">Port ${port} · ${state}</div>
      </div>
      <div class="action-zone">
        ${this.actionsRender()}
      </div>
    `;
  }
  
  private actionsRender(): TemplateResult {
    const actions = this.actionsDiscover();
    return html`${actions.map(action => html`
      <button @click=${() => this.actionInvoke(action)}>
        ${action.icon} ${action.label}
      </button>
    `)}`;
  }
  
  actionsDiscover(): Action[] {
    return [{ method: 'peerStop', label: 'Stop', icon: '🛑', style: 'danger' }];
  }
  
  modelConnect(scenario: Scenario<any>): void {
    this.scenario = scenario;
    this.requestUpdate();
  }
  
  private async actionInvoke(action: Action): Promise<void> {
    const ior = this.scenario?.ior;
    if (!ior) return;
    // IOR-based method invocation
  }
}
```

**2.4 Action Interface (layer3/Action.interface.ts)**
```typescript
/**
 * Web4 Action - Method exposed as UI action
 * 
 * Actions are methods with NO REQUIRED PARAMETERS
 * that can be mapped directly to buttons.
 * 
 * @ior ior:esm:/ONCE/{version}/Action
 */
export interface Action {
  /** Method name to invoke via IOR */
  method: string;
  
  /** Human-readable label */
  label: string;
  
  /** Emoji or icon */
  icon: string;
  
  /** Visual style: primary, secondary, danger */
  style: 'primary' | 'secondary' | 'danger' | 'warning';
  
  /** Keyboard shortcut (optional) */
  shortcut?: string;
  
  /** Confirmation required before invoke */
  confirmRequired?: boolean;
  
  /** Confirmation message */
  confirmMessage?: string;
}
