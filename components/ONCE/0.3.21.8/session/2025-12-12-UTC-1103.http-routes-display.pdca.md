## **📝 PHASE RO.HTTP.1: /routes Endpoint**

### **Add Route to HTTPServer**

Create a ScenarioRoute or dedicated route to expose registered HTTP routes.

```typescript
// Option A: Add to existing routes registration in NodeJsOnce.ts

// Register /routes endpoint
httpRouter.registerRoute(new ScenarioRoute().init({
  model: {
    pattern: '/routes',
    method: HttpMethod.GET,
    priority: 100,
    handler: 'routesGet'
  }
}));
```

### **Response Format**

```typescript
// GET /routes response:
interface HttpRoutesResponse {
  routes: HttpRouteModel[];
}

interface HttpRouteModel {
  pattern: string;           // e.g., "/{Component}/{version}/**"
  method: string;            // "GET", "POST", etc.
  type: string;              // "StaticFileRoute", "IORRoute", etc.
  description?: string;      // Human-readable description
  priority: number;          // Route matching priority
}
```

### **Implementation in HTTPRouter**

```typescript
// layer2/HTTPRouter.ts - Add method

/**
 * Get all registered routes as models
 * Web4 P16: routesGet() accessor pattern
 */
public routesGet(): RouteModel[] {
  return Array.from(this.routes.values()).map(function(route) {
    return {
      pattern: route.model.pattern,
      method: route.model.method,
      type: route.constructor.name,
      priority: route.model.priority ?? 0,
      description: route.model.description ?? ''
    };
  });
}
```

// File: layer5/views/HttpRouteItemView.ts

/**
 * HttpRouteItemView.ts - HTTP Route Display
 * 
 * Displays a single HTTP route (static, IOR, HTML, scenario).
 * Unlike RouteItemView (SPA), these are NOT navigable - just informational.
 * 
 * @component Router
 * @layer 5
 * @pdca 2025-12-12-UTC-1103.http-routes-display.pdca.md
 */

import { html, TemplateResult } from 'lit';
import { customElement, property } from 'lit/decorators.js';
import { UcpView } from './UcpView.js';

interface HttpRouteModel {
  pattern: string;
  method: string;
  type: string;
  description?: string;
  priority: number;
}

@customElement('http-route-item-view')
export class HttpRouteItemView extends UcpView<HttpRouteModel> {
  
  static cssPath = 'HttpRouteItemView.css';
  
  // ═══════════════════════════════════════════════════════════════
  // COMPUTED ACCESSORS (Web4 P16)
  // ═══════════════════════════════════════════════════════════════
  
  get routePattern(): string {
    return this.model?.pattern ?? '';
  }
  
  get routeMethod(): string {
    return this.model?.method ?? 'GET';
  }
  
  get routeType(): string {
    return this.model?.type ?? 'Route';
  }
  
  get routeTypeIcon(): string {
    switch (this.routeType) {
      case 'StaticFileRoute': return '📁';
      case 'IORRoute': return '🔗';
      case 'HTMLRoute': return '📄';
      case 'ScenarioRoute': return '📊';
      default: return '🔌';
    }
  }
  
  // ═══════════════════════════════════════════════════════════════
  // RENDER
  // ═══════════════════════════════════════════════════════════════
  
  render(): TemplateResult {
    return html`
      <div class="http-route">
        <span class="route-icon">${this.routeTypeIcon}</span>
        <span class="route-method method-${this.routeMethod.toLowerCase()}">${this.routeMethod}</span>
        <code class="route-pattern">${this.routePattern}</code>
        <span class="route-type">${this.routeType}</span>
      </div>
    `;
  }
}

## **📝 PHASE RO.HTTP.3: HttpRouteOverView Component**

### **Component Structure**

```typescript
// File: layer5/views/HttpRouteOverView.ts

/**
 * HttpRouteOverView.ts - HTTP Routes Container
 * 
 * Fetches and displays all registered HTTP routes from /routes endpoint.
 * Groups routes by type (Static, IOR, HTML, Scenario).
 * 
 * @component Router
 * @layer 5
 * @pdca 2025-12-12-UTC-1103.http-routes-display.pdca.md
 */

import { html, TemplateResult } from 'lit';
import { customElement, state } from 'lit/decorators.js';
import { UcpView } from './UcpView.js';
import './HttpRouteItemView.js';

interface HttpRouteModel {
  pattern: string;
  method: string;
  type: string;
  priority: number;
}

@customElement('http-route-over-view')
export class HttpRouteOverView extends UcpView<any> {
  
  static cssPath = 'HttpRouteOverView.css';
  
  @state() private httpRoutes: HttpRouteModel[] = [];
  @state() private loading: boolean = true;
  @state() private error: string = '';
  
  // ═══════════════════════════════════════════════════════════════
  // LIFECYCLE
  // ═══════════════════════════════════════════════════════════════
  
  connectedCallback(): void {
    super.connectedCallback();
    this.routesFetch();
  }
  
  // ═══════════════════════════════════════════════════════════════
  // DATA FETCHING (Web4 P4: Methods, not arrow functions)
  // ═══════════════════════════════════════════════════════════════
  
  /**
   * Fetch HTTP routes from /routes endpoint
   */
  private async routesFetch(): Promise<void> {
    try {
      const response = await fetch('/routes');
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}`);
      }
      const data = await response.json();
      this.httpRoutes = data.routes || [];
      this.loading = false;
    } catch (err) {
      this.error = `Failed to load routes: ${err}`;
      this.loading = false;
    }
  }
  
  /**
   * Group routes by type
   */
  private routesByTypeGet(): Map<string, HttpRouteModel[]> {
    const grouped = new Map<string, HttpRouteModel[]>();
    
    this.httpRoutes.forEach(this.routeGroupAdd.bind(this, grouped));
    
    return grouped;
  }
  
  /**
   * Add route to group
   * Web4 P4: Method for forEach callback
   */
  private routeGroupAdd(grouped: Map<string, HttpRouteModel[]>, route: HttpRouteModel): void {
    const type = route.type;
    if (!grouped.has(type)) {
      grouped.set(type, []);
    }
    grouped.get(type)!.push(route);
  }
  
  /**
   * Render route item
   * Web4 P4: Method for map callback
   */
  private routeItemRender(route: HttpRouteModel): TemplateResult {
    return html`
      <http-route-item-view .model=${route}></http-route-item-view>
    `;
  }
  
  /**
   * Render route group
   */
  private routeGroupRender(type: string, routes: HttpRouteModel[]): TemplateResult {
    return html`
      <div class="route-group">
        <h4 class="group-title">${type}</h4>
        ${routes.map(this.routeItemRender.bind(this))}
      </div>
    `;
  }
  
  // ═══════════════════════════════════════════════════════════════
  // RENDER
  // ═══════════════════════════════════════════════════════════════
  
  render(): TemplateResult {
    if (this.loading) {
      return html`<p class="loading">Loading HTTP routes...</p>`;
    }
    
    if (this.error) {
      return html`<p class="error">${this.error}</p>`;
    }
    
    if (this.httpRoutes.length === 0) {
      return html`<p class="no-routes">No HTTP routes registered</p>`;
    }
    
    const grouped = this.routesByTypeGet();
    const groups: TemplateResult[] = [];
    grouped.forEach((routes, type) => {
      groups.push(this.routeGroupRender(type, routes));
    });
    
    return html`${groups}`;
  }
}
```
