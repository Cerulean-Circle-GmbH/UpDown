## **📝 PHASE RO.HTTP.2: /routes Endpoint**

### **Add to HTTPRouter**

```typescript
// layer2/HTTPRouter.ts - Add method

/**
 * Get all registered routes as unified models
 * Web4 P16: routesGet() accessor pattern
 * Web4 P4: Method, not arrow function
 */
public routesGet(): RouteModel[] {
  const routes: RouteModel[] = [];
  
  this.routes.forEach(this.routeModelExtract.bind(this, routes));
  
  return routes;
}

/**
 * Extract RouteModel from Route instance
 * Web4 P4: Method for forEach callback
 */
private routeModelExtract(routes: RouteModel[], route: Route): void {
  routes.push({
    pattern: route.model.pattern,
    method: route.model.method || 'GET',
    type: this.routeTypeDetect(route),
    title: route.model.name || route.constructor.name,
    priority: route.model.priority ?? 0,
    description: route.model.description ?? ''
  });
}

/**
 * Detect route type from class name
 */
private routeTypeDetect(route: Route): RouteType {
  const name = route.constructor.name;
  if (name.includes('Static')) return RouteType.STATIC;
  if (name.includes('IOR')) return RouteType.IOR;
  if (name.includes('HTML')) return RouteType.HTML;
  if (name.includes('Scenario')) return RouteType.SCENARIO;
  return RouteType.API;
}
```

## **📝 PHASE RO.HTTP.3: Update RouteItemView (Unified)**

### **Extend Existing RouteItemView for ALL Route Types**

```typescript
// File: layer5/views/RouteItemView.ts - UPDATED

/**
 * RouteItemView.ts - UNIFIED Route Item View
 * 
 * Displays ANY route type: SPA, HTTP, IOR, Static, Scenario
 * Uses RouteType to determine display and behavior.
 * 
 * Web4 Principles:
 * - P4: Radical OOP (method references)
 * - P16: TypeScript accessors
 * - P27: Web Components ARE Radical OOP
 */

import { html, TemplateResult } from 'lit';
import { customElement, property } from 'lit/decorators.js';
import { UcpView } from './UcpView.js';
import type { RouteModel } from '../../layer3/RouteScenario.interface.js';
import { RouteType } from '../../layer3/RouteType.enum.js';

@customElement('route-item-view')
export class RouteItemView extends UcpView<RouteModel> {
  
  static cssPath = 'RouteItemView.css';
  
  @property({ attribute: false }) router: any = null;
  
  // ═══════════════════════════════════════════════════════════════
  // COMPUTED ACCESSORS (Web4 P16)
  // ═══════════════════════════════════════════════════════════════
  
  get routeType(): RouteType {
    return this.model?.type ?? RouteType.SPA;
  }
  
  get isSpaRoute(): boolean {
    return this.routeType === RouteType.SPA;
  }
  
  get routeIcon(): string {
    switch (this.routeType) {
      case RouteType.SPA: return '🌐';
      case RouteType.STATIC: return '📁';
      case RouteType.IOR: return '🔗';
      case RouteType.HTML: return '📄';
      case RouteType.SCENARIO: return '📊';
      case RouteType.API: return '🔌';
      default: return '📍';
    }
  }
  
  get routeMethod(): string {
    return this.model?.method ?? 'GET';
  }
  
  get showMethodBadge(): boolean {
    // Show method badge for HTTP routes, not for SPA
    return !this.isSpaRoute;
  }
  
  get routeTitle(): string {
    if (this.model?.title) return this.model.title;
    if (this.model?.pattern === '/') return 'Main';
    const name = this.model?.pattern?.slice(1) ?? '';
    return name.charAt(0).toUpperCase() + name.slice(1);
  }
  
  // ═══════════════════════════════════════════════════════════════
  // NAVIGATION (Web4 P4)
  // ═══════════════════════════════════════════════════════════════
  
  private linkClickHandle(event: Event): void {
    if (this.isSpaRoute) {
      // SPA navigation
      event.preventDefault();
      if (this.router?.navigateTo) {
        this.router.navigateTo(this.model?.pattern, false);
      } else {
        window.history.pushState({}, '', this.model?.pattern);
        window.dispatchEvent(new PopStateEvent('popstate'));
      }
    }
    // HTTP routes: let default behavior (open in new tab)
  }
  
  // ═══════════════════════════════════════════════════════════════
  // RENDER
  // ═══════════════════════════════════════════════════════════════
  
  render(): TemplateResult {
    const activeClass = this.model?.isActive ? 'active' : '';
    const target = this.isSpaRoute ? '' : '_blank';
    
    return html`
      <a href="${this.model?.pattern ?? '#'}" 
         class="route-link ${activeClass} route-type-${this.routeType}"
         target="${target}"
         @click=${this.linkClickHandle.bind(this)}>
        <span class="route-icon">${this.routeIcon}</span>
        ${this.showMethodBadge ? html`
          <span class="method-badge method-${this.routeMethod.toLowerCase()}">${this.routeMethod}</span>
        ` : ''}
        <span class="route-title">${this.routeTitle}</span>
        <code class="route-path">${this.model?.pattern}</code>
      </a>
    `;
  }
}
```

## **📝 PHASE RO.HTTP.4: Update RouteOverView (Unified)**

### **Extend Existing RouteOverView to Include HTTP Routes**

```typescript
// File: layer5/views/RouteOverView.ts - UPDATED

/**
 * RouteOverView.ts - UNIFIED Route Overview
 * 
 * Displays ALL routes: SPA (from router) + HTTP (from /routes endpoint)
 * Uses single <route-item-view> for all route types.
 * 
 * Web4 Principles:
 * - P4: Method references
 * - P27: Declarative composition
 */

import { html, TemplateResult } from 'lit';
import { customElement, property, state } from 'lit/decorators.js';
import { UcpView } from './UcpView.js';
import type { RouteModel } from '../../layer3/RouteScenario.interface.js';
import { RouteType } from '../../layer3/RouteType.enum.js';
import './RouteItemView.js';

@customElement('route-over-view')
export class RouteOverView extends UcpView<any> {
  
  static cssPath = 'RouteOverView.css';
  
  /** Router reference for SPA routes */
  @property({ attribute: false }) router: any = null;
  
  /** Whether to include HTTP routes from /routes endpoint */
  @property({ type: Boolean }) includeHttpRoutes: boolean = false;
  
  /** Whether to group routes by type */
  @property({ type: Boolean }) groupByType: boolean = true;
  
  @state() private httpRoutes: RouteModel[] = [];
  @state() private loading: boolean = false;
  
  // ═══════════════════════════════════════════════════════════════
  // LIFECYCLE
  // ═══════════════════════════════════════════════════════════════
  
  connectedCallback(): void {
    super.connectedCallback();
    if (this.includeHttpRoutes) {
      this.httpRoutesFetch();
    }
  }
  
  // ═══════════════════════════════════════════════════════════════
  // DATA ACCESS (Web4 P4: Methods)
  // ═══════════════════════════════════════════════════════════════
  
  /**
   * Get SPA routes from router
   */
  private spaRoutesGet(): RouteModel[] {
    if (!this.router?.routesGet) return [];
    const routeMap = this.router.routesGet() as Map<string, any>;
    const routes: RouteModel[] = [];
    routeMap.forEach(this.spaRouteExtract.bind(this, routes));
    return routes;
  }
  
  private spaRouteExtract(routes: RouteModel[], route: any): void {
    routes.push({ ...route.model, type: RouteType.SPA });
  }
  
  /**
   * Fetch HTTP routes from /routes endpoint
   */
  private async httpRoutesFetch(): Promise<void> {
    this.loading = true;
    try {
      const response = await fetch('/routes');
      if (response.ok) {
        const data = await response.json();
        this.httpRoutes = data.routes || [];
      }
    } catch (err) {
      console.warn('[RouteOverView] Failed to fetch HTTP routes:', err);
    }
    this.loading = false;
  }
  
  /**
   * Get all routes (SPA + HTTP if enabled)
   */
  private allRoutesGet(): RouteModel[] {
    const spaRoutes = this.spaRoutesGet();
    if (!this.includeHttpRoutes) return spaRoutes;
    return [...spaRoutes, ...this.httpRoutes];
  }
  
  /**
   * Group routes by type
   */
  private routesByTypeGet(): Map<RouteType, RouteModel[]> {
    const grouped = new Map<RouteType, RouteModel[]>();
    const allRoutes = this.allRoutesGet();
    allRoutes.forEach(this.routeGroupAdd.bind(this, grouped));
    return grouped;
  }
  
  private routeGroupAdd(grouped: Map<RouteType, RouteModel[]>, route: RouteModel): void {
    const type = route.type ?? RouteType.SPA;
    if (!grouped.has(type)) grouped.set(type, []);
    grouped.get(type)!.push(route);
  }
  
  /**
   * Render single route
   * Web4 P4: Method for map callback
   */
  private routeItemRender(route: RouteModel): TemplateResult {
    return html`
      <route-item-view .model=${route} .router=${this.router}></route-item-view>
    `;
  }
  
  /**
   * Render route group
   */
  private routeGroupRender(type: RouteType, routes: RouteModel[]): TemplateResult {
    const typeLabels: Record<RouteType, string> = {
      [RouteType.SPA]: '🌐 SPA Routes',
      [RouteType.STATIC]: '📁 Static Files',
      [RouteType.IOR]: '🔗 IOR Methods',
      [RouteType.HTML]: '📄 HTML Pages',
      [RouteType.SCENARIO]: '📊 Scenario Endpoints',
      [RouteType.API]: '🔌 API Endpoints'
    };
    
    return html`
      <div class="route-group">
        <h4 class="group-title">${typeLabels[type] ?? type}</h4>
        ${routes.map(this.routeItemRender.bind(this))}
      </div>
    `;
  }
  
  // ═══════════════════════════════════════════════════════════════
  // RENDER
  // ═══════════════════════════════════════════════════════════════
  
  render(): TemplateResult {
    if (this.loading) {
      return html`<p class="loading">Loading routes...</p>`;
    }
    
    const allRoutes = this.allRoutesGet();
    
    if (allRoutes.length === 0) {
      return html`<p class="no-routes">No routes registered</p>`;
    }
    
    // Grouped view
    if (this.groupByType) {
      const grouped = this.routesByTypeGet();
      const groups: TemplateResult[] = [];
      grouped.forEach((routes, type) => {
        groups.push(this.routeGroupRender(type, routes));
      });
      return html`${groups}`;
    }
    
    // Flat view
    return html`${allRoutes.map(this.routeItemRender.bind(this))}`;
  }
}
```
