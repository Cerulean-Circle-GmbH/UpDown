   ```typescript
   import { DefaultWeb4TestCase } from '../../../../Web4Test/0.1.0.0/src/ts/layer2/DefaultWeb4TestCase';
   import { execSync } from 'child_process';
   import * as fs from 'fs/promises';
   import * as path from 'path';
   
   /**
    * ONCETestCase - Base for all ONCE lifecycle tests
    * Provides ONCE-specific utilities (scenario observation, CLI invocation)
    */
   export abstract class ONCETestCase extends DefaultWeb4TestCase {
     protected testDataDir!: string;
     protected onceCLI!: string;
     
     init(scenario: any): this {
       super.init(scenario);
       this.testDataDir = scenario.testDataScenario.testDataDir;
       this.onceCLI = path.join(this.testDataDir, 'scripts/once');
       return this;
     }
     
     // Scenario observation (NO HTTP!)
     protected async observeScenarioFile(serverName: string): Promise<any> {
       const scenariosDir = path.join(this.testDataDir, 'scenarios/ONCE');
       const files = await fs.readdir(scenariosDir);
       const matchingFile = files.find(f => 
         f.includes(serverName) && f.endsWith('.scenario.json')
       );
       
       if (!matchingFile) {
         throw new Error(`Scenario file not found for ${serverName}`);
       }
       
       const scenarioPath = path.join(scenariosDir, matchingFile);
       const content = await fs.readFile(scenarioPath, 'utf-8');
       return JSON.parse(content);
     }
     
     // CLI invocation (IOR pattern)
     protected invokeONCEMethod(method: string, ...args: string[]): string {
       const argsStr = args.join(' ');
       return execSync(`${this.onceCLI} ${method} ${argsStr}`, {
         cwd: this.testDataDir,
         encoding: 'utf-8'
       });
     }
     
     // Quality assessment helper
     protected assessQuality(verdict: boolean, confidence: number, reason: string): any {
       return {
         verdict: verdict ? 'GOOD' : 'POOR',
         confidence,
         reason,
         timestamp: new Date().toISOString()
       };
     }
   }
   ```

#### **Step 3: Create 15 ONCE Test Cases** (6-8 hours)

**Pattern for Each Test:**
```typescript
import { ONCETestCase } from '../base/ONCETestCase';

export class ServerStartupTest extends ONCETestCase {
  protected async executeTestLogic(): Promise<any> {
    // 1. RECORD INPUT EVIDENCE
    this.recordEvidence('input', 'Server startup configuration', {
      port: this.scenario?.testDataScenario.port,
      hostname: this.scenario?.testDataScenario.hostname
    });
    
    // 2. INVOKE SERVER (CLI = IOR pattern)
    const startTime = Date.now();
    try {
      this.invokeONCEMethod('startServer');
    } catch (error) {
      this.recordEvidence('error', 'Server start failed', { error });
      throw error;
    }
    const startupDuration = Date.now() - startTime;
    
    // 3. RECORD STEP EVIDENCE
    this.recordEvidence('step', 'Server startup process', {
      duration: startupDuration,
      timestamp: new Date().toISOString()
    });
    
    // 4. OBSERVE SCENARIO (NO HTTP!)
    await this.waitForScenarioFile('primary', 5000);
    const serverScenario = await this.observeScenarioFile('primary');
    
    // 5. RECORD OUTPUT EVIDENCE
    this.recordEvidence('output', 'Server scenario observed', serverScenario);
    
    // 6. ASSESS QUALITY
    const isHealthy = serverScenario.model.primaryServer !== null;
    const quality = this.assessQuality(
      isHealthy,
      0.95,
      isHealthy ? 'Server started successfully' : 'Server not healthy'
    );
    
    // 7. RECORD ASSERTION EVIDENCE
    this.recordEvidence('assertion', 'Server health assessment', quality);
    
    return {
      success: isHealthy,
      startupDuration,
      serverScenario,
      quality
    };
  }
  
  private async waitForScenarioFile(serverName: string, timeout: number): Promise<void> {
    // Poll for scenario file creation
    const startTime = Date.now();
    while (Date.now() - startTime < timeout) {
      try {
        await this.observeScenarioFile(serverName);
        return;
      } catch {
        await new Promise(resolve => setTimeout(resolve, 100));
      }
    }
    throw new Error(`Scenario file for ${serverName} not created within ${timeout}ms`);
  }
}
```

#### **Step 4: Quality Oracle Integration** (2-3 hours)

**Goal**: Add Tootsie Quality Oracle post-test assessment

**Create:** `test/tootsie/quality/ONCEQualityAssessor.ts`
```typescript
import { QualityOracle } from '../../../../Tootsie/0.1.0.0/src/ts/layer2/QualityOracle';
import { QualityScenario } from '../../../../Tootsie/0.1.0.0/src/ts/layer3/QualityScenario';

export class ONCEQualityAssessor {
  private oracle: QualityOracle;
  
  constructor() {
    this.oracle = new QualityOracle();
  }
  
  async assessTestSuite(testResults: any[]): Promise<any> {
    // Awaken Quality Oracle
    const qualityScenario = this.createQualityScenario();
    await this.oracle.awaken(qualityScenario);
    
    // Judge each test result
    const judgments = [];
    for (const result of testResults) {
      const judgment = await this.oracle.judge(result, qualityScenario);
      judgments.push(judgment);
    }
    
    // Overall quality assessment
    const overallQuality = this.calculateOverallQuality(judgments);
    
    // Evolve oracle (learn from this assessment)
    await this.oracle.evolve();
    
    // Hibernate oracle state
    const oracleIOR = await this.oracle.hibernate();
    
    return {
      judgments,
      overallQuality,
      oracleIOR,
      consciousnessLevel: await this.oracle.getConsciousnessLevel()
    };
  }
  
  private createQualityScenario(): QualityScenario {
    // Create quality assessment context
    return {
      getConsciousnessConfiguration: () => ({
        intelligenceLevel: 50,
        learningRate: 0.1,
        wisdomThreshold: 0.8
      }),
      getQualityContext: () => ({
        domain: 'ONCE Lifecycle Testing',
        standards: ['Web4 Radical OOP', 'Black-Box Testing', 'IOR-based'],
        expectations: {
          passRate: 1.0,
          evidenceCompleteness: 1.0,
          protocolIndependence: 1.0
        }
      }),
      hibernate: async () => ({} as any)
    } as any;
  }
  
  private calculateOverallQuality(judgments: any[]): any {
    const totalConfidence = judgments.reduce((sum, j) => 
      sum + j.getConfidence(), 0
    );
    const avgConfidence = totalConfidence / judgments.length;
    
    return {
      verdict: avgConfidence > 0.8 ? 'EXCELLENT' : 'ACCEPTABLE',
      confidence: avgConfidence,
      passRate: judgments.filter(j => 
        j.getVerdict() === 'GOOD'
      ).length / judgments.length
    };
  }
}
```
