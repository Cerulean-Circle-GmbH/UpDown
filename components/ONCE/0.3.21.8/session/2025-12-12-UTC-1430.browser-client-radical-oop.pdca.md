### **Fixed HTTPSLoader.ts (P4 Compliant)**

```typescript
/**
 * HTTPSLoader.ts
 * 
 * Web4 HTTPS Loader - ISOMORPHIC (Browser + Node.js)
 * Uses fetch() API which works in both environments.
 * 
 * Protocol: 'https'
 * 
 * Web4 P4: No arrow functions - all callbacks use bound methods
 */

// ... existing imports ...

export class HTTPSLoader implements Loader {
    // ... existing properties ...
    
    /**
     * Extract IOR profiles from URL
     * 
     * Web4 P4: Use bound method instead of arrow function
     */
    private extractProfiles(ior: string): IORProfile[] {
        const match = ior.match(/https:\/\/([^\/]+)/);
        if (!match) {
            throw new Error(`Invalid HTTPS IOR format: ${ior}`);
        }
        
        const hostPortString = match[1];
        const hostPorts = hostPortString.split(',');
        
        // P4 FIX: Use bound method instead of arrow function
        const profiles: IORProfile[] = [];
        hostPorts.forEach(this.profileParse.bind(this, profiles));
        return profiles;
    }
    
    /**
     * Parse single host:port into profile
     * 
     * Web4 P4: Named method for .forEach() callback
     */
    private profileParse(profiles: IORProfile[], hostPort: string): void {
        const [host, portStr] = hostPort.split(':');
        profiles.push({
            host: host.trim(),
            port: parseInt(portStr, 10)
        });
    }
    
    // ... rest of existing implementation unchanged ...
}
```

```typescript
/**
 * Initialize browser loader chain
 * Web4 P4: All method references

private loaderChainInit(): void {
  // Create browser-specific HTTPS loader
  const browserHttpsLoader = new BrowserHTTPSLoader().init({
    model: { uuid: this.uuidGenerate(), name: 'BrowserHTTPSLoader' }
  });
  
  // Create REST loader (reuses existing)
  const restLoader = new RESTLoader().init({
    model: { uuid: this.uuidGenerate(), name: 'RESTLoader' }
  });
  restLoader.registerLoader(browserHttpsLoader);
  
  // Create Scenario loader (reuses existing)
  const scenarioLoader = new ScenarioLoader().init({
    model: { uuid: this.uuidGenerate(), name: 'ScenarioLoader' }
  });
  scenarioLoader.registerLoader(restLoader);
  
  // Register with kernel
  global.ONCE.loaderRegistry.set('scenario', scenarioLoader);
  global.ONCE.loaderRegistry.set('REST', restLoader);
  global.ONCE.loaderRegistry.set('https', browserHttpsLoader);
}
```
