## **📊 SUMMARY**

### **The Problem**

Current PWA PDCA has this pattern:

```typescript
// ❌ WRONG: Anonymous functions in .then() chains
fetch(ior, { method: 'POST', body: JSON.stringify(scenario) })
  .then(function(response) { return response.json(); })  // Anonymous!
  .then(function(result) { console.log(result); });       // Anonymous!
```

Web4 Principle 4 says: **No arrow functions, no anonymous functions**.

### **The Solution**

Use the existing **Loader pattern** (already Radical OOP) for browser HTTP:

```typescript
// ✅ CORRECT: IOR Loader chain (existing pattern)
// ior:scenario:REST:https://host/ONCE/0.3.21.8/{uuid}/scenarioReceive

const scenarioLoader = global.ONCE.loaderRegistry.get('scenario');
scenarioLoader.save(scenario, ior);  // No anonymous functions!
```

## **📝 PHASE BC.1: BrowserHTTPSLoader Implementation**

### **BrowserHTTPSLoader.ts** (layer2)

```typescript
// layer2/BrowserHTTPSLoader.ts

/**
 * BrowserHTTPSLoader
 * 
 * Browser-side HTTPS Loader using fetch() API
 * Wraps fetch() in Radical OOP - NO anonymous functions!
 * 
 * @component ONCE
 * @layer 2
 * Web4 P4: All callbacks are method references
 * Web4 P19: One file, one type
 */

import { Loader } from '../layer3/Loader.interface.js';
import { LoaderModel } from '../layer3/LoaderModel.interface.js';
import { Scenario } from '../layer3/Scenario.interface.js';

export class BrowserHTTPSLoader implements Loader {
  public protocol = 'https';
  public model: LoaderModel;
  
  // Response handling state (for method references)
  private pendingResponse: Response | null = null;
  private pendingResolve: ((value: string) => void) | null = null;
  private pendingReject: ((error: Error) => void) | null = null;
  
  constructor() {
    const now = new Date().toISOString();
    this.model = {
      uuid: '',
      name: 'BrowserHTTPSLoader',
      protocol: 'https',
      statistics: {
        totalOperations: 0,
        successCount: 0,
        errorCount: 0,
        lastOperationAt: '',
        lastErrorAt: '',
        createdAt: now,
        updatedAt: now
      }
    };
  }
  
  public init(scenario?: Scenario<LoaderModel>): this {
    if (scenario?.model) {
      Object.assign(this.model, scenario.model);
    }
    return this;
  }
  
  // ═══════════════════════════════════════════════════════════════
  // LOAD (GET request)
  // ═══════════════════════════════════════════════════════════════
  
  /**
   * Load via HTTPS (browser fetch)
   * 
   * Web4 P4: Method references, not anonymous functions
   */
  public load(ior: string, options?: any): Promise<string> {
    this.model.statistics.totalOperations++;
    this.model.statistics.lastOperationAt = new Date().toISOString();
    
    const url = this.urlFromIOR(ior);
    
    return new Promise(this.loadExecutor.bind(this, url, options));
  }
  
  /**
   * Promise executor - bound method reference
   */
  private loadExecutor(
    url: string, 
    options: any,
    resolve: (value: string) => void, 
    reject: (error: Error) => void
  ): void {
    this.pendingResolve = resolve;
    this.pendingReject = reject;
    
    fetch(url, {
      method: options?.method || 'GET',
      headers: options?.headers || {}
    })
      .then(this.responseHandle.bind(this))
      .catch(this.errorHandle.bind(this));
  }
  
  /**
   * Handle fetch response - method reference
   */
  private responseHandle(response: Response): void {
    if (!response.ok) {
      this.errorHandle(new Error(`HTTP ${response.status}: ${response.statusText}`));
      return;
    }
    
    this.pendingResponse = response;
    response.text()
      .then(this.textResolve.bind(this))
      .catch(this.errorHandle.bind(this));
  }
  
  /**
   * Resolve with text body - method reference
   */
  private textResolve(text: string): void {
    this.model.statistics.successCount++;
    if (this.pendingResolve) {
      this.pendingResolve(text);
    }
    this.pendingCleanup();
  }
  
  /**
   * Handle errors - method reference
   */
  private errorHandle(error: Error): void {
    this.model.statistics.errorCount++;
    this.model.statistics.lastErrorAt = new Date().toISOString();
    
    if (this.pendingReject) {
      this.pendingReject(error);
    }
    this.pendingCleanup();
  }
  
  /**
   * Cleanup pending state
   */
  private pendingCleanup(): void {
    this.pendingResponse = null;
    this.pendingResolve = null;
    this.pendingReject = null;
  }
  
  // ═══════════════════════════════════════════════════════════════
  // SAVE (POST request)
  // ═══════════════════════════════════════════════════════════════
  
  /**
   * Save via HTTPS (browser fetch POST)
   */
  public save(data: string, ior: string, options?: any): Promise<void> {
    this.model.statistics.totalOperations++;
    this.model.statistics.lastOperationAt = new Date().toISOString();
    
    const url = this.urlFromIOR(ior);
    
    return new Promise(this.saveExecutor.bind(this, url, data, options));
  }
  
  private saveExecutor(
    url: string,
    data: string,
    options: any,
    resolve: () => void,
    reject: (error: Error) => void
  ): void {
    fetch(url, {
      method: options?.method || 'POST',
      headers: {
        'Content-Type': 'application/json',
        ...options?.headers
      },
      body: data
    })
      .then(this.saveResponseHandle.bind(this, resolve))
      .catch(this.saveErrorHandle.bind(this, reject));
  }
  
  private saveResponseHandle(resolve: () => void, response: Response): void {
    if (!response.ok) {
      this.model.statistics.errorCount++;
      return;
    }
    this.model.statistics.successCount++;
    resolve();
  }
  
  private saveErrorHandle(reject: (error: Error) => void, error: Error): void {
    this.model.statistics.errorCount++;
    this.model.statistics.lastErrorAt = new Date().toISOString();
    reject(error);
  }
  
  // ═══════════════════════════════════════════════════════════════
  // IOR PARSING
  // ═══════════════════════════════════════════════════════════════
  
  /**
   * Extract URL from IOR
   * 
   * @example
   * ior:scenario:REST:https://host/path → https://host/path
   */
  private urlFromIOR(ior: string): string {
    const match = ior.match(/https:\/\/.+/);
    if (!match) {
      throw new Error(`Invalid HTTPS IOR: ${ior}`);
    }
    return match[0];
  }
  
  // ═══════════════════════════════════════════════════════════════
  // SCENARIO CONVERSION
  // ═══════════════════════════════════════════════════════════════
  
  public async toScenario(): Promise<Scenario<LoaderModel>> {
    return {
      ior: {
        uuid: this.model.uuid,
        component: 'BrowserHTTPSLoader',
        version: '0.3.21.8'
      },
      owner: '',
      model: this.model
    };
  }
}
```

## **📝 PHASE BC.3: Loader Registry in Browser**

### **BrowserOnceOrchestrator Integration**

```typescript
// layer4/BrowserOnceOrchestrator.ts

/**
 * Initialize browser loader chain
 * Web4 P4: All method references
 */
private loaderChainInit(): void {
  // Create browser-specific HTTPS loader
  const browserHttpsLoader = new BrowserHTTPSLoader().init({
    model: { uuid: this.uuidGenerate(), name: 'BrowserHTTPSLoader' }
  });
  
  // Create REST loader (reuses existing)
  const restLoader = new RESTLoader().init({
    model: { uuid: this.uuidGenerate(), name: 'RESTLoader' }
  });
  restLoader.registerLoader(browserHttpsLoader);
  
  // Create Scenario loader (reuses existing)
  const scenarioLoader = new ScenarioLoader().init({
    model: { uuid: this.uuidGenerate(), name: 'ScenarioLoader' }
  });
  scenarioLoader.registerLoader(restLoader);
  
  // Register with kernel
  global.ONCE.loaderRegistry.set('scenario', scenarioLoader);
  global.ONCE.loaderRegistry.set('REST', restLoader);
  global.ONCE.loaderRegistry.set('https', browserHttpsLoader);
}
```
