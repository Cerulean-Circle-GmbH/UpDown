### **SNIManager.ts**

```typescript
// layer2/SNIManager.ts

/**
 * SNIManager - Server Name Indication (SNI) context management
 * 
 * Manages TLS SecureContext per domain, enabling one HTTPS server
 * to serve multiple domains with different certificates.
 * 
 * Radical OOP:
 * - Empty constructor + init() (P6)
 * - No arrow functions (P4)
 * 
 * @component HTTPSServer
 * @layer 2
 */

import * as tls from 'tls';
import { DomainCertificate } from '../layer3/DomainCertificate.interface.js';

export interface SNIManagerModel {
    uuid: string;
    name: string;
    domains: string[];
    defaultDomain?: string;
}

export class SNIManager {
    public model: SNIManagerModel;
    
    private contexts: Map<string, tls.SecureContext> = new Map();
    private defaultContext: tls.SecureContext | null = null;
    
    constructor() {
        // Empty constructor - Web4 P6
        this.model = {
            uuid: '',
            name: 'SNIManager',
            domains: []
        };
    }
    
    /**
     * Add or update certificate for domain
     * Supports hot-reload without server restart
     */
    public certificateSet(domain: string, cert: DomainCertificate): void {
        const context = tls.createSecureContext({
            cert: cert.certificate,
            key: cert.privateKey,
            ca: cert.caChain,
            minVersion: 'TLSv1.2'
        });
        
        this.contexts.set(domain.toLowerCase(), context);
        
        if (!this.model.domains.includes(domain)) {
            this.model.domains.push(domain);
        }
        
        console.log(`[SNIManager] Certificate set for: ${domain}`);
    }
    
    /**
     * Set default (fallback) certificate
     */
    public defaultCertificateSet(cert: DomainCertificate): void {
        this.defaultContext = tls.createSecureContext({
            cert: cert.certificate,
            key: cert.privateKey,
            ca: cert.caChain,
            minVersion: 'TLSv1.2'
        });
        
        this.model.defaultDomain = cert.domain;
        console.log(`[SNIManager] Default certificate set: ${cert.domain}`);
    }
    
    /**
     * SNI callback for https.createServer()
     * Web4 P4: Bound method reference
     * 
     * Usage: https.createServer({ SNICallback: sniManager.sniCallback.bind(sniManager) })
     */
    public sniCallback(
        servername: string, 
        callback: (err: Error | null, ctx?: tls.SecureContext) => void
    ): void {
        const domain = servername.toLowerCase();
        
        // Try exact match
        let context = this.contexts.get(domain);
        
        // Try wildcard match
        if (!context) {
            context = this.wildcardContextFind(domain);
        }
        
        // Fallback to default
        if (!context) {
            context = this.defaultContext ?? undefined;
        }
        
        if (context) {
            callback(null, context);
        } else {
            callback(new Error(`No certificate for: ${domain}`));
        }
    }
    
    /**
     * Find wildcard certificate for domain
     */
    private wildcardContextFind(domain: string): tls.SecureContext | undefined {
        const parts = domain.split('.');
        if (parts.length < 2) return undefined;
        
        // Try *.example.com for sub.example.com
        const wildcard = '*.' + parts.slice(1).join('.');
        return this.contexts.get(wildcard);
    }
    
    /**
     * Remove certificate for domain
     */
    public certificateRemove(domain: string): void {
        this.contexts.delete(domain.toLowerCase());
        const idx = this.model.domains.indexOf(domain);
        if (idx !== -1) {
            this.model.domains.splice(idx, 1);
        }
    }
    
    /**
     * Get all configured domains
     */
    public domainsGet(): string[] {
        return [...this.model.domains];
    }
}
```

```typescript
// Update HTTPRouter.ts

/**
 * Enhanced route matching with domain support
 */
private findMatchingRoute(req: IncomingMessage): Route | undefined {
    const path = this.pathExtract(req);
    const method = this.methodExtract(req);
    const hostname = this.hostnameExtract(req);  // NEW
    
    const allRoutes = Array.from(this.routes.values());
    allRoutes.sort((a, b) => a.model.priority - b.model.priority);
    
    for (const route of allRoutes) {
        // Check domain match first (if domains specified)
        if (!this.domainMatches(hostname, route.model.domains)) {
            continue;
        }
        
        // Then check path and method
        if (route.matches(path, method)) {
            return route;
        }
    }
    
    return undefined;
}

/**
 * Check if hostname matches route domains
 */
private domainMatches(hostname: string, domains?: string[]): boolean {
    // No domain restriction = matches all
    if (!domains || domains.length === 0) {
        return true;
    }
    
    const hostLower = hostname.toLowerCase();
    
    for (const domain of domains) {
        const domainLower = domain.toLowerCase();
        
        // Exact match
        if (hostLower === domainLower) {
            return true;
        }
        
        // Wildcard match (*.example.com)
        if (domainLower.startsWith('*.')) {
            const suffix = domainLower.slice(1);  // .example.com
            if (hostLower.endsWith(suffix) && hostLower.length > suffix.length) {
                return true;
            }
        }
    }
    
    return false;
}

/**
 * Extract hostname from request
 */
private hostnameExtract(req: IncomingMessage): string {
    const host = req.headers.host || '';
    // Remove port if present
    return host.split(':')[0].toLowerCase();
}
```

```typescript
// layer2/LetsEncryptProvider.ts

/**
 * LetsEncryptProvider - ACME v2 client for Let's Encrypt
 * 
 * Implements HTTP-01 and DNS-01 challenges.
 * Uses Node.js native crypto (no external ACME library).
 * 
 * @component CertificateManager
 * @layer 2
 * Web4 P4: No arrow functions
 */

import * as crypto from 'crypto';
import * as https from 'https';
import { CertificateProvider } from '../layer3/CertificateProvider.interface.js';
import { DomainCertificate } from '../layer3/DomainCertificate.interface.js';
import { DomainConfig } from '../layer3/DomainConfig.interface.js';

/** ACME Directory URLs */
const ACME_PRODUCTION = 'https://acme-v02.api.letsencrypt.org/directory';
const ACME_STAGING = 'https://acme-staging-v02.api.letsencrypt.org/directory';

export class LetsEncryptProvider implements CertificateProvider {
    public readonly name = 'letsencrypt';
    
    private accountKey: crypto.KeyObject | null = null;
    private accountUrl: string | null = null;
    private directory: any = null;
    
    // HTTP-01 challenge tokens (populated during challenge)
    private pendingChallenges: Map<string, string> = new Map();
    
    constructor() {
        // Empty constructor - Web4 P6
    }
    
    public canHandle(config: DomainConfig): boolean {
        return config.certificateSource === 'letsencrypt';
    }
    
    /**
     * Request new certificate
     * ACME v2 flow: newAccount → newOrder → authorizations → finalize
     */
    public async certificateRequest(config: DomainConfig): Promise<DomainCertificate> {
        const acmeUrl = config.acmeStaging ? ACME_STAGING : ACME_PRODUCTION;
        
        // 1. Fetch directory
        this.directory = await this.directoryFetch(acmeUrl);
        
        // 2. Create or get account
        await this.accountEnsure(config.acmeEmail || 'admin@localhost');
        
        // 3. Create order
        const domains = config.isWildcard 
            ? [config.domain, config.domain.replace('*.', '')]
            : [config.domain];
        const order = await this.orderCreate(domains);
        
        // 4. Complete authorizations (challenges)
        for (const authUrl of order.authorizations) {
            await this.authorizationComplete(authUrl, config);
        }
        
        // 5. Finalize order with CSR
        const { privateKey, csr } = this.csrGenerate(domains);
        const certificate = await this.orderFinalize(order.finalize, csr);
        
        // 6. Build result
        return {
            domain: config.domain,
            altNames: domains.filter(d => d !== config.domain),
            certificate: certificate.cert,
            privateKey: privateKey,
            caChain: certificate.chain,
            expiresAt: this.expiryExtract(certificate.cert),
            issuedAt: new Date().toISOString(),
            issuer: "Let's Encrypt",
            isSelfSigned: false,
            fingerprint: this.fingerprintCompute(certificate.cert)
        };
    }
    
    // ... Additional ACME methods (directoryFetch, accountEnsure, etc.)
    // Implementation follows RFC 8555
    
    /**
     * Get pending HTTP-01 challenge token
     * Called by ACMEChallengeRoute
     */
    public challengeTokenGet(token: string): string | undefined {
        return this.pendingChallenges.get(token);
    }
    
    /**
     * Set pending challenge (called during authorization)
     */
    public challengeTokenSet(token: string, keyAuth: string): void {
        this.pendingChallenges.set(token, keyAuth);
    }
}
```

### **ACMEChallengeRoute.ts**

```typescript
// layer2/ACMEChallengeRoute.ts

/**
 * ACMEChallengeRoute - Serves ACME HTTP-01 challenges
 * 
 * Matches: GET /.well-known/acme-challenge/{token}
 * Returns: Key authorization for the token
 * 
 * @component CertificateManager
 * @layer 2
 */

import { IncomingMessage, ServerResponse } from 'http';
import { Route } from './Route.js';
import type { LetsEncryptProvider } from './LetsEncryptProvider.js';

export class ACMEChallengeRoute extends Route {
    private provider: LetsEncryptProvider | null = null;
    
    constructor() {
        super();
        this.model.name = 'ACMEChallengeRoute';
        this.model.pattern = '/.well-known/acme-challenge/**';
        this.model.priority = 1;  // Highest priority
    }
    
    public providerSet(provider: LetsEncryptProvider): this {
        this.provider = provider;
        return this;
    }
    
    public override iconGet(): string {
        return '🔐';
    }
    
    public override labelGet(): string {
        return '🔐 ACME Challenges';
    }
    
    protected override async handleRequest(
        req: IncomingMessage, 
        res: ServerResponse
    ): Promise<void> {
        const url = req.url || '';
        const token = url.split('/').pop() || '';
        
        if (!this.provider) {
            res.writeHead(500, { 'Content-Type': 'text/plain' });
            res.end('ACME provider not configured');
            return;
        }
        
        const keyAuth = this.provider.challengeTokenGet(token);
        
        if (keyAuth) {
            res.writeHead(200, { 'Content-Type': 'text/plain' });
            res.end(keyAuth);
        } else {
            res.writeHead(404, { 'Content-Type': 'text/plain' });
            res.end('Challenge not found');
        }
    }
}
```

```typescript
// layer2/CertificateRenewalScheduler.ts

/**
 * CertificateRenewalScheduler - Schedules automatic certificate renewal
 * 
 * Runs periodic checks for expiring certificates.
 * Triggers renewal X days before expiry (default: 30).
 * 
 * @component CertificateManager
 * @layer 2
 * Web4 P4: No arrow functions (uses bound methods for intervals)
 */

import { DomainCertificate } from '../layer3/DomainCertificate.interface.js';
import { DomainConfig } from '../layer3/DomainConfig.interface.js';

export interface RenewalTask {
    config: DomainConfig;
    certificate: DomainCertificate;
    scheduledAt: string;
}

export type RenewalCallback = (task: RenewalTask) => Promise<void>;

export class CertificateRenewalScheduler {
    private checkIntervalMs: number = 12 * 60 * 60 * 1000;  // 12 hours
    private renewBeforeDays: number = 30;
    private intervalId: NodeJS.Timeout | null = null;
    private certificates: Map<string, { config: DomainConfig; cert: DomainCertificate }> = new Map();
    private onRenewalNeeded: RenewalCallback | null = null;
    
    constructor() {
        // Empty constructor - Web4 P6
    }
    
    /**
     * Set callback for when renewal is needed
     */
    public onRenewalNeededSet(callback: RenewalCallback): void {
        this.onRenewalNeeded = callback;
    }
    
    /**
     * Register certificate for monitoring
     */
    public certificateRegister(config: DomainConfig, cert: DomainCertificate): void {
        this.certificates.set(config.domain, { config, cert });
    }
    
    /**
     * Start the scheduler
     */
    public start(): void {
        if (this.intervalId) return;
        
        // Initial check
        this.expiringCertificatesCheck();
        
        // Schedule periodic checks
        this.intervalId = setInterval(
            this.expiringCertificatesCheck.bind(this),
            this.checkIntervalMs
        );
        
        console.log('[CertificateRenewalScheduler] Started');
    }
    
    /**
     * Stop the scheduler
     */
    public stop(): void {
        if (this.intervalId) {
            clearInterval(this.intervalId);
            this.intervalId = null;
        }
        console.log('[CertificateRenewalScheduler] Stopped');
    }
    
    /**
     * Check for expiring certificates
     * Web4 P4: Bound method for interval callback
     */
    private expiringCertificatesCheck(): void {
        const now = Date.now();
        const threshold = now + (this.renewBeforeDays * 24 * 60 * 60 * 1000);
        
        for (const [domain, { config, cert }] of this.certificates) {
            const expiresAt = new Date(cert.expiresAt).getTime();
            
            if (expiresAt < threshold) {
                console.log(`[CertificateRenewalScheduler] Certificate expiring soon: ${domain}`);
                this.renewalTrigger(config, cert);
            }
        }
    }
    
    /**
     * Trigger renewal callback
     */
    private renewalTrigger(config: DomainConfig, cert: DomainCertificate): void {
        if (!this.onRenewalNeeded) return;
        
        const task: RenewalTask = {
            config,
            certificate: cert,
            scheduledAt: new Date().toISOString()
        };
        
        this.onRenewalNeeded(task).catch(this.renewalErrorHandle.bind(this, config.domain));
    }
    
    private renewalErrorHandle(domain: string, error: Error): void {
        console.error(`[CertificateRenewalScheduler] Renewal failed for ${domain}:`, error.message);
    }
}
```

### **CertificateOrchestrator.ts**

```typescript
// layer4/CertificateOrchestrator.ts

/**
 * CertificateOrchestrator - High-level certificate lifecycle management
 * 
 * Coordinates:
 * - Certificate providers (LetsEncrypt, manual)
 * - SNI manager (hot-reload)
 * - Renewal scheduler
 * - Persistence (scenario storage)
 * 
 * Exposes IOR methods for management.
 * 
 * @component CertificateManager
 * @layer 4
 * Web4 P7: async only in Layer 4
 */

import { SNIManager } from '../layer2/SNIManager.js';
import { LetsEncryptProvider } from '../layer2/LetsEncryptProvider.js';
import { CertificateRenewalScheduler, RenewalTask } from '../layer2/CertificateRenewalScheduler.js';
import { DomainConfig } from '../layer3/DomainConfig.interface.js';
import { DomainCertificate } from '../layer3/DomainCertificate.interface.js';
import { Scenario } from '../layer3/Scenario.interface.js';

export interface CertificateOrchestratorModel {
    uuid: string;
    name: string;
    domains: DomainConfig[];
    certificates: { [domain: string]: DomainCertificate };
}

export class CertificateOrchestrator {
    public model: CertificateOrchestratorModel;
    
    private sniManager: SNIManager;
    private letsEncryptProvider: LetsEncryptProvider;
    private scheduler: CertificateRenewalScheduler;
    
    constructor() {
        this.model = {
            uuid: '',
            name: 'CertificateOrchestrator',
            domains: [],
            certificates: {}
        };
        
        this.sniManager = new SNIManager();
        this.letsEncryptProvider = new LetsEncryptProvider();
        this.scheduler = new CertificateRenewalScheduler();
        
        // Wire up renewal callback
        this.scheduler.onRenewalNeededSet(this.renewalHandle.bind(this));
    }
    
    /**
     * Initialize with scenario
     */
    public async init(scenario: Scenario<CertificateOrchestratorModel>): Promise<this> {
        if (scenario?.model) {
            Object.assign(this.model, scenario.model);
        }
        
        // Load existing certificates into SNI manager
        for (const [domain, cert] of Object.entries(this.model.certificates)) {
            this.sniManager.certificateSet(domain, cert);
            const config = this.model.domains.find(d => d.domain === domain);
            if (config) {
                this.scheduler.certificateRegister(config, cert);
            }
        }
        
        return this;
    }
    
    /**
     * Get SNI callback for HTTPSServer
     */
    public sniCallbackGet(): (servername: string, cb: any) => void {
        return this.sniManager.sniCallback.bind(this.sniManager);
    }
    
    // ═══════════════════════════════════════════════════════════════
    // IOR Methods (exposed via IORRoute)
    // ═══════════════════════════════════════════════════════════════
    
    /**
     * IOR: Request certificate for domain
     * ior:esm:/ONCE/0.3.21.8/{uuid}/certificateRequest
     */
    public async certificateRequest(config: DomainConfig): Promise<DomainCertificate> {
        console.log(`[CertificateOrchestrator] Requesting certificate for: ${config.domain}`);
        
        const cert = await this.letsEncryptProvider.certificateRequest(config);
        
        // Update SNI manager (hot-reload)
        this.sniManager.certificateSet(config.domain, cert);
        
        // Store in model
        this.model.certificates[config.domain] = cert;
        if (!this.model.domains.find(d => d.domain === config.domain)) {
            this.model.domains.push(config);
        }
        
        // Register for renewal
        this.scheduler.certificateRegister(config, cert);
        
        console.log(`[CertificateOrchestrator] Certificate issued for: ${config.domain}`);
        return cert;
    }
    
    /**
     * IOR: Add domain configuration
     * ior:esm:/ONCE/0.3.21.8/{uuid}/domainAdd
     */
    public async domainAdd(config: DomainConfig): Promise<void> {
        this.model.domains.push(config);
        
        if (config.certificateSource === 'letsencrypt') {
            await this.certificateRequest(config);
        }
    }
    
    /**
     * IOR: List all domains
     * ior:esm:/ONCE/0.3.21.8/{uuid}/domainsGet
     */
    public async domainsGet(): Promise<DomainConfig[]> {
        return this.model.domains;
    }
    
    /**
     * IOR: Get certificate status
     * ior:esm:/ONCE/0.3.21.8/{uuid}/certificateStatusGet
     */
    public async certificateStatusGet(domain: string): Promise<{ 
        domain: string; 
        expiresAt: string; 
        daysRemaining: number;
        issuer: string;
    } | null> {
        const cert = this.model.certificates[domain];
        if (!cert) return null;
        
        const expiresAt = new Date(cert.expiresAt);
        const now = new Date();
        const daysRemaining = Math.floor((expiresAt.getTime() - now.getTime()) / (24 * 60 * 60 * 1000));
        
        return {
            domain,
            expiresAt: cert.expiresAt,
            daysRemaining,
            issuer: cert.issuer
        };
    }
    
    /**
     * Start lifecycle management
     */
    public async start(): Promise<void> {
        this.scheduler.start();
        console.log('[CertificateOrchestrator] Started');
    }
    
    /**
     * Stop lifecycle management
     */
    public async stop(): Promise<void> {
        this.scheduler.stop();
        console.log('[CertificateOrchestrator] Stopped');
    }
    
    /**
     * Handle renewal request
     * Web4 P4: Bound method for callback
     */
    private async renewalHandle(task: RenewalTask): Promise<void> {
        console.log(`[CertificateOrchestrator] Renewing certificate for: ${task.config.domain}`);
        
        const newCert = await this.letsEncryptProvider.certificateRenew(
            task.config, 
            task.certificate
        );
        
        // Hot-reload into SNI manager
        this.sniManager.certificateSet(task.config.domain, newCert);
        
        // Update stored certificate
        this.model.certificates[task.config.domain] = newCert;
        
        // Re-register with new expiry
        this.scheduler.certificateRegister(task.config, newCert);
        
        console.log(`[CertificateOrchestrator] Certificate renewed for: ${task.config.domain}`);
    }
}
```
