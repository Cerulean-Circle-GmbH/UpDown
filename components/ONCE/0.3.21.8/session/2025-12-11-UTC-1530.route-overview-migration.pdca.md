## **📝 PHASE RO.1: Create RouteItemView Component**

**File:** `layer5/views/RouteItemView.ts`

**Key Points:**
- Uses **SPA RouteModel** from `RouteScenario.interface.ts`
- Extends `DefaultItemView<RouteModel>`
- Web4 P4: Object method references (no arrow functions)

```typescript
/**
 * RouteItemView.ts - Route Item View Component (SPA Routes)
 * 
 * Uses SPA RouteModel from RouteScenario.interface.ts
 * NOT HTTP RouteModel from RouteModel.interface.ts!
 * 
 * @component Router
 * @layer 5
 */

import { html, TemplateResult } from 'lit';
import { customElement, property } from 'lit/decorators.js';
import { UcpView } from './UcpView.js';
import type { RouteModel } from '../../layer3/RouteScenario.interface.js';
import type { Reference } from '../../layer3/Reference.interface.js';

@customElement('route-item-view')
export class RouteItemView extends UcpView<RouteModel> {
  
  static cssPath = 'RouteItemView.css';
  
  /** Router reference for navigation */
  @property({ attribute: false }) router: Reference<HTMLElement> = null;
  
  // Computed accessors (Web4 P16)
  get path(): string { return this.model?.pattern ?? ''; }
  get routeTitle(): string { 
    if (this.model?.title) return this.model.title;
    if (this.path === '/') return 'Server Status';
    return this.path.slice(1).charAt(0).toUpperCase() + this.path.slice(2);
  }
  get isActive(): boolean { return this.model?.isActive ?? false; }
  
  // Navigation (Web4 P4: method references)
  private linkClickHandle(event: Event): void {
    event.preventDefault();
    if (this.router && (this.router as any).navigateTo) {
      (this.router as any).navigateTo(this.path, false);
    } else {
      window.history.pushState({}, '', this.path);
      window.dispatchEvent(new PopStateEvent('popstate'));
    }
  }
  
  private linkClickHandlerGet(): (event: Event) => void {
    return this.linkClickHandle.bind(this);
  }
  
  render(): TemplateResult {
    return html`
      <div class="route-item ${this.isActive ? 'active' : ''}">
        <a href="${this.path}" @click=${this.linkClickHandlerGet()}>
          <strong>${this.routeTitle}</strong>
          <code>${this.path}</code>
        </a>
      </div>
    `;
  }
}
```

```typescript
/** Router reference - set by UcpRouter when creating view */
@property({ attribute: false }) router: Reference<HTMLElement> = null;

private routesSectionRender(): TemplateResult {
  // Get routes from router
  const routes = this.routesGet();
  
  return html`
    <div class="endpoints-section">
      <h3><span class="emoji">📡</span> Available Routes</h3>
      ${routes.map(this.routeItemRender.bind(this))}
    </div>
    ...
  `;
}

/**
 * Get registered routes from router
 * Web4 P4: Method, not arrow function
 */
private routesGet(): RouteModel[] {
  if (!this.router) return [];
  const routerAny = this.router as any;
  if (!routerAny.routesGet) return [];
  
  const routeMap = routerAny.routesGet() as Map<string, any>;
  const routes: RouteModel[] = [];
  
  // Convert Route instances to RouteModel for view
  routeMap.forEach(this.routeModelExtract.bind(this, routes));
  return routes;
}

/**
 * Extract RouteModel from Route instance
 * Web4 P4: Method for forEach callback
 */
private routeModelExtract(routes: RouteModel[], route: any): void {
  routes.push(route.model);
}

/**
 * Render single route using RouteItemView
 * Web4 P4: Method for map callback
 */
private routeItemRender(routeModel: RouteModel): TemplateResult {
  return html`
    <route-item-view
      .model=${routeModel}
      .router=${this.router}
    ></route-item-view>
  `;
}
```
