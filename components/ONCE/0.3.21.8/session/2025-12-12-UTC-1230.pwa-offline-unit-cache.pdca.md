## **🔗 KEY CONCEPT: Unit = Cacheable Asset with IOR**

### **Unit as Cache Entry**

```typescript
interface Unit {
  uuid: string;           // Unique ID
  ior: string;            // /EAMD.ucp/components/ONCE/0.3.21.8/dist/...
  type: UnitType;         // 'js' | 'css' | 'html' | 'json' | 'scenario'
  version: SemanticVersion;
  hash: string;           // Content hash for cache invalidation
  size: number;           // Bytes
  dependencies: string[]; // Other Unit IORs this depends on
}
```

### **Route → Unit Mapping**

Every StaticFileRoute serves a file that is tracked as a Unit:

```
StaticFileRoute pattern: /EAMD.ucp/components/ONCE/0.3.21.8/**
                              │
                              ▼
Request: GET /EAMD.ucp/components/ONCE/0.3.21.8/dist/ts/layer1/ONCE.js
                              │
                              ▼
Unit IOR: /EAMD.ucp/components/ONCE/0.3.21.8/dist/ts/layer1/ONCE.js
                              │
                              ▼
Unit: {
  uuid: "abc123...",
  ior: "/EAMD.ucp/components/ONCE/0.3.21.8/dist/ts/layer1/ONCE.js",
  type: "js",
  version: "0.3.21.8",
  hash: "sha256-xyz...",
  dependencies: ["./DefaultONCE.js", "./ONCE.interface.js"]
}
```

### **Cache Strategy: Unit Versioning**

```
Unit v0.3.21.7 (cached)   vs   Unit v0.3.21.8 (server)
                              │
                              ▼
Version mismatch detected → Fetch new, update cache, notify UI
```

### **Expose Units via /units Endpoint**

```typescript
// In NodeJsOnce.ts routes registration

/**
 * GET /units - Returns all cacheable Units
 * Service Worker fetches this to know what to cache
 * 
 * Web4 P4: Method reference
 * JsInterface: Units ARE their type (no enum)
 */
httpRouter.registerRoute(new ScenarioRoute().init({
  model: {
    pattern: '/units',
    method: HttpMethod.GET,
    priority: 100
  },
  handler: this.unitsEndpointHandle.bind(this)
}));

private unitsEndpointHandle(req: IncomingMessage, res: ServerResponse): void {
  const units = this.unitRegistryGet();
  res.writeHead(200, { 'Content-Type': 'application/json' });
  res.end(JSON.stringify({
    version: this.model.version,
    units: units.map(this.unitSerialize.bind(this))
  }));
}

private unitSerialize(unit: Unit): object {
  return {
    uuid: unit.uuid,
    ior: unit.ior,
    type: unit.constructor.name,  // JsInterface pattern
    version: unit.version.toString(),
    hash: unit.hash,
    size: unit.size,
    icon: unit.iconGet(),         // Polymorphism
    label: unit.labelGet()        // Polymorphism
  };
}
```

### **UnitRegistry Integration**

```typescript
// layer2/UnitRegistry.ts

class UnitRegistry extends UcpComponent<UnitRegistryModel> {
  
  private units: Map<string, Unit> = new Map();  // IOR → Unit
  
  /**
   * Get all registered Units
   * Web4 P16: unitsGet() accessor pattern
   */
  public unitsGet(): Unit[] {
    const result: Unit[] = [];
    this.units.forEach(this.unitCollect.bind(this, result));
    return result;
  }
  
  private unitCollect(result: Unit[], unit: Unit): void {
    result.push(unit);
  }
  
  /**
   * Register Unit from Route
   * Creates Unit for each static file served
   */
  public unitFromRouteRegister(route: StaticFileRoute, filePath: string): Unit {
    const unit = new CSSUnit().init({
      model: {
        uuid: this.uuidGenerate(),
        ior: route.iorFromPath(filePath),
        hash: this.hashCompute(filePath),
        version: this.version
      }
    });
    this.units.set(unit.ior, unit);
    return unit;
  }
}
```

### **OnceServiceWorker Component**

```typescript
// layer2/OnceServiceWorker.ts

/**
 * OnceServiceWorker - ONCE kernel for Service Worker context
 * 
 * Manages Unit-based caching with IOR keys.
 * Follows Web4 Radical OOP - no arrow functions.
 * 
 * @component ONCE
 * @layer 2
 */

export class OnceServiceWorker extends UcpComponent<ServiceWorkerModel> {
  
  private unitCache: Map<string, CachedUnit> = new Map();
  private cacheVersion: string = '0.3.21.8';
  private cacheName: string = `once-cache-${this.cacheVersion}`;
  
  // ═══════════════════════════════════════════════════════════════
  // LIFECYCLE (Web4 P4: Method references)
  // ═══════════════════════════════════════════════════════════════
  
  /**
   * Install event - pre-cache critical Units
   */
  public installHandle(event: ExtendableEvent): void {
    console.log('[SW] Installing...');
    event.waitUntil(this.criticalUnitsCache());
  }
  
  /**
   * Activate event - clean old caches
   */
  public activateHandle(event: ExtendableEvent): void {
    console.log('[SW] Activating...');
    event.waitUntil(this.oldCachesClean());
  }
  
  // fetchHandle is defined below in IOR-BASED COMMUNICATION section
  // (handles both IOR calls AND cacheable static assets)
  
  // ═══════════════════════════════════════════════════════════════
  // CACHE STRATEGY (Radical OOP - methods, not arrow functions)
  // ═══════════════════════════════════════════════════════════════
  
  /**
   * Check if request should be cached
   */
  private isCacheableRequest(url: URL): boolean {
    // Cache static assets from EAMD.ucp
    if (url.pathname.startsWith('/EAMD.ucp/')) return true;
    // Cache component assets
    if (url.pathname.match(/\/components\/.*\/dist\//)) return true;
    return false;
  }
  
  /**
   * Get cached response or fetch and cache
   * Strategy: Cache-first with network fallback
   */
  private async cachedResponseGet(request: Request): Promise<Response> {
    const cache = await caches.open(this.cacheName);
    const ior = this.iorFromRequest(request);
    
    // Check cache first
    const cached = await cache.match(request);
    if (cached) {
      console.log('[SW] Cache hit:', ior);
      // Background refresh (stale-while-revalidate)
      this.backgroundRefresh(request, cache);
      return cached;
    }
    
    // Not cached - fetch from network
    console.log('[SW] Cache miss, fetching:', ior);
    try {
      const response = await fetch(request);
      if (response.ok) {
        // Cache the response
        await cache.put(request, response.clone());
        // Create Unit entry
        this.unitFromResponse(ior, response);
      }
      return response;
    } catch (err) {
      console.warn('[SW] Network failed, no cache:', ior);
      return new Response('Offline - resource not cached', { status: 503 });
    }
  }
  
  /**
   * Background refresh - update cache without blocking
   */
  private backgroundRefresh(request: Request, cache: Cache): void {
    fetch(request)
      .then(this.cacheUpdateHandle.bind(this, request, cache))
      .catch(function(err) {
        console.warn('[SW] Background refresh failed:', err);
      });
  }
  
  /**
   * Handle cache update from background refresh
   */
  private cacheUpdateHandle(request: Request, cache: Cache, response: Response): void {
    if (response.ok) {
      cache.put(request, response.clone());
    }
  }
  
  /**
   * Extract IOR from request
   */
  private iorFromRequest(request: Request): string {
    return new URL(request.url).pathname;
  }
  
  /**
   * Create Unit from fetched response
   */
  private unitFromResponse(ior: string, response: Response): void {
    const contentType = response.headers.get('content-type') || '';
    const cachedUnit: CachedUnit = {
      ior: ior,
      type: this.typeFromContentType(contentType),
      cachedAt: Date.now(),
      size: parseInt(response.headers.get('content-length') || '0', 10)
    };
    this.unitCache.set(ior, cachedUnit);
  }
  
  /**
   * Detect Unit type from content-type header
   */
  private typeFromContentType(contentType: string): string {
    if (contentType.includes('javascript')) return 'js';
    if (contentType.includes('css')) return 'css';
    if (contentType.includes('html')) return 'html';
    if (contentType.includes('json')) return 'json';
    return 'other';
  }
  
  // ═══════════════════════════════════════════════════════════════
  // PRE-CACHING
  // ═══════════════════════════════════════════════════════════════
  
  /**
   * Pre-cache critical Units during install
   */
  private async criticalUnitsCache(): Promise<void> {
    const cache = await caches.open(this.cacheName);
    
    // Fetch Unit list from server
    try {
      const response = await fetch('/units');
      const data = await response.json();
      const criticalUrls = data.units
        .filter(this.isCriticalUnit.bind(this))
        .map(this.urlFromUnit.bind(this));
      
      await cache.addAll(criticalUrls);
      console.log('[SW] Pre-cached', criticalUrls.length, 'critical Units');
    } catch (err) {
      console.warn('[SW] Failed to pre-cache:', err);
    }
  }
  
  /**
   * Check if Unit is critical (must be pre-cached)
   */
  private isCriticalUnit(unit: any): boolean {
    // Critical: main entry points, core framework
    return unit.ior.includes('ONCE.js') ||
           unit.ior.includes('index.html') ||
           unit.ior.includes('main.css');
  }
  
  /**
   * Get URL from Unit for caching
   */
  private urlFromUnit(unit: any): string {
    return unit.ior;
  }
  
  // ═══════════════════════════════════════════════════════════════
  // CACHE CLEANUP
  // ═══════════════════════════════════════════════════════════════
  
  /**
   * Clean old version caches
   */
  private async oldCachesClean(): Promise<void> {
    const keys = await caches.keys();
    const deletePromises = keys
      .filter(this.isOldCache.bind(this))
      .map(this.cacheDelete.bind(this));
    
    await Promise.all(deletePromises);
  }
  
  private isOldCache(cacheName: string): boolean {
    return cacheName.startsWith('once-cache-') && cacheName !== this.cacheName;
  }
  
  private cacheDelete(cacheName: string): Promise<boolean> {
    console.log('[SW] Deleting old cache:', cacheName);
    return caches.delete(cacheName);
  }
  
  // ═══════════════════════════════════════════════════════════════
  // IOR-BASED COMMUNICATION (Web4 Radical OOP - No Messages!)
  // ═══════════════════════════════════════════════════════════════
  
  /**
   * SW exposes IOR endpoints via fetch interception
   * 
   * Web4 Radical OOP:
   * ❌ postMessage({ type: 'scenario-sync', ... })  // Message + switch = anti-pattern
   * ✅ fetch('/ONCE/sw-kernel/scenarioReceive')     // IOR method call via REST
   * 
   * The Scenario IS the object - no message wrapper needed!
   */
  
  /**
   * Extended fetch handler - intercepts IOR calls to SW kernel
   * Pattern: /ONCE/{version}/{sw-uuid}/{method}
   */
  public fetchHandle(event: FetchEvent): void {
    const url = new URL(event.request.url);
    
    // Check if this is an IOR call to the SW kernel
    if (this.isKernelIOR(url)) {
      event.respondWith(this.iorMethodHandle(event.request, url));
      return;
    }
    
    // Standard cache handling for static assets
    if (this.isCacheableRequest(url)) {
      event.respondWith(this.cachedResponseGet(event.request));
    }
  }
  
  /**
   * Check if URL is an IOR call to this SW kernel
   */
  private isKernelIOR(url: URL): boolean {
    // Pattern: /ONCE/{version}/{sw-uuid}/{method}
    return url.pathname.includes(this.model.uuid);
  }
  
  /**
   * Handle IOR method calls to SW kernel
   * Each method is a REST endpoint - no switch on message types!
   */
  private async iorMethodHandle(request: Request, url: URL): Promise<Response> {
    const pathParts = url.pathname.split('/');
    const method = pathParts[pathParts.length - 1];
    
    // Dynamic method dispatch via IOR
    // The method name IS the action - Radical OOP!
    const handler = (this as any)[method];
    if (typeof handler === 'function') {
      const body = request.method === 'POST' 
        ? await request.json() 
        : {};
      const result = await handler.call(this, body);
      return new Response(JSON.stringify(result), {
        headers: { 'Content-Type': 'application/json' }
      });
    }
    
    return new Response('Method not found', { status: 404 });
  }
  
  // ═══════════════════════════════════════════════════════════════
  // IOR METHODS (Exposed as REST endpoints)
  // ═══════════════════════════════════════════════════════════════
  
  /**
   * IOR: /ONCE/{version}/{sw-uuid}/scenarioReceive
   * POST body: Scenario object
   * 
   * Receives scenario from any kernel (main thread, worker, peer)
   */
  public scenarioReceive(scenario: Scenario<any>): { status: string } {
    console.log('[SW] IOR scenarioReceive:', scenario.uuid);
    this.scenarioService.scenarioStore(scenario);
    return { status: 'stored' };
  }
  
  /**
   * IOR: /ONCE/{version}/{sw-uuid}/scenariosGet
   * GET - returns all scenarios in SW kernel
   */
  public scenariosGet(): Scenario<any>[] {
    return this.scenarioService.scenariosGet();
  }
  
  /**
   * IOR: /ONCE/{version}/{sw-uuid}/componentLoad
   * POST body: { ior: string }
   */
  public async componentLoad(params: { ior: string }): Promise<{ status: string }> {
    const module = await import(params.ior);
    console.log('[SW] Loaded component:', params.ior);
    
    if (module.default?.start) {
      module.default.start();
    }
    return { status: 'loaded' };
  }
  
  /**
   * IOR: /ONCE/{version}/{sw-uuid}/cacheInvalidate
   * POST body: { pattern: string }
   */
  public async cacheInvalidate(params: { pattern: string }): Promise<{ status: string, deleted: number }> {
    const cache = await caches.open(this.cacheName);
    const keys = await cache.keys();
    let deleted = 0;
    
    for (const request of keys) {
      if (request.url.includes(params.pattern)) {
        await cache.delete(request);
        deleted++;
      }
    }
    
    return { status: 'invalidated', deleted };
  }
}
```

## **📝 PHASE PWA.3c: Multi-Kernel IOR Communication (Radical OOP)**

### **Web4 Principle: No Messages, Only IOR/REST**

```
❌ WRONG: Message-based communication
   postMessage({ type: 'scenario-sync', scenario: ... })
   switch (message.type) { ... }  // Anti-pattern!

✅ CORRECT: IOR-based communication (REST)
   fetch('/ONCE/0.3.21.8/{sw-uuid}/scenarioReceive', {
     method: 'POST',
     body: JSON.stringify(scenario)  // Scenario IS the object!
   })
```

**Why?**
- The Scenario IS the object - no message wrapper needed
- Each method is an IOR endpoint - REST semantics
- No switch statements on "message type" - Polymorphism!
- SW intercepts fetch → same pattern as server IOR routes

```typescript
    throw new Error('IOR call failed: ' + response.status);
  }

  return fetch(ior)
    .then(function(r) { return r.json(); });
}
```
