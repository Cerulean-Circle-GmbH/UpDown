### ✅ B. Design Tootsie Delegation Pattern

#### Proposed Flow (After DelegationProxy Upgrade):
```
User → once tootsie file → DelegationProxy → Web4TSComponent.tootsie() → Tootsie.testFile()
```

**Automatic Delegation** (modern pattern):
1. **ONCECLI** - Wraps NodeJsOnce in DelegationProxy (upgrade needed!)
2. **DelegationProxy** - Automatically intercepts `tootsie` method → delegates to Web4TSComponent
3. **Web4TSComponent** - Add `tootsie()` method that loads and invokes Tootsie
4. **Tootsie** - Quality consciousness implementation

**NO CODE NEEDED IN NodeJsOnce!** (Proxy handles everything automatically)

### ✅ C. Implementation Architecture

#### C.1: ONCE Layer (ONCECLI) - UPGRADE TO DelegationProxy

**Replace manual delegation with modern DelegationProxy pattern:**

```typescript
// components/ONCE/0.3.21.8/src/ts/layer5/ONCECLI.ts

import { DelegationProxy } from '../layer2/DelegationProxy.js';  // ADD THIS

export class ONCECLI extends DefaultCLI {
  /**
   * Empty constructor - initialize component asynchronously
   */
  constructor() {
    super();
    this.init();
    // Component initialized in initComponent() now
  }
  
  /**
   * Initialize component with DelegationProxy (modern pattern)
   * @pdca 2025-12-02-UTC-1115.iteration-10.4-tootsie-cli-delegation-pattern.pdca.md
   */
  async initComponent(): Promise<void> {
    // STEP 1: Create component
    const component = new NodeJsOnce();
    
    // STEP 2: Set Path Authority
    component.model.projectRoot = this.model.projectRoot;
    component.model.targetDirectory = this.model.projectRoot;
    component.model.isTestIsolation = this.model.projectRoot?.includes('/test/data') || false;
    
    // STEP 3: Pre-load Web4TSComponent for delegation
    await (component as any).getWeb4TSComponent();
    
    // STEP 4: Wrap with DelegationProxy
    this.component = await DelegationProxy.start(component as any) as any;
    
    // STEP 5: Discover methods
    this.discoverMethods();
    
    // STEP 6: Discover component's own methods
    if (this.component) {
      const ownMethods = this.component.listMethods();
      for (const methodName of ownMethods) {
        const signature = this.component.getMethodSignature(methodName);
        if (signature) {
          this.methodSignatures.set(methodName, signature);
        }
      }
    }
    
    // STEP 7: Discover delegated methods
    if (this.component && (this.component as any).hasDelegation()) {
      const delegationTarget = (this.component as any).getDelegationTarget();
      if (delegationTarget) {
        const delegatedMethods = delegationTarget.listMethods();
        for (const methodName of delegatedMethods) {
          const signature = delegationTarget.getMethodSignature(methodName);
          if (signature) {
            signature.isDelegated = true;
            this.methodSignatures.set(methodName, signature);
          }
        }
      }
    }
  }
  
  /**
   * Static start method
   */
  static async start(args: string[]): Promise<void> {
    const cli = new ONCECLI();
    await cli.initComponent();  // NEW: Initialize asynchronously
    await cli.execute(args);
  }
  
  /**
   * Override shCompletion to ensure component is initialized
   */
  async shCompletion(cword: string, ...words: string[]): Promise<void> {
    if (!this.component) {
      await this.initComponent();
    }
    await super.shCompletion(cword, ...words);
  }
}
```

```typescript
// components/Web4TSComponent/0.3.20.6/src/ts/layer2/DefaultWeb4TSComponent.ts

/**
 * Tootsie: Quality consciousness testing
 * Loads Tootsie component and delegates quality assessment
 * 
 * @param command - Tootsie command (file, describe, assess, oracle)
 * @param args - Command arguments
 */
async tootsie(command?: string, ...args: string[]): Promise<this> {
  const tootsieVersion = '0.3.20.6'; // Or discover from latest symlink
  
  // Load Tootsie component dynamically
  const TootsieClass = await this.componentLoad('Tootsie', tootsieVersion);
  const tootsie = new TootsieClass();
  
  // Initialize Tootsie with component context
  const tootsieScenario = {
    targetComponent: this.model.name,
    targetVersion: this.model.version,
    projectRoot: this.model.projectRoot,
    componentRoot: this.model.componentRoot,
    command,
    args
  };
  
  await tootsie.init(tootsieScenario);
  
  // Delegate to Tootsie based on command
  switch (command) {
    case 'file':
      await tootsie.testFile(args[0]);
      break;
    case 'describe':
      await tootsie.testDescribe(args[0]);
      break;
    case 'assess':
      await tootsie.assessTest(args[0]);
      break;
    case 'oracle':
      await tootsie.oracle(args[0], ...args.slice(1));
      break;
    default:
      await tootsie.usage();
  }
  
  return this;
}
```

#### C.3: Tootsie Layer (DefaultTootsie)

```typescript
// components/Tootsie/0.3.20.6/src/ts/layer2/DefaultTootsie.ts

/**
 * Test a specific file with quality consciousness
 * 
 * @param testFile - Test file name or path
 */
async testFile(testFile: string): Promise<this> {
  console.log(`🎯 Tootsie: Running test file with quality assessment: ${testFile}`);
  
  // 1. Awaken Quality Oracle
  const oracle = await this.awakenOracle();
  
  // 2. Load test object (not a function!)
  const testClass = await this.loadTest(testFile);
  const test = new testClass();
  const scenario = this.createTestScenario(testFile);
  test.init(scenario);
  
  // 3. Execute test
  console.log(`🚀 Executing test: ${test.getUUID()}`);
  const execution = await test.execute();
  
  // 4. Oracle assesses quality
  console.log(`⚖️ Quality Oracle assessing...`);
  const judgment = await oracle.judge(execution, this.model.qualityScenario);
  
  // 5. Oracle learns from evidence
  for (const evidence of execution.evidenceScenarios) {
    await oracle.learn(this.convertToQualityEvidence(evidence));
  }
  
  // 6. Oracle evolves
  await oracle.evolve();
  
  // 7. Hibernate state for next run
  await this.hibernateState(test, oracle, judgment);
  
  // 8. Report results
  this.reportResults(execution, judgment, oracle);
  
  return this;
}

/**
 * Quality Oracle operations
 */
async oracle(operation: string, ...args: string[]): Promise<this> {
  const oracle = await this.restoreOracle() || await this.awakenOracle();
  
  switch (operation) {
    case 'evolve':
      await oracle.evolve();
      console.log(`✨ Oracle evolved to consciousness level: ${await oracle.getConsciousnessLevel()}`);
      break;
    case 'wisdom':
      console.log(`🧠 Oracle wisdom level: ${await oracle.getWisdomLevel()}`);
      break;
    case 'history':
      const history = await oracle.getQualityHistory();
      console.log(`📚 Oracle has ${history.length} quality assessments in history`);
      break;
    default:
      console.log('🔮 Oracle operations: evolve, wisdom, history');
  }
  
  return this;
}
```
