## **P - PLAN**

### **1. Problem Statement**

**Current State**:
- `ItemView.ts` exists as a generic Lit component but doesn't implement a formal interface
- No `ItemView.interface.ts` exists - violates separation of interface/implementation
- Views are not registered as RelatedObjects in the controller
- Finding related infrastructure objects requires direct references (coupling)
- `DefaultCli` has direct delegation references that should be lookups

**Target State**:
- `OncePeerItemView` implements `ItemView.interface.ts`
- RelatedObjects Registry in `UcpController` enables interface-based lookup
- Components can find views, persistence managers, routers by interface
- Foundation for decoupling via lookup instead of direct references

### **2. User Requirements (TRON Feedback)**

### **3. Web4 Principles Applied**

| # | Principle | Application |
|---|-----------|-------------|
| P1 | Everything is a Scenario | RelatedObjects registry state as scenario |
| P4 | Radical OOP | No arrow functions, method references |
| P5 | Reference<T> | Nullable related objects |
| P6 | Empty Constructor | RelatedObjects init via init() |
| P7 | Async Layer 4 only | Registry operations sync (Layer 2) |
| P16 | Object-Action naming | `relatedObjectRegister()`, `relatedObjectLookup()` |
| P19 | One File One Type | ItemView.interface.ts separate from implementation |
| **P24** | **RelatedObjects Registry** | **NEW PRINCIPLE** |

### **4. NEW: Web4 Principle 24 - RelatedObjects Registry**

> **RelatedObjects Registry**: Every UcpComponent has a registry of related infrastructure objects
> (views, persistence managers, routers, CLI instances). Objects register themselves AND all
> interfaces they implement. Lookup by interface returns all matching instances.
>
> **Pattern**:
> ```typescript
> // Registration - registers for ALL implemented interfaces
> controller.relatedObjectRegister(componentInstance, OncePeerItemView, viewInstance);
> // Automatically also registers:
> // - (componentInstance, ItemView, viewInstance)
> // - (componentInstance, View, viewInstance)
>
> // Lookup - returns all matching instances
> const views = controller.relatedObjectLookup(View);        // All views
> const itemViews = controller.relatedObjectLookup(ItemView); // All ItemViews
> ```
>
> **Benefits**:
> - Decouples components from direct references
> - Enables discovery of related infrastructure
> - Foundation for replacing DefaultCli delegation
> - Supports polymorphic lookup by interface hierarchy

### **5.3 RelatedObjects Interface**

```typescript
// layer3/RelatedObjects.interface.ts
/**
 * RelatedObjects.interface.ts - Registry for related infrastructure objects
 * 
 * Web4 Principle 24: RelatedObjects Registry
 * 
 * Enables components to find related views, persistence managers, routers
 * by interface rather than direct references.
 * 
 * @ior ior:esm:/ONCE/{version}/RelatedObjects.interface
 */

/**
 * Constructor type for interface matching
 */
export type InterfaceConstructor<T = any> = new (...args: any[]) => T;

/**
 * RelatedObjects - Registry for infrastructure objects
 */
export interface RelatedObjects {
  
  /**
   * Register a related object and all its implemented interfaces
   * @param interfaceType The interface/class type to register under
   * @param instance The object instance
   */
  relatedObjectRegister<T>(interfaceType: InterfaceConstructor<T>, instance: T): void;
  
  /**
   * Lookup all objects implementing an interface
   * @param interfaceType The interface/class type to lookup
   * @returns Array of all matching instances
   */
  relatedObjectLookup<T>(interfaceType: InterfaceConstructor<T>): T[];
  
  /**
   * Lookup single object implementing an interface
   * @param interfaceType The interface/class type to lookup
   * @returns First matching instance or null
   */
  relatedObjectLookupFirst<T>(interfaceType: InterfaceConstructor<T>): T | null;
  
  /**
   * Unregister a related object from all interfaces
   * @param instance The object instance to unregister
   */
  relatedObjectUnregister(instance: any): void;
}
```

### **5.4 OncePeerItemView Implementation**

```typescript
// layer5/views/OncePeerItemView.ts
/**
 * OncePeerItemView.ts - ONCE Peer item view implementation
 * 
 * Concrete implementation of ItemView for ONCE peer display.
 * Registered in RelatedObjects as: OncePeerItemView, ItemView, View
 * 
 * Web4 Principles:
 * - P4: Radical OOP (no arrow functions)
 * - P19: CSS in separate file (OncePeerItemView.css)
 * - P24: RelatedObjects registration
 * 
 * @ior ior:esm:/ONCE/{version}/OncePeerItemView
 */

import { html, TemplateResult } from 'lit';
import { customElement, property } from 'lit/decorators.js';
import { UcpView } from './UcpView.js';
import { ItemView } from '../../layer3/ItemView.interface.js';
import { Action } from '../../layer3/Action.interface.js';
import { ActionStyle } from '../../layer3/ActionStyle.enum.js';
import { ONCEServerModel } from '../../layer3/ONCEServerModel.js';

/**
 * OncePeerItemView - Compact peer item for list/grid display
 * 
 * Shows:
 * - Status indicator (running/stopped)
 * - Peer UUID (truncated)
 * - Port
 * - Stop action button (for non-primary peers)
 */
@customElement('once-peer-item-view')
export class OncePeerItemView extends UcpView<ONCEServerModel> implements ItemView<ONCEServerModel> {
  
  /** CSS path for adoptedStyleSheets */
  static cssPath = 'OncePeerItemView.css';
  
  /** Touch/drag support enabled */
  @property({ type: Boolean }) touchEnabled = true;
  
  /** View name for CSS/template paths */
  get viewName(): string {
    return 'OncePeerItemView';
  }
  
  /**
   * Discover available actions
   * Returns actions with no required parameters
   */
  actionsDiscover(): Action[] {
    const model = this.hasModel ? this.model : null;
    const isPrimary = model?.isPrimaryServer || false;
    
    if (isPrimary) {
      return [];  // No actions for primary server
    }
    
    return [{
      method: 'peerStop',
      label: 'Stop',
      icon: '🛑',
      style: ActionStyle.DANGER,
      shortcut: null,
      confirmRequired: null,
      confirmMessage: null
    }];
  }
  
  /**
   * Render the peer item
   */
  render(): TemplateResult {
    if (!this.hasModel) {
      return html`<div class="loading">Loading peer...</div>`;
    }
    
    const model = this.model;
    const state = model.state || 'UNKNOWN';
    const uuid = model.uuid || '???';
    const shortUuid = uuid.substring(0, 8);
    const portCap = model.capabilities?.find(
      function(c: any) { return c.capability === 'httpPort'; }
    );
    const port = portCap?.port || 'N/A';
    const isPrimary = model.isPrimaryServer || false;
    
    return html`
      <div class="peer-item">
        <div class="status-dot ${state.toLowerCase()}"></div>
        <div class="peer-info">
          <div class="peer-title">Peer ${shortUuid}${isPrimary ? ' (Primary)' : ''}</div>
          <div class="peer-subtitle">Port ${port} · ${state}</div>
        </div>
        <div class="action-zone">
          ${this.actionsRender()}
        </div>
      </div>
    `;
  }
  
  /**
   * Render action buttons
   */
  private actionsRender(): TemplateResult {
    const actions = this.actionsDiscover();
    if (actions.length === 0) {
      return html``;
    }
    
    const buttons: TemplateResult[] = [];
    actions.forEach(this.actionButtonCreate.bind(this, buttons));
    return html`${buttons}`;
  }
  
  /**
   * Create action button - called via method reference
   */
  private actionButtonCreate(buttons: TemplateResult[], action: Action): void {
    buttons.push(html`
      <button 
        class="action-btn ${action.style?.toLowerCase() || ''}"
        @click=${this.actionInvokeHandler.bind(this, action)}
      >
        ${action.icon} ${action.label}
      </button>
    `);
  }
  
  /**
   * Handle action button click
   */
  private actionInvokeHandler(action: Action): void {
    const uuid = this.model?.uuid;
    if (!uuid) {
      console.error('[OncePeerItemView] No UUID for action');
      return;
    }
    
    this.dispatchEvent(new CustomEvent('action-invoke', {
      detail: { action, uuid },
      bubbles: true,
      composed: true
    }));
  }
}
```

### **5.5 UcpController with RelatedObjects**

```typescript
// Additions to layer2/UcpController.ts

import { RelatedObjects, InterfaceConstructor } from '../layer3/RelatedObjects.interface.js';

export class UcpController<TModel extends object> implements Controller<TModel>, RelatedObjects {
  
  // ... existing code ...
  
  /** RelatedObjects registry - keyed by interface constructor */
  private relatedRegistry: Map<InterfaceConstructor, Set<any>> = new Map();
  
  /**
   * Register a related object for an interface
   * Also registers for all parent interfaces in prototype chain
   * 
   * Web4 Principle 24: RelatedObjects Registry
   */
  relatedObjectRegister<T>(interfaceType: InterfaceConstructor<T>, instance: T): void {
    // Register for the specific type
    this.relatedRegistryAdd(interfaceType, instance);
    
    // Walk prototype chain to register for parent interfaces
    let proto = Object.getPrototypeOf(instance);
    while (proto && proto.constructor !== Object) {
      this.relatedRegistryAdd(proto.constructor, instance);
      proto = Object.getPrototypeOf(proto);
    }
  }
  
  /**
   * Add to registry (internal helper)
   */
  private relatedRegistryAdd(interfaceType: InterfaceConstructor, instance: any): void {
    if (!this.relatedRegistry.has(interfaceType)) {
      this.relatedRegistry.set(interfaceType, new Set());
    }
    this.relatedRegistry.get(interfaceType)!.add(instance);
  }
  
  /**
   * Lookup all objects implementing an interface
   */
  relatedObjectLookup<T>(interfaceType: InterfaceConstructor<T>): T[] {
    const instances = this.relatedRegistry.get(interfaceType);
    if (!instances) {
      return [];
    }
    return Array.from(instances) as T[];
  }
  
  /**
   * Lookup first object implementing an interface
   */
  relatedObjectLookupFirst<T>(interfaceType: InterfaceConstructor<T>): T | null {
    const instances = this.relatedObjectLookup(interfaceType);
    return instances.length > 0 ? instances[0] : null;
  }
  
  /**
   * Unregister object from all interfaces
   */
  relatedObjectUnregister(instance: any): void {
    this.relatedRegistry.forEach(function(instances) {
      instances.delete(instance);
    });
  }
}
```

export abstract class UcpComponent<TModel extends object> {

   * Register a related object - delegates to controller
   * Web4 Principle 24
   */

    this.controller.relatedObjectRegister(interfaceType, instance);
  }

   * Lookup related objects by interface - delegates to controller
   */

    return this.controller.relatedObjectLookup(interfaceType);

   * Lookup first related object by interface
   */

}
