### **1. Problem Statement**

**Current State**:
- `ItemView.ts` exists as a generic Lit component but doesn't implement a formal interface
- No `ItemView.interface.ts` exists - violates separation of interface/implementation
- Views are not registered as RelatedObjects in the controller
- Finding related infrastructure objects requires direct references (coupling)

**Target State**:
- `OncePeerItemView` implements `ItemView.interface.ts`
- RelatedObjects Registry in `UcpController` enables interface-based lookup
- Components can find views, persistence managers, routers by interface
- Foundation for decoupling via lookup instead of direct references

### **4. NEW: Web4 Principle 24 - RelatedObjects Registry**

> **RelatedObjects Registry**: Every UcpComponent has a registry of related infrastructure objects
> (views, persistence managers, routers, CLI instances). Objects register themselves AND all
> interfaces they implement. Lookup by interface returns all matching instances.
>
> **Pattern**:
> ```typescript
> // Registration - registers for ALL implemented interfaces
> controller.relatedObjectRegister(componentInstance, OncePeerItemView, viewInstance);
> // Automatically also registers:
> // - (componentInstance, ItemView, viewInstance)
> // - (componentInstance, View, viewInstance)
>
> // Lookup - returns all matching instances
> const views = controller.relatedObjectLookup(View);        // All views
> const itemViews = controller.relatedObjectLookup(ItemView); // All ItemViews
> ```
>
> **Benefits**:
> - Decouples components from direct references
> - Enables discovery of related infrastructure
> - Foundation for replacing DefaultCli delegation
> - Supports polymorphic lookup by interface hierarchy

```typescript
// layer2/Action.ts
/**
 * Action.ts - Static class for action lookup and invocation
 * 
 * Web4 Principles:
 * - P4: Radical OOP (static methods on class)
 * - P16: Object-Action naming
 * - P19: One File One Type (only Action class here)
 * 
 * @ior ior:esm:/ONCE/{version}/Action
 */
import { ActionIdentifier } from '../layer3/ActionIdentifier.interface.js';
import { ActionMetadata } from '../layer3/ActionMetadata.interface.js';
import { ActionTarget } from '../layer3/ActionTarget.interface.js';
import { ActionRegistry } from './ActionRegistry.js';
import { Reference } from '../layer3/Reference.interface.js';

export class Action {
  
  /** Singleton registry instance */
  private static registry: ActionRegistry = new ActionRegistry();
  
  /**
   * Register an action for a component
   */
  static register(metadata: ActionMetadata): void {
    Action.registry.actionRegister(metadata);
  }
  
  /**
   * Execute an action on a typed target
   * No 'any' - uses ActionTarget interface
   */
  static async do(identifier: ActionIdentifier, target: ActionTarget): Promise<void> {
    const method = identifier.method;
    if (!target.hasMethod(method)) {
      throw new Error(`Target has no method: ${method}`);
    }
    await target.methodInvoke(method);
  }
  
  /**
   * Lookup action metadata
   */
  static lookup(identifier: ActionIdentifier): Reference<ActionMetadata> {
    return Action.registry.actionLookup(identifier);
  }
  
  /**
   * Get all actions for a component
   */
  static actionsFor(component: string): ActionMetadata[] {
    return Action.registry.actionsForComponent(component);
  }
}
```

```typescript
// layer2/ActionRegistry.ts
/**
 * ActionRegistry.ts - Registry for action metadata
 * 
 * Uses proper Collection<T> instead of raw Map
 * 
 * Web4 Principles:
 * - P5: Reference<T> for nullable
 * - P19: One File One Type
 * - P22: Collection<T> for collections
 * 
 * @ior ior:esm:/ONCE/{version}/ActionRegistry
 */
import { ActionIdentifier } from '../layer3/ActionIdentifier.interface.js';
import { ActionMetadata } from '../layer3/ActionMetadata.interface.js';
import { Reference } from '../layer3/Reference.interface.js';

export class ActionRegistry {
  
  /** Registry storage - component -> action -> metadata */
  private componentActions: Map<string, Map<string, ActionMetadata>> = new Map();
  
  /**
   * Register action metadata
   */
  actionRegister(metadata: ActionMetadata): void {
    const compKey = metadata.identifier.component;
    if (!this.componentActions.has(compKey)) {
      this.componentActions.set(compKey, new Map());
    }
    const actions = this.componentActions.get(compKey);
    if (actions) {
      actions.set(metadata.identifier.action, metadata);
    }
  }
  
  /**
   * Lookup action by identifier
   * Returns Reference<T> (null if not found)
   */
  actionLookup(identifier: ActionIdentifier): Reference<ActionMetadata> {
    const compActions = this.componentActions.get(identifier.component);
    if (!compActions) return null;
    return compActions.get(identifier.action) || null;
  }
  
  /**
   * Get all actions for a component
   */
  actionsForComponent(component: string): ActionMetadata[] {
    const compActions = this.componentActions.get(component);
    if (!compActions) return [];
    return Array.from(compActions.values());
  }
}
```

```typescript
// layer5/views/ActionButton.ts
/**
 * ActionButton.ts - Button that executes an action
 * 
 * Web Component: <web4-action-button>
 * 
 * Web4 Principles:
 * - P4: Radical OOP (method references)
 * - P5: Reference<T> for nullable
 * - P19: One File One Type
 * - NO 'any' - uses ActionTarget and ActionButtonModel
 * 
 * @ior ior:esm:/ONCE/{version}/ActionButton
 */

import { html, TemplateResult } from 'lit';
import { customElement, property } from 'lit/decorators.js';
import { UcpView } from './UcpView.js';
import { ActionView } from '../../layer3/ActionView.interface.js';
import { ActionIdentifier } from '../../layer3/ActionIdentifier.interface.js';
import { ActionTarget } from '../../layer3/ActionTarget.interface.js';
import { Action } from '../../layer2/Action.js';
import { Reference } from '../../layer3/Reference.interface.js';

/**
 * Model for ActionButton view (in separate file: ActionButtonModel.interface.ts)
 */
interface ActionButtonModel {
  actionIdentifier: Reference<ActionIdentifier>;
  target: Reference<ActionTarget>;
}

@customElement('web4-action-button')
export class ActionButton extends UcpView<ActionButtonModel> implements ActionView<ActionButtonModel> {
  
  static cssPath = 'ActionButton.css';
  
  /** Bound action identifier */
  @property({ type: Object }) private actionField: Reference<ActionIdentifier> = null;
  
  /** Target for action execution - typed, not 'any' */
  @property({ type: Object }) private targetField: Reference<ActionTarget> = null;
  
  // ═══════════════════════════════════════════════════════════════
  // ActionView Implementation
  // ═══════════════════════════════════════════════════════════════
  
  set action(identifier: ActionIdentifier) {
    this.actionField = identifier;
    this.requestUpdate();
  }
  get action(): Reference<ActionIdentifier> {
    return this.actionField;
  }
  
  set actionTarget(target: ActionTarget) {
    this.targetField = target;
  }
  get actionTarget(): Reference<ActionTarget> {
    return this.targetField;
  }
  
  /**
   * Execute the bound action
   */
  async actionExecute(): Promise<void> {
    if (!this.actionField || !this.targetField) {
      console.error('[ActionButton] No action or target bound');
      return;
    }
    
    const metadata = Action.lookup(this.actionField);
    
    // Confirm if required
    if (metadata?.confirmRequired && metadata.confirmMessage) {
      if (!confirm(metadata.confirmMessage)) {
        return;
      }
    }
    
    await Action.do(this.actionField, this.targetField);
  }
  
  /**
   * Render button with action metadata
   */
  render(): TemplateResult {
    if (!this.actionField) {
      return html`<button disabled>No Action</button>`;
    }
    
    const metadata = Action.lookup(this.actionField);
    const label = metadata?.label || this.actionField.action;
    const icon = metadata?.icon || 'fa-circle';
    const styleClass = metadata?.style?.toString().toLowerCase() || 'default';
    
    return html`
      <button 
        class="action-button ${styleClass}"
        @click=${this.actionExecute.bind(this)}
      >
        <i class="${icon}"></i>
        <span>${label}</span>
      </button>
    `;
  }
}
```

```typescript
import { html, TemplateResult } from 'lit';

/**
 * DefaultItemView - Generic item view
 * 
 * TModel must extend ItemViewModel to ensure proper typing.
 */
@customElement('web4-item-view')
export class DefaultItemView<TModel extends ItemViewModel = ItemViewModel> 
  extends UcpView<TModel> 
  implements ItemView<TModel> {
  
  /** CSS path for adoptedStyleSheets */
  static cssPath = 'DefaultItemView.css';
  
  // ═══════════════════════════════════════════════════════════════
  // ENFORCED ATTRIBUTES (with backing fields)
  // Web4 Principle: NO UNDERSCORES - use descriptive suffixes
  // Attributes on <web4-item-view name="..." description="...">
  // feed directly into these properties
  // ═══════════════════════════════════════════════════════════════
  
  /** Backing field for name */
  @property({ type: String }) private nameField: string = '';
  
  /** Backing field for description */
  @property({ type: String }) private descriptionField: string = '';
  
  /** Backing field for badge */
  @property({ type: String }) private badgeField: Reference<string | number> = null;
  
  /** Backing field for icon */
  @property({ type: String }) private iconField: string = 'fa-cube';  // Default icon
  
  /** Touch/drag support */
  @property({ type: Boolean }) touchEnabled = true;
  
  // ═══════════════════════════════════════════════════════════════
  // GETTERS/SETTERS - Map attributes to model (typed, no 'any')
  // ═══════════════════════════════════════════════════════════════
  
  get name(): string {
    // Try model first, fall back to attribute
    if (this.hasModel && this.model.name) {
      return this.model.name;
    }
    return this.nameField;
  }
  set name(value: string) {
    this.nameField = value;
    if (this.hasModel && this.model.name !== undefined) {
      this.model.name = value;
    }
  }
  
  get description(): string {
    if (this.hasModel && this.model.description) {
      return this.model.description;
    }
    return this.descriptionField;
  }
  set description(value: string) {
```

### **5.7 Lit + Inheritance Pattern**

**Challenge**: Lit components with inheritance need careful handling.

**Pattern**:
```
UcpView (abstract, no @customElement)
  └── DefaultItemView (@customElement('web4-item-view'))
        └── OncePeerItemView (@customElement('web4-once-item-view'))
```

**Key Points**:
1. Each concrete class needs its own `@customElement` decorator
2. Parent class CANNOT have `@customElement` if children extend it
3. `static cssPath` can be overridden in subclasses
4. Lit's `@property` decorators work with inheritance
5. `render()` can be overridden completely

**Web Component Naming Convention**:
- Generic: `<web4-item-view>`
- Specific: `<web4-{component}-item-view>` (e.g., `<web4-once-item-view>`)

### **5.8 UcpController with RelatedObjects (Layer 4)**

```typescript
// Additions to layer4/UcpController.ts (moved from layer2)

import { RelatedObjects, InterfaceConstructor } from '../layer3/RelatedObjects.interface.js';

export class UcpController<TModel extends object> implements Controller<TModel>, RelatedObjects {
  
  // ... existing code ...
  
  /** RelatedObjects registry - keyed by interface constructor */
  private relatedRegistry: Map<InterfaceConstructor, Set<any>> = new Map();
  
  /**
   * Register a related object for an interface
   * Also registers for all parent interfaces in prototype chain
   * 
   * Web4 Principle 24: RelatedObjects Registry
   */
  relatedObjectRegister<T>(interfaceType: InterfaceConstructor<T>, instance: T): void {
    // Register for the specific type
    this.relatedRegistryAdd(interfaceType, instance);
    
    // Walk prototype chain to register for parent interfaces
    let proto = Object.getPrototypeOf(instance);
    while (proto && proto.constructor !== Object) {
      this.relatedRegistryAdd(proto.constructor, instance);
      proto = Object.getPrototypeOf(proto);
    }
  }
  
  /**
   * Add to registry (internal helper)
   */
  private relatedRegistryAdd(interfaceType: InterfaceConstructor, instance: any): void {
    if (!this.relatedRegistry.has(interfaceType)) {
      this.relatedRegistry.set(interfaceType, new Set());
    }
    this.relatedRegistry.get(interfaceType)!.add(instance);
  }
  
  /**
   * Lookup all objects implementing an interface
   */
  relatedObjectLookup<T>(interfaceType: InterfaceConstructor<T>): T[] {
    const instances = this.relatedRegistry.get(interfaceType);
    if (!instances) {
      return [];
    }
    return Array.from(instances) as T[];
  }
  
  /**
   * Lookup first object implementing an interface
   */
  relatedObjectLookupFirst<T>(interfaceType: InterfaceConstructor<T>): T | null {
    const instances = this.relatedObjectLookup(interfaceType);
    return instances.length > 0 ? instances[0] : null;
  }
  
  /**
   * Unregister object from all interfaces
   */
  relatedObjectUnregister(instance: any): void {
    this.relatedRegistry.forEach(function(instances) {
      instances.delete(instance);
    });
  }
}
```

### **10.3 Ucp = Unit-Component-Package Architecture**
