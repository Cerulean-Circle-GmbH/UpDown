### **4. NEW: Web4 Principle 24 - RelatedObjects Registry**

> **RelatedObjects Registry**: Every UcpComponent has a registry of related infrastructure objects
> (views, persistence managers, routers, CLI instances). Objects register themselves AND all
> interfaces they implement. Lookup by interface returns all matching instances.
>
> **Pattern**:
> ```typescript
> // Registration - registers for ALL implemented interfaces
> controller.relatedObjectRegister(componentInstance, OncePeerItemView, viewInstance);
> // Automatically also registers:
> // - (componentInstance, ItemView, viewInstance)
> // - (componentInstance, View, viewInstance)
>
> // Lookup - returns all matching instances
> const views = controller.relatedObjectLookup(View);        // All views
> const itemViews = controller.relatedObjectLookup(ItemView); // All ItemViews
> ```
>
> **Benefits**:
> - Decouples components from direct references
> - Enables discovery of related infrastructure
> - Foundation for replacing DefaultCli delegation
> - Supports polymorphic lookup by interface hierarchy

### **5.6 Action Registration in static start()**

**Pattern**: Actions are discovered and registered as RelatedObjects during class loading

```typescript
// layer1/UcpComponent.ts
export abstract class UcpComponent<TModel extends object> {
  
  /**
   * Static start() - called when class is loaded
   * Discovers actions from TSDoc and registers as RelatedObjects
   * 
   * Web4 Pattern: Dynamic action discovery, not static constants
   */
  static async start(): Promise<void> {
    const actions = this.actionsDiscover();
    
    // Register each action as a RelatedObject
    actions.forEach(function(action: ActionMetadata) {
      // Use controller's RelatedObjects registry
      // Actions are registered under ActionMetadata interface
      this.prototype.controller?.relatedObjectRegister(ActionMetadata, action);
    }.bind(this));
  }
  
  /**
   * Discover actions from class methods via TSDoc
   * Uses TSCompletion-like parsing (from Unit component)
   */
  private static actionsDiscover(): ActionMetadata[] {
    const actions: ActionMetadata[] = [];
    const componentName = this.name.replace('Default', '').replace('NodeJs', '');
    
    // Get method names from prototype
    const methodNames = Object.getOwnPropertyNames(this.prototype)
      .filter(function(name) {
        return typeof (this.prototype as Record<string, unknown>)[name] === 'function';
      }.bind(this));
    
    // For each method, check for @action annotation
    // (Actual implementation would use TSCompletion to parse TSDoc)
    methodNames.forEach(function(methodName: string) {
      const metadata = this.actionMetadataFromMethod(componentName, methodName);
      if (metadata) {
        actions.push(metadata);
      }
    }.bind(this));
    
    return actions;
  }
  
  /**
   * Extract action metadata from method TSDoc
   * Returns null if method is not an action
   */
  private static actionMetadataFromMethod(
    component: string, 
    methodName: string
  ): Reference<ActionMetadata> {
    // Parse TSDoc for @action annotation
    // This would use TSCompletion from Unit component
    // For now, simplified example:
    
    // Convert methodName to ACTION_NAME
    // peerStart → PEER_START
    const actionName = methodName
      .replace(/([A-Z])/g, '_$1')
      .toUpperCase()
      .replace(/^_/, '');
    
    // Return metadata (or null if no @action annotation)
    return {
      component,
      action: actionName,
      method: methodName,
      label: methodName,     // Would come from @action TSDoc
      icon: 'fa-circle',     // Would come from @actionIcon TSDoc
      style: ActionStyle.DEFAULT,
      shortcut: null,
      confirmRequired: false,
      confirmMessage: null
    };
  }
}
```

### **5.7 Action Lookup via RelatedObjects**

```typescript
// layer2/Action.ts - Simplified, uses RelatedObjects
/**
 * Action.ts - Static helpers for action invocation
 * 
 * NO separate registry - uses RelatedObjects
 * 
 * @ior ior:esm:/ONCE/{version}/Action
 */
import { ActionMetadata } from '../layer3/ActionMetadata.interface.js';
import { ActionTarget } from '../layer3/ActionTarget.interface.js';
import { Reference } from '../layer3/Reference.interface.js';

export class Action {
  
  /**
   * Execute an action on a typed target
   */
  static async do(action: ActionMetadata, target: ActionTarget): Promise<void> {
    if (!target.hasMethod(action.method)) {
      throw new Error(`Target has no method: ${action.method}`);
    }
    await target.methodInvoke(action.method);
  }
  
  /**
   * Lookup action from RelatedObjects by method name
   * @param controller UcpController with RelatedObjects
   * @param methodName Method name to find action for
   */
  static lookup(
    controller: RelatedObjects, 
    methodName: string
  ): Reference<ActionMetadata> {
    const allActions = controller.relatedObjectLookup(ActionMetadata);
    return allActions.find(function(a) { return a.method === methodName; }) || null;
  }
  
  /**
   * Get all actions from RelatedObjects
   */
  static all(controller: RelatedObjects): ActionMetadata[] {
    return controller.relatedObjectLookup(ActionMetadata);
  }
}
```

import { html, TemplateResult } from 'lit';
import { customElement, property } from 'lit/decorators.js';
import { UcpView } from './UcpView.js';
import { ActionView } from '../../layer3/ActionView.interface.js';
import { ActionMetadata } from '../../layer3/ActionMetadata.interface.js';
import { ActionTarget } from '../../layer3/ActionTarget.interface.js';
import { Action } from '../../layer2/Action.js';
import { Reference } from '../../layer3/Reference.interface.js';

/**
 * Model for ActionButton view (in separate file: ActionButtonModel.interface.ts)
 */

  actionMetadata: Reference<ActionMetadata>;
  target: Reference<ActionTarget>;
}

@customElement('web4-action-button')
export class ActionButton extends UcpView<ActionButtonModel> implements ActionView<ActionButtonModel> {
  
  static cssPath = 'ActionButton.css';
  
  /** Bound action metadata (discovered, not static constant) */
  @property({ type: Object }) private actionField: Reference<ActionMetadata> = null;
  
  /** Target for action execution - typed, not 'any' */
  @property({ type: Object }) private targetField: Reference<ActionTarget> = null;
  
  // ═══════════════════════════════════════════════════════════════
  // ActionView Implementation
  // ═══════════════════════════════════════════════════════════════
  
  set action(metadata: ActionMetadata) {
    this.actionField = metadata;
    this.requestUpdate();
  }
  get action(): Reference<ActionMetadata> {
    return this.actionField;
  }
  
  set actionTarget(target: ActionTarget) {
    this.targetField = target;
  }
  get actionTarget(): Reference<ActionTarget> {
    return this.targetField;
  }
  
  /**
   * Execute the bound action
   */
  async actionExecute(): Promise<void> {
    if (!this.actionField || !this.targetField) {
      console.error('[ActionButton] No action or target bound');
      return;
    }
    
    // Confirm if required (metadata has all info)
    if (this.actionField.confirmRequired && this.actionField.confirmMessage) {
      if (!confirm(this.actionField.confirmMessage)) {
        return;
      }
    }
    
    await Action.do(this.actionField, this.targetField);
  }
  
  /**
   * Render button - action metadata has all display info
   */
  render(): TemplateResult {
    if (!this.actionField) {
      return html`<button disabled>No Action</button>`;
    }
    
    const label = this.actionField.label;
    const icon = this.actionField.icon;
    const styleClass = this.actionField.style?.toString().toLowerCase() || 'default';
    
    return html`
      <button 
        class="action-button ${styleClass}"
        @click=${this.actionExecute.bind(this)}
      >
        <i class="${icon}"></i>
        <span>${label}</span>
      </button>
    `;
  }
}

/**

 * - P5: Reference<T> for nullable

export interface RelatedObjects {

  relatedObjectRegister<T>(interfaceType: InterfaceConstructor<T>, instance: T): void;

  relatedObjectLookup<T>(interfaceType: InterfaceConstructor<T>): T[];

  relatedObjectLookupFirst<T>(interfaceType: InterfaceConstructor<T>): Reference<T>;
  
  /**
   * Unregister a related object from all interfaces
   * @param instance The object instance to unregister (typed via generic)
   */
  relatedObjectUnregister<T>(instance: T): void;
}

### **5.7 Lit + Inheritance Pattern**

**Challenge**: Lit components with inheritance need careful handling.

**Pattern**:
```
UcpView (abstract, no @customElement)
  └── DefaultItemView (@customElement('web4-item-view'))
        └── OncePeerItemView (@customElement('web4-once-item-view'))
```

**Key Points**:
1. Each concrete class needs its own `@customElement` decorator
2. Parent class CANNOT have `@customElement` if children extend it
3. `static cssPath` can be overridden in subclasses
4. Lit's `@property` decorators work with inheritance
5. `render()` can be overridden completely

**Web Component Naming Convention**:
- Generic: `<web4-item-view>`
- Specific: `<web4-{component}-item-view>` (e.g., `<web4-once-item-view>`)

### **5.8 UcpController with RelatedObjects (Layer 4)**

```typescript
// Additions to layer4/UcpController.ts (moved from layer2)

import { RelatedObjects, InterfaceConstructor } from '../layer3/RelatedObjects.interface.js';

export class UcpController<TModel extends object> implements Controller<TModel>, RelatedObjects {
  
  // ... existing code ...
  
  /** RelatedObjects registry - keyed by interface constructor */
  private relatedRegistry: Map<InterfaceConstructor, Set<any>> = new Map();
  
  /**
   * Register a related object for an interface
   * Also registers for all parent interfaces in prototype chain
   * 
   * Web4 Principle 24: RelatedObjects Registry
   */
  relatedObjectRegister<T>(interfaceType: InterfaceConstructor<T>, instance: T): void {
    // Register for the specific type
    this.relatedRegistryAdd(interfaceType, instance);
    
    // Walk prototype chain to register for parent interfaces
    let proto = Object.getPrototypeOf(instance);
    while (proto && proto.constructor !== Object) {
      this.relatedRegistryAdd(proto.constructor, instance);
      proto = Object.getPrototypeOf(proto);
    }
  }
  
  /**
   * Add to registry (internal helper)
   */
  private relatedRegistryAdd(interfaceType: InterfaceConstructor, instance: any): void {
    if (!this.relatedRegistry.has(interfaceType)) {
      this.relatedRegistry.set(interfaceType, new Set());
    }
    this.relatedRegistry.get(interfaceType)!.add(instance);
  }
  
  /**
   * Lookup all objects implementing an interface
   */
  relatedObjectLookup<T>(interfaceType: InterfaceConstructor<T>): T[] {
    const instances = this.relatedRegistry.get(interfaceType);
    if (!instances) {
      return [];
    }
    return Array.from(instances) as T[];
  }
  
  /**
   * Lookup first object implementing an interface
   */
  relatedObjectLookupFirst<T>(interfaceType: InterfaceConstructor<T>): T | null {
    const instances = this.relatedObjectLookup(interfaceType);
    return instances.length > 0 ? instances[0] : null;
  }
  
  /**
   * Unregister object from all interfaces
   */
  relatedObjectUnregister(instance: any): void {
    this.relatedRegistry.forEach(function(instances) {
      instances.delete(instance);
    });
  }
}
```
