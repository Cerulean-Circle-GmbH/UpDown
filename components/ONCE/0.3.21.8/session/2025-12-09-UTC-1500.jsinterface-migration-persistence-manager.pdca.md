### **G.4.1: Convert PersistenceManager to JsInterface**

**File:** `src/ts/layer3/PersistenceManager.interface.ts`

```typescript
// BEFORE: TypeScript interface (erased at runtime)
export interface PersistenceManager { ... }
export const PersistenceManager = Symbol('PersistenceManager');

// AFTER: Abstract class extending JsInterface (exists at runtime)
import { JsInterface } from './JsInterface.js';

export abstract class PersistenceManager extends JsInterface {
  // All methods become abstract
  abstract scenarioSave<T extends Model>(...): Promise<void>;
  abstract scenarioLoad<T extends Model>(...): Promise<Scenario<T>>;
  abstract scenarioFind<T extends Model>(...): Promise<Scenario<T>[]>;
  abstract scenarioDelete(...): Promise<void>;
  abstract scenarioExists(...): Promise<boolean>;
  abstract typePathBuild(...): string;
  abstract domainPathBuild(...): string;
  abstract capabilityPathBuild(...): string;
}

/** @deprecated Use PersistenceManager class directly. Will be removed in 0.4.0.0 */
export const PersistenceManagerSymbol = Symbol('PersistenceManager');
```

### **G.4.3: Update RelatedObjects Registry**

**File:** `src/ts/layer4/UcpController.ts`

```typescript
// Update InterfaceKey type to accept class constructors
type InterfaceKey<T> = symbol | (abstract new (...args: any[]) => T);

// relatedObjectRegister accepts both Symbol and Class
relatedObjectRegister<T>(
  key: InterfaceKey<T>,
  instance: T
): void {
  if (typeof key === 'symbol') {
    console.warn('@deprecated: Use class key instead of Symbol for RelatedObjects');
  }
  this.relatedObjects.set(key, instance);
}
```
