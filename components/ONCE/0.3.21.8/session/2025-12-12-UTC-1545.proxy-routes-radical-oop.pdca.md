## **🏛️ ARCHITECTURE: Proxy as Routes (Radical OOP)**

### **Why Routes, Not Server Feature?**

| Approach | Radical OOP? | Reason |
|----------|--------------|--------|
| **ProxyRoute** | ✅ Yes | Route knows how to proxy - polymorphism! |
| **Server feature** | ❌ No | Server should only delegate, not contain logic |


### **HeaderRewriter.ts**

```typescript
// layer2/HeaderRewriter.ts

/**
 * HeaderRewriter - Rewrites HTTP headers based on rules
 * 
 * Radical OOP: HeaderRewriter knows how to rewrite headers.
 * 
 * @component Router
 * @layer 2
 * Web4 P4: No arrow functions
 */

import { IncomingHttpHeaders, OutgoingHttpHeaders } from 'http';
import { RewriteRule } from '../layer3/RewriteRule.interface.js';

export interface RewriteContext {
  clientIp?: string;
  originalHost?: string;
  originalUrl?: string;
  protocol?: 'http' | 'https';
}

export class HeaderRewriter {
  
  /**
   * Rewrite headers based on rules
   * Web4 P4: Method references in iteration
   */
  public rewrite(
    headers: IncomingHttpHeaders | OutgoingHttpHeaders,
    rules: RewriteRule[],
    context: RewriteContext
  ): OutgoingHttpHeaders {
    const result = { ...headers } as OutgoingHttpHeaders;
    
    for (const rule of rules) {
      this.ruleApply(result, rule, context);
    }
    
    return result;
  }
  
  /**
   * Apply single rewrite rule
   */
  private ruleApply(
    headers: OutgoingHttpHeaders,
    rule: RewriteRule,
    context: RewriteContext
  ): void {
    const value = rule.value ? this.variableSubstitute(rule.value, context) : undefined;
    
    switch (rule.operation) {
      case 'set':
        headers[rule.target.toLowerCase()] = value;
        break;
        
      case 'add':
        const existing = headers[rule.target.toLowerCase()];
        if (existing) {
          headers[rule.target.toLowerCase()] = `${existing}, ${value}`;
        } else {
          headers[rule.target.toLowerCase()] = value;
        }
        break;
        
      case 'remove':
        delete headers[rule.target.toLowerCase()];
        break;
        
      case 'replace':
        if (rule.pattern && value) {
          const current = headers[rule.target.toLowerCase()];
          if (typeof current === 'string') {
            headers[rule.target.toLowerCase()] = current.replace(
              new RegExp(rule.pattern, 'g'),
              value
            );
          }
        }
        break;
    }
  }
  
  /**
   * Substitute ${variables} in value
   */
  private variableSubstitute(value: string, context: RewriteContext): string {
    return value
      .replace('${client.ip}', context.clientIp || '')
      .replace('${original.host}', context.originalHost || '')
      .replace('${original.url}', context.originalUrl || '')
      .replace('${protocol}', context.protocol || 'http');
  }
}
```

```typescript
// layer2/HrefRewriter.ts

/**
 * HrefRewriter - Rewrites URLs in HTML response bodies
 * 
 * Radical OOP: HrefRewriter knows how to rewrite hrefs.
 * 
 * @component Router
 * @layer 2
 * Web4 P4: No arrow functions
 */

import { RewriteRule } from '../layer3/RewriteRule.interface.js';

export class HrefRewriter {
  
  /**
   * Rewrite URLs in HTML body
   */
  public rewrite(body: string, rules: RewriteRule[]): string {
    let result = body;
    
    for (const rule of rules) {
      result = this.ruleApply(result, rule);
    }
    
    return result;
  }
  
  /**
   * Apply single href rewrite rule
   */
  private ruleApply(body: string, rule: RewriteRule): string {
    if (rule.operation === 'replace' && rule.pattern && rule.value) {
      // Replace in href="..." and src="..." attributes
      const pattern = new RegExp(
        `(href|src|action)=["']${rule.pattern}([^"']*)["']`,
        'gi'
      );
      return body.replace(pattern, `$1="${rule.value}$2"`);
    }
    
    return body;
  }
}
```

## **📝 PHASE PX.4: ProxyRoute (Layer 2)**

### **ProxyRoute.ts** (Forward Proxy)

```typescript
// layer2/ProxyRoute.ts

/**
 * ProxyRoute - Forward proxy route
 * 
 * Client → ONCE → Upstream Server
 * 
 * Radical OOP: ProxyRoute knows how to proxy.
 * Server just delegates - ONE LINE!
 * 
 * @component Router
 * @layer 2
 * Web4 P4: No arrow functions
 */

import * as http from 'http';
import * as https from 'https';
import { IncomingMessage, ServerResponse } from 'http';
import { Route } from './Route.js';
import { HeaderRewriter, RewriteContext } from './HeaderRewriter.js';
import { ProxyRouteModel } from '../layer3/ProxyRouteModel.interface.js';
import { Scenario } from '../layer3/Scenario.interface.js';

export class ProxyRoute extends Route {
  public override model!: ProxyRouteModel;
  
  private headerRewriter: HeaderRewriter;
  private upstreamIndex: number = 0;  // For round-robin
  
  constructor() {
    super();
    this.headerRewriter = new HeaderRewriter();
  }
  
  public override init(scenario?: Scenario<ProxyRouteModel>): this {
    super.init(scenario);
    return this;
  }
  
  /**
   * Handle proxy request
   * Web4 P4: Bound methods for callbacks
   */
  public override async handle(req: IncomingMessage, res: ServerResponse): Promise<void> {
    const upstream = this.upstreamSelect();
    const upstreamUrl = new URL(upstream);
    
    // Build context for header rewriting
    const context: RewriteContext = {
      clientIp: this.clientIpExtract(req),
      originalHost: req.headers.host,
      originalUrl: req.url,
      protocol: upstreamUrl.protocol === 'https:' ? 'https' : 'http'
    };
    
    // Rewrite request headers
    const headers = this.headerRewriter.rewrite(
      req.headers,
      this.model.requestHeaderRules || [],
      context
    );
    
    // Add standard proxy headers
    headers['x-forwarded-for'] = context.clientIp;
    headers['x-forwarded-host'] = context.originalHost;
    headers['x-forwarded-proto'] = context.protocol;
    
    if (!this.model.preserveHost) {
      headers['host'] = upstreamUrl.host;
    }
    
    // Build upstream path
    let path = req.url || '/';
    if (this.model.stripPrefix) {
      path = path.replace(new RegExp(`^${this.model.stripPrefix}`), '');
    }
    if (this.model.addPrefix) {
      path = this.model.addPrefix + path;
    }
    
    // Make upstream request
    const options: http.RequestOptions = {
      hostname: upstreamUrl.hostname,
      port: upstreamUrl.port || (upstreamUrl.protocol === 'https:' ? 443 : 80),
      path: path,
      method: req.method,
      headers: headers,
      timeout: this.model.timeout || 30000
    };
    
    const transport = upstreamUrl.protocol === 'https:' ? https : http;
    const proxyReq = transport.request(options, this.upstreamResponseHandle.bind(this, res, context));
    
    proxyReq.on('error', this.proxyErrorHandle.bind(this, res));
    
    // Pipe request body
    req.pipe(proxyReq);
  }
  
  /**
   * Handle upstream response - bound method
   */
  private upstreamResponseHandle(
    res: ServerResponse,
    context: RewriteContext,
    upstreamRes: IncomingMessage
  ): void {
    // Rewrite response headers
    const headers = this.headerRewriter.rewrite(
      upstreamRes.headers,
      this.model.responseHeaderRules || [],
      context
    );
    
    res.writeHead(upstreamRes.statusCode || 200, headers);
    
    // TODO: If hrefRules exist, buffer and rewrite body
    // For now, just pipe through
    upstreamRes.pipe(res);
  }
  
  /**
   * Handle proxy error - bound method
   */
  private proxyErrorHandle(res: ServerResponse, error: Error): void {
    console.error('[ProxyRoute] Upstream error:', error.message);
    
    if (!res.headersSent) {
      res.writeHead(502, { 'Content-Type': 'application/json' });
      res.end(JSON.stringify({
        error: 'Bad Gateway',
        message: error.message
      }));
    }
  }
  
  /**
   * Select upstream (load balancing)
   */
  private upstreamSelect(): string {
    const upstreams = this.model.upstreams || [this.model.upstream];
    
    if (upstreams.length === 1) {
      return upstreams[0];
    }
    
    switch (this.model.loadBalancing) {
      case 'round-robin':
        this.upstreamIndex = (this.upstreamIndex + 1) % upstreams.length;
        return upstreams[this.upstreamIndex];
        
      case 'random':
        return upstreams[Math.floor(Math.random() * upstreams.length)];
        
      default:
        return upstreams[0];
    }
  }
  
  /**
   * Extract client IP from request
   */
  private clientIpExtract(req: IncomingMessage): string {
    const forwarded = req.headers['x-forwarded-for'];
    if (forwarded) {
      return (Array.isArray(forwarded) ? forwarded[0] : forwarded).split(',')[0].trim();
    }
    return req.socket.remoteAddress || '127.0.0.1';
  }
}
```

```typescript
// layer2/ReverseProxyRoute.ts

/**
 * ReverseProxyRoute - Reverse proxy with body rewriting
 * 
 * External Client → ONCE → Internal Service
 * 
 * Extends ProxyRoute, adds:
 * - Response body buffering
 * - Href rewriting in HTML
 * 
 * @component Router
 * @layer 2
 */

import { IncomingMessage, ServerResponse } from 'http';
import { ProxyRoute } from './ProxyRoute.js';
import { HrefRewriter } from './HrefRewriter.js';

export class ReverseProxyRoute extends ProxyRoute {
  
  private hrefRewriter: HrefRewriter;
  
  constructor() {
    super();
    this.hrefRewriter = new HrefRewriter();
  }
  
  /**
   * Override to add body rewriting for HTML responses
   */
  protected override upstreamResponseHandle(
    res: ServerResponse,
    context: any,
    upstreamRes: IncomingMessage
  ): void {
    const contentType = upstreamRes.headers['content-type'] || '';
    const needsRewrite = this.model.hrefRules?.length && contentType.includes('text/html');
    
    if (needsRewrite) {
      // Buffer response for rewriting
      this.bufferedResponseHandle(res, context, upstreamRes);
    } else {
      // Stream through (no rewriting)
      super.upstreamResponseHandle(res, context, upstreamRes);
    }
  }
  
  /**
   * Buffer and rewrite HTML response
   */
  private bufferedResponseHandle(
    res: ServerResponse,
    context: any,
    upstreamRes: IncomingMessage
  ): void {
    const chunks: Buffer[] = [];
    
    upstreamRes.on('data', this.chunkCollect.bind(this, chunks));
    upstreamRes.on('end', this.bufferedResponseSend.bind(this, res, context, upstreamRes, chunks));
  }
  
  private chunkCollect(chunks: Buffer[], chunk: Buffer): void {
    chunks.push(chunk);
  }
  
  private bufferedResponseSend(
    res: ServerResponse,
    context: any,
    upstreamRes: IncomingMessage,
    chunks: Buffer[]
  ): void {
    let body = Buffer.concat(chunks).toString('utf-8');
    
    // Rewrite hrefs
    body = this.hrefRewriter.rewrite(body, this.model.hrefRules || []);
    
    // Update content-length
    const headers = { ...upstreamRes.headers };
    headers['content-length'] = Buffer.byteLength(body).toString();
    
    res.writeHead(upstreamRes.statusCode || 200, headers);
    res.end(body);
  }
}
```
