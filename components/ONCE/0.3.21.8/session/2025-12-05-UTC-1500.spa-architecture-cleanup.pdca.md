## 📋 PLAN

### Problem Statement

1. **Floating Functional Code**: `demo-lit.html` has functional code outside classes:
   - `handleModelUpdate()` function
   - `setInterval()` polling fallback
   - Direct DOM manipulation
   
2. **Layer Violation**: `OnceOverView` (Layer 5) contains async methods:
   - `webSocketConnect()`, `webSocketOnMessage()`, etc.
   - `serversFetch()`, `iorCall()`
   - Should be synchronous views only

3. **Multiple Entry Points**: Multiple HTML files in `src/view/html/`:
   - `demo-hub.html` (legacy)
   - `demo-lit.html` (current)
   - Should be ONE minimal entry point

4. **SPA Architecture**: Need proper Single Page App pattern:
   - One boot sequence
   - Views swap in/out
   - URL routing without page reloads

### B.4: UcpRouter Design

```typescript
// layer5/views/UcpRouter.ts
import { LitElement, html, TemplateResult } from 'lit';
import { customElement, state } from 'lit/decorators.js';
import { Reference } from '../../layer3/Reference.interface.js';

/**
 * Route configuration
 */
export interface RouteConfig {
  /** URL pattern with optional dynamic segments like :id */
  pattern: string;
  /** Factory to create the view TemplateResult */
  viewFactory: (params: RouteParams) => TemplateResult;
  /** Page title for this route */
  title?: string;
  /** Should server be notified of this route? */
  serverRoute?: boolean;
}

/**
 * Parameters passed to route view factory
 */
export interface RouteParams {
  /** Current URL path */
  path: string;
  /** Extracted URL parameters (from :segments) */
  params: Record<string, string>;
  /** Query string parameters */
  query: URLSearchParams;
}

/**
 * UcpRouter - Web4 SPA Router Component
 * 
 * Custom router following Web4 principles:
 * - No external dependencies
 * - Full control over routing logic
 * - Server-side route integration
 * - API compatible with @lit-labs/router for future migration
 * 
 * @pdca 2025-12-05-UTC-1500.spa-architecture-cleanup.pdca.md
 */
@customElement('ucp-router')
export class UcpRouter extends LitElement {
  
  @state() private currentRoute: string = '/';
  @state() private currentView: Reference<TemplateResult> = null;
  
  private routes: Map<string, RouteConfig> = new Map();
  private kernel: Reference<any> = null;  // Reference to BrowserOnce
  
  connectedCallback(): void {
    super.connectedCallback();
    window.addEventListener('popstate', this.routeChangeHandle.bind(this));
    this.routeResolve(window.location.pathname);
  }
  
  disconnectedCallback(): void {
    super.disconnectedCallback();
    window.removeEventListener('popstate', this.routeChangeHandle.bind(this));
  }
  
  /**
   * Set kernel reference for server integration
   */
  kernelSet(kernel: any): void {
    this.kernel = kernel;
  }
  
  /**
   * Register a route
   */
  routeRegister(pattern: string, viewFactory: (params: RouteParams) => TemplateResult, options?: Partial<RouteConfig>): void {
    this.routes.set(pattern, {
      pattern,
      viewFactory,
      ...options
    });
  }
  
  /**
   * Navigate to a path
   */
  navigateTo(path: string, replace: boolean = false): void {
    if (replace) {
      window.history.replaceState({}, '', path);
    } else {
      window.history.pushState({}, '', path);
    }
    this.routeResolve(path);
  }
  
  /**
   * Handle browser back/forward
   */
  private routeChangeHandle(event: PopStateEvent): void {
    this.routeResolve(window.location.pathname);
  }
  
  /**
   * Resolve current path to a view
   */
  private routeResolve(path: string): void {
    this.currentRoute = path;
    
    // Try exact match first
    let config = this.routes.get(path);
    let params: Record<string, string> = {};
    
    // Try pattern matching if no exact match
    if (!config) {
      for (const [pattern, routeConfig] of this.routes) {
        const match = this.patternMatch(pattern, path);
        if (match) {
          config = routeConfig;
          params = match;
          break;
        }
      }
    }
    
    if (config) {
      // Update page title if specified
      if (config.title) {
        document.title = config.title;
      }
      
      // Create view with route params
      const routeParams: RouteParams = {
        path,
        params,
        query: new URLSearchParams(window.location.search)
      };
      this.currentView = config.viewFactory(routeParams);
      
      // Notify server if configured
      if (config.serverRoute && this.kernel) {
        this.serverRouteNotify(path);
      }
    } else {
      // 404 fallback
      this.currentView = html`<div class="not-found">404 - Route not found: ${path}</div>`;
    }
    
    this.requestUpdate();
  }
  
  /**
   * Match URL path against pattern with :segments
   */
  private patternMatch(pattern: string, path: string): Record<string, string> | null {
    const patternParts = pattern.split('/');
    const pathParts = path.split('/');
    
    if (patternParts.length !== pathParts.length) {
      return null;
    }
    
    const params: Record<string, string> = {};
    
    for (let i = 0; i < patternParts.length; i++) {
      const patternPart = patternParts[i];
      const pathPart = pathParts[i];
      
      if (patternPart.startsWith(':')) {
        // Dynamic segment
        params[patternPart.slice(1)] = pathPart;
      } else if (patternPart !== pathPart) {
        // Static segment doesn't match
        return null;
      }
    }
    
    return params;
  }
  
  /**
   * Notify server of route change (for SSR/analytics)
   */
  private serverRouteNotify(path: string): void {
    // TODO: Implement server notification via WebSocket
    console.log(`[UcpRouter] Server route notify: ${path}`);
  }
  
  render(): TemplateResult {
    return html`${this.currentView}`;
  }
}
```

```typescript
// layer5/views/UcpRouter.ts - processes child <ucp-route> elements
@customElement('ucp-router')
export class UcpRouter extends LitElement {
  
  connectedCallback(): void {
    super.connectedCallback();
    this.routesDiscover();  // Find all <ucp-route> children
    window.addEventListener('popstate', this.routeChangeHandle.bind(this));
    this.routeResolve(window.location.pathname);
  }
  
  /**
   * Discover routes from child <ucp-route> elements
   * Web4: No arrow functions, use method binding
   */
  private routesDiscover(): void {
    const routeElements = this.querySelectorAll('ucp-route') as NodeListOf<UcpRoute>;
    routeElements.forEach(this.routeElementProcess.bind(this));
  }
  
  private routeElementProcess(routeElement: UcpRoute): void {
    const config = routeElement.toRouteConfig();
    this.routes.set(config.pattern, config);
  }
  
  /**
   * Create view element for route
   * Web4: Method instead of factory function
   */
  private viewCreate(config: RouteConfig, params: RouteParams): TemplateResult {
    const viewTag = config.viewTag;
    // Dynamic element creation with property binding
    return html`
      <${unsafeStatic(viewTag)} 
        .model=${this.model} 
        .kernel=${this.kernel}
        .routeParams=${params}>
      </${unsafeStatic(viewTag)}>
    `;
  }
}
```

```typescript
@customElement('ucp-static-route')
export class UcpStaticRoute extends LitElement {
  @property({ type: String }) pattern: string = '';
  @property({ type: String }) base: string = '';
  @property({ type: String }) index: string = 'index.html';  // Default index file
  @property({ type: Boolean, attribute: 'directory-listing' }) directoryListing: boolean = false;
  
  /**
   * Resolve file path from URL
   */
  pathResolve(urlPath: string): string {
    // Extract the ** part from pattern match
    const patternRegex = this.patternToRegex();
    const match = urlPath.match(patternRegex);
    if (!match) return null;
    
    const relativePath = match[1] || '';
    return `${this.base}${relativePath}`;
  }
  
  private patternToRegex(): RegExp {
    // Convert /css/** to /css/(.*)
    const regexStr = this.pattern
      .replace(/\*\*/g, '(.*)')
      .replace(/\*/g, '([^/]*)')
      .replace(/\{(\w+)\}/g, '(?<$1>[^/]+)');
    return new RegExp(`^${regexStr}$`);
  }
}
```

```typescript
@customElement('ucp-ior-route')
export class UcpIorRoute extends LitElement {
  @property({ type: String }) pattern: string = '';
  @property({ type: String }) component: string = '';  // Target component name
  @property({ type: String }) method: string = '';      // Method to invoke
  @property({ type: String, attribute: 'param-map' }) paramMap: string = '';  // URL params to method args
  @property({ type: String, attribute: 'response-type' }) responseType: string = 'json';
  
  /**
   * Build IOR from URL match
   */
  iorBuild(urlPath: string, params: Record<string, string>): IOR {
    const component = params.component || this.component;
    const version = params.version || 'latest';
    const uuid = params.uuid || 'singleton';
    const method = params.method || this.method;
    
    return new IOR({
      protocol: 'ior',
      host: window.location.host,
      component,
      version,
      uuid,
      method
    });
  }
  
  /**
   * Map URL parameters to method arguments
   */
  argsMap(params: Record<string, string>): Record<string, any> {
    if (!this.paramMap) return params;
    
    const args: Record<string, any> = {};
    const mappings = this.paramMap.split(',');
    mappings.forEach(this.mappingProcess.bind(this, params, args));
    return args;
  }
  
  private mappingProcess(params: Record<string, string>, args: Record<string, any>, mapping: string): void {
    const [urlParam, argName] = mapping.split(':').map(s => s.trim());
    if (params[urlParam] !== undefined) {
      args[argName || urlParam] = params[urlParam];
    }
  }
}
```

```typescript
// layer5/views/UcpLoader.ts
@customElement('ucp-loader')
export class UcpLoader extends LitElement {
  @property({ type: String, attribute: 'ior-pattern' }) iorPattern: string = '';
  @property({ type: String, attribute: 'static-path' }) staticPath: string = '';
  @property({ type: String, attribute: 'param-map' }) paramMap: string = '';
  @property({ type: String, attribute: 'target-property' }) targetProperty: string = 'data';
  @property({ type: Boolean }) required: boolean = true;  // Fail route if load fails?
  @property({ type: Number }) timeout: number = 5000;     // Load timeout in ms
  
  private orchestrator: Reference<BrowserOnceOrchestrator> = null;
  
  /**
   * Load data for route
   * Called by UcpRouter before rendering view
   * @returns Data to set on view's targetProperty
   */
  async load(routeParams: RouteParams): Promise<any> {
    if (this.staticPath) {
      return this.staticLoad();
    }
    if (this.iorPattern) {
      return this.iorLoad(routeParams);
    }
    return null;
  }
  
  /**
   * Load from static file
   */
  private async staticLoad(): Promise<any> {
    const response = await fetch(this.staticPath);
    if (!response.ok) {
      if (this.required) {
        throw new Error(`Failed to load: ${this.staticPath}`);
      }
      return null;
    }
    return response.json();
  }
  
  /**
   * Load via IOR method invocation
   */
  private async iorLoad(routeParams: RouteParams): Promise<any> {
    const args = this.argsMap(routeParams.params);
    const ior = this.iorBuild(args);
    
    // Delegate to orchestrator for actual IOR call
    if (!this.orchestrator) {
      this.orchestrator = this.orchestratorFind();
    }
    
    return this.orchestrator?.iorCall(ior.method, args);
  }
  
  private orchestratorFind(): Reference<BrowserOnceOrchestrator> {
    // Find orchestrator from parent router or kernel
    const router = this.closest('ucp-router') as UcpRouter;
    return router?.kernel?.orchestrator || null;
  }
  
  private iorBuild(args: Record<string, any>): IOR {
    // Build IOR from pattern, substituting args
    let iorStr = this.iorPattern;
    Object.entries(args).forEach(this.iorSubstitute.bind(this, iorStr));
    return IOR.parse(iorStr);
  }
  
  private iorSubstitute(iorStr: string, entry: [string, any]): void {
    const [key, value] = entry;
    iorStr = iorStr.replace(`:${key}`, String(value));
  }
  
  private argsMap(params: Record<string, string>): Record<string, any> {
    if (!this.paramMap) return { ...params };
    
    const args: Record<string, any> = {};
    const mappings = this.paramMap.split(',');
    mappings.forEach(this.mappingProcess.bind(this, params, args));
    return args;
  }
  
  private mappingProcess(params: Record<string, string>, args: Record<string, any>, mapping: string): void {
    const [urlParam, argName] = mapping.split(':').map(s => s.trim());
    if (params[urlParam] !== undefined) {
      args[argName || urlParam] = params[urlParam];
    }
  }
}
```

#### UcpRouter Processing Loaders

```typescript
// In UcpRouter.ts - process loaders before rendering view
@customElement('ucp-router')
export class UcpRouter extends LitElement {
  
  /**
   * Resolve route with loaders
   */
  private async routeResolve(path: string): Promise<void> {
    this.currentRoute = path;
    
    const { config, params } = this.routeMatch(path);
    
    if (config) {
      if (config.title) {
        document.title = config.title;
      }
      
      // Load data from all loaders
      const loadedData = await this.loadersExecute(config, params);
      
      // Create view element with loaded data
      this.currentView = this.viewCreate(config, params, loadedData);
      
      if (config.serverRoute && this.kernel) {
        this.serverRouteNotify(path);
      }
    } else {
      this.currentView = this.notFoundViewCreate(path);
    }
    
    this.requestUpdate();
  }
  
  /**
   * Execute all loaders for a route
   */
  private async loadersExecute(config: RouteConfig, params: RouteParams): Promise<Record<string, any>> {
    const loadedData: Record<string, any> = {};
    
    if (!config.loaders || config.loaders.length === 0) {
      return loadedData;
    }
    
    // Execute loaders in parallel
    const loadPromises = config.loaders.map(this.loaderExecute.bind(this, params));
    const results = await Promise.all(loadPromises);
    
    // Merge results into loadedData
    results.forEach(this.loaderResultMerge.bind(this, loadedData));
    
    return loadedData;
  }
  
  private async loaderExecute(params: RouteParams, loader: UcpLoader): Promise<{ property: string; data: any }> {
    const data = await loader.load(params);
    return { property: loader.targetProperty, data };
  }
  
  private loaderResultMerge(loadedData: Record<string, any>, result: { property: string; data: any }): void {
    loadedData[result.property] = result.data;
  }
  
  /**
   * Create view element with loaded data
   */
  private viewCreate(config: RouteConfig, params: RouteParams, loadedData: Record<string, any>): TemplateResult {
    return staticHtml`
      <${unsafeStatic(config.viewTag)}
        .model=${this.model}
        .kernel=${this.kernel}
        .routeParams=${params}
        .loadedData=${loadedData}>
      </${unsafeStatic(config.viewTag)}>
    `;
  }
}
```

```typescript
  }
```

```typescript
   * Handle action event from views
   * Web4: Async method instead of async arrow function
   */

    await this.actionHandle(action, model);

   * Execute action via IOR call
   */
  private async actionHandle(action: ActionMetadata, model: any): Promise<void> {

    Object.assign(this.component.model, result);
    
    // Views will auto-update via model property binding
    this.router?.modelSet(this.component.model);

   * Cleanup - remove event listeners

  dispose(): void {
    if (this.container) {
      this.container.removeEventListener('navigate', this.navigateEventHandle.bind(this));
      this.container.removeEventListener('action-request', this.actionEventHandle.bind(this));
```
