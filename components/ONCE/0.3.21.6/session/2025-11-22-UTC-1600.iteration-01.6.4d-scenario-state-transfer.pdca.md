**Why This is Wrong**:

- Sends model, not full scenario
- Primary has to parse message protocol
- Violates Web4 Principle 11: Protocol-Less Communication

class DefaultONCE {
    private notifyPrimaryOfStateChange() {
        if (!this.isPrimaryServer && this.primaryWebSocket) {
            const scenario = this.getScenario();  // Full scenario object
            this.primaryWebSocket.send(JSON.stringify(scenario));  // NO protocol wrapper!
        }

        this.notifyPrimaryOfStateChange();  // Send updated scenario
    }
    
    async stopServer() {
        this.model.state.state = 'stopping';
        this.notifyPrimaryOfStateChange();  // Send scenario with state=stopping
        // ... stop logic ...
        this.model.state.state = 'stopped';
        this.notifyPrimaryOfStateChange();  // Send scenario with state=stopped (triggers deregistration)
    }
}

// ✅ Primary receives scenarios and updates registry
class ServerHierarchyManager {
    private handleWebSocketMessage(ws: WebSocket, data: any) {
        // NO switch/case for message types!
        // Data IS a scenario (has ior, owner, model)
        const scenario = data;
        
        if (this.isScenario(scenario)) {
            this.handleScenarioStateUpdate(scenario);
        }
    }
    
    private handleScenarioStateUpdate(scenario: any) {
        const state = scenario.model?.state?.state;
        const uuid = scenario.model?.uuid;
        
        if (state === 'stopped' || state === 'shutdown') {
            // Deregister
            this.serverRegistry.delete(uuid);
            console.log(`🗑️  Server ${uuid} ${state} - deregistered`);
        } else if (state === 'running' || state === 'client_server') {
            // Register or update
            this.serverRegistry.set(uuid, {
                scenario: scenario,  // Store full scenario
                lastSeen: new Date().toISOString(),
                websocket: ws  // Keep WebSocket for real-time updates
            });
            console.log(`📡 Server ${uuid} ${state} - registered/updated`);
        }
    }
}
```

/**
 * Handle incoming scenario from client server
 * ✅ Protocol-Less: No message types, just scenarios
 */
private handleWebSocketMessage(ws: WebSocket, data: any): void {
    try {
        // Check if data is a scenario (has ior, model structure)
        if (data.ior && data.model) {
            this.handleScenarioStateUpdate(ws, data);
        } else {
            console.warn('⚠️  Received non-scenario WebSocket data:', Object.keys(data));
        }
    } catch (error) {
        console.error('❌ Error handling WebSocket message:', error);
    }
}

/**
 * Handle scenario state update from client server
 * Registers, updates, or deregisters based on scenario state
 */
private handleScenarioStateUpdate(ws: WebSocket, scenario: any): void {
    const state = scenario.model?.state?.state;
    const uuid = scenario.model?.uuid;
    const port = scenario.model?.state?.capabilities?.find((c: any) => c.capability === 'httpPort')?.port;
    
    if (!uuid) {
        console.warn('⚠️  Received scenario without UUID');
        return;
    }
    
    console.log(`📥 Received scenario from ${uuid} (state: ${state}, port: ${port})`);
    
    // State-based actions
    if (state === LifecycleState.STOPPED || state === LifecycleState.SHUTDOWN) {
        // Deregister server
        if (this.serverRegistry.has(uuid)) {
            this.serverRegistry.delete(uuid);
            console.log(`🗑️  Server ${uuid} deregistered (state: ${state})`);
            
            // Broadcast deregistration to other clients
            this.broadcastToBrowserClients({
                type: 'server-deregistered',  // For browser UI only
                uuid: uuid
            });
        }
    } else if (state === LifecycleState.RUNNING || state === LifecycleState.CLIENT_SERVER) {
        // Register or update server
        this.serverRegistry.set(uuid, {
            scenario: scenario,  // ✅ Store full scenario
            lastSeen: new Date().toISOString(),
            websocket: ws  // Keep connection for real-time updates
        });
        
        const action = this.serverRegistry.has(uuid) ? 'updated' : 'registered';
        console.log(`📡 Server ${uuid} ${action} (state: ${state}, port: ${port})`);
        
        // Broadcast to browser clients
        this.broadcastToBrowserClients({
            type: 'server-registered',  // For browser UI only
            scenario: scenario
        });
    }
}
```
