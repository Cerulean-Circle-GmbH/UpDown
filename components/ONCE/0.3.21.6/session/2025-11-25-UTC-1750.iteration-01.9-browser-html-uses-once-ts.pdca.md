async init(scenario?: any): Promise<this> {
    // 1. Initialize model (ALL state here)
    this.model = {
        peerHost: scenario?.peerHost || window.location.host,
        isConnected: false,
        peers: [],
        messages: [],

#### **Step 3: Protocol-Less Scenario State Transfer** (Web4 Pattern)

**WebSocket Push for Scenario Updates** (NOT polling, NOT message passing):

```typescript
private async connectWebSocket(): Promise<void> {
    const wsUrl = `ws://${this.model.peerHost}`;
    this.model.ws = new WebSocket(wsUrl);
    
    const self = this;
    
    // ✅ Connection opened
    this.model.ws.onopen = function() {
        self.model.isConnected = true;
        self.addMessageToLog('connection', '✅ WebSocket connected');
    };
    
    // ✅ Connection closed
    this.model.ws.onclose = function() {
        self.model.isConnected = false;
        self.addMessageToLog('error', '❌ WebSocket disconnected');
    };
    
    // ✅ Error handling
    this.model.ws.onerror = function(error) {
        console.error('WebSocket error:', error);
        self.addMessageToLog('error', '❌ WebSocket error');
    };
}

private listenForScenarioUpdates(): void {
    const self = this;
    
    // ✅ Receive scenario state pushes (NOT protocol messages!)
    this.model.ws.onmessage = function(event) {
        try {
            // ✅ Parse incoming scenario (Web4 Scenario format)
            const scenario = JSON.parse(event.data);
            
            // ✅ Handle scenario state update
            self.handleScenarioUpdate(scenario);
        } catch (error) {
            console.error('Failed to parse scenario:', error);
        }
    };
}

private handleScenarioUpdate(scenario: any): void {
    // ✅ Determine scenario type from IOR component
    const component = scenario.ior?.component;
    
    if (component === 'ONCE') {
        // ✅ Peer state update
        this.updatePeerState(scenario);
    } else if (component === 'ONCEMessage') {
        // ✅ Message scenario received
        this.handleMessageScenario(scenario);
    }
    
    // ✅ Update UI
    this.updatePeersDisplay();
    this.updateStatsDisplay();
}

private updatePeerState(scenario: any): void {
    // ✅ Update or add peer in model
    const existingIndex = this.model.peers.findIndex(
        p => p.ior.uuid === scenario.ior.uuid
    );
    
    if (existingIndex >= 0) {
        this.model.peers[existingIndex] = scenario;
    } else {
        this.model.peers.push(scenario);
    }
}

private handleMessageScenario(scenario: any): void {
    // ✅ Extract message from scenario model
    const message = scenario.model;
    
    // ✅ Update stats
    this.model.stats.received++;
    this.model.messages.push(message);
    
    // ✅ Display message
    const from = message.from?.peerHost || 'unknown';
    this.addMessageToLog('message', `📩 Message from ${from}: ${message.content}`);
    
    // ✅ Auto-acknowledge (send ACK scenario back)
    this.sendAcknowledgment(scenario);
}
```

#### **Step 5: Scenario-Based State Updates** (NOT Protocol Messages)

**Broadcasting to All Peers** (Scenario Transfer via WebSocket):

```typescript
async broadcastToAll(): Promise<void> {
    // ✅ Create a scenario representing the message
    const messageScenario = {
        ior: {
            uuid: this.generateUUID(),
            component: 'ONCEMessage',
            version: '0.3.21.6'
        },
        owner: this.model.userUUID,
        model: {
            uuid: this.generateUUID(),
            name: 'Broadcast Message',
            type: 'broadcast',
            content: 'Hello all peers!',
            timestamp: new Date().toISOString(),
            from: {
                peerHost: this.model.peerHost,
                peerUUID: this.model.peerUUID
            }
        }
    };
    
    // ✅ Send scenario via WebSocket (NOT HTTP fetch!)
    if (this.model.ws && this.model.ws.readyState === WebSocket.OPEN) {
        this.model.ws.send(JSON.stringify(messageScenario));
        
        // ✅ Update local model (increment stats)
        this.model.stats.sent++;
        this.model.messages.push(messageScenario.model);
        
        // ✅ Update UI
        this.updateStatsDisplay();
        this.addMessageToLog('broadcast', '📡 Broadcast sent to all peers');
    } else {
        this.addMessageToLog('error', '❌ WebSocket not connected');
    }
}

private async sendAcknowledgment(originalScenario: any): Promise<void> {
    // ✅ Create ACK scenario
    const ackScenario = {
        ior: {
            uuid: this.generateUUID(),
            component: 'ONCEAcknowledgment',
            version: '0.3.21.6'
        },
        owner: this.model.userUUID,
        model: {
            uuid: this.generateUUID(),
            name: 'Message Acknowledgment',
            originalMessageUUID: originalScenario.ior.uuid,
            timestamp: new Date().toISOString(),
            from: {
                peerHost: this.model.peerHost,
                peerUUID: this.model.peerUUID
            }
        }
    };
    
    // ✅ Send ACK via WebSocket
    if (this.model.ws && this.model.ws.readyState === WebSocket.OPEN) {
        this.model.ws.send(JSON.stringify(ackScenario));
        this.model.stats.acks++;
        this.updateStatsDisplay();
    }
}
```

#### **Step 6: UI Updates** (Methods Operating on Model)

**Display Update Methods** - Pure methods, no arrow functions:

```typescript
private updatePeersDisplay(): void {
    const el = this.model.elements.peersDisplay;
    if (!el) return;
    
    const count = this.model.peers.length;
    if (count === 0) {
        el.textContent = 'No peers connected';
    } else {
        const peerList = this.model.peers
            .map(p => `${p.model.peerHost}:${p.model.port}`)
            .join(', ');
        el.textContent = `${count} peer${count === 1 ? '' : 's'}: ${peerList}`;
    }
}

private updateStatsDisplay(): void {
    const stats = this.model.stats;
    
    if (this.model.elements.messagesSent) {
        this.model.elements.messagesSent.textContent = String(stats.sent);
    }
    if (this.model.elements.messagesReceived) {
        this.model.elements.messagesReceived.textContent = String(stats.received);
    }
    if (this.model.elements.acknowledgmentsSent) {
        this.model.elements.acknowledgmentsSent.textContent = String(stats.acks);
    }
}

private addMessageToLog(type: string, text: string): void {
    const el = this.model.elements.messageLog;
    if (!el) return;
    
    const now = new Date();
    const timestamp = `${String(now.getHours()).padStart(2, '0')}:${String(now.getMinutes()).padStart(2, '0')}:${String(now.getSeconds()).padStart(2, '0')}`;
    
    const messageDiv = document.createElement('div');
    messageDiv.className = type;
    messageDiv.innerHTML = `<span class="timestamp">[${timestamp}]</span> ${text}`;
    
    el.appendChild(messageDiv);
    el.scrollTop = el.scrollHeight; // Auto-scroll
}
```

        const response = await fetch(url, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(params)
        });
        
        if (!response.ok) {
            throw new Error(`IOR invocation failed: ${response.statusText}`);

        return await response.json();
    } catch (error) {
        console.error(`IOR method invocation failed: ${method}`, error);
        return null;
    }
}

#### **Summary: Web4 Radical OOP Pattern**

**✅ Principles Upheld**:
1. **Empty Constructor** - All initialization in `init()`
2. **State in Model** - Everything in `this.model`, including WebSocket reference
3. **Methods Only** - No arrow functions, no callbacks (except for WebSocket event handlers)
4. **Scenario-Based** - All communication via scenario transfer via WebSocket, not protocol messages
5. **IOR-Based** - Scenario identification via IOR (component type determines handling)
6. **Protocol-Less** - No message types like `type: 'broadcast'`, just scenario state pushed via WebSocket
7. **WebSocket Push** - Real-time scenario updates pushed from server, not polling
8. **Global Singleton** - Accessed via `ONCE.getInstance()` or `window.global.ONCE`

**❌ What's Eliminated**:
- Polling for updates (replaced with WebSocket push)
- Protocol message types (replaced with scenario IOR component type)
- Arrow functions in HTML (replaced with method references)
- Direct `fetch()` calls (replaced with WebSocket.send(JSON.stringify(scenario)))
- Inline JavaScript (all in compiled TypeScript module)
- `window.onceKernel` (replaced with `ONCE.getInstance()`)
