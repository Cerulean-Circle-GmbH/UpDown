### **✅ APPROVED: ONCEPeerModel (Single Source of Truth)**

```typescript
/**
 * ONCEPeerModel - Unified model for ALL ONCE peers
 * 
 * Occam's Razor: One model to rule them all.
 * Works for: Browser, Node.js, Worker, Service Worker
 * 
 * @layer3
 * Web4 P19: One File One Type
 */
export interface ONCEPeerModel extends Model {
    // === IDENTITY (from Model) ===
    // uuid: string;   // inherited
    // name: string;   // inherited
    
    // === VERSION ===
    version: string;
    
    // === ENVIRONMENT ===
    environment: EnvironmentInfo;  // Replaces platform, envType confusion
    
    // === LIFECYCLE ===
    state: LifecycleState;
    startTime: Reference<Date>;
    connectionTime: Reference<Date>;
    
    // === NETWORK ===
    host: string;           // hostname (McDonges-3.fritz.box)
    port: number;           // primary port
    isPrimary: boolean;     // is this the primary peer?
    primaryPeerUuid: Reference<string>;  // UUID of primary peer's scenario
    
    // === P2P ===
    peers: ONCEPeerModel[];  // Connected peers (recursive!)
    
    // === COMPONENTS ===
    components: Map<string, Component>;  // Registered components
    
    // === BROWSER-ONLY (optional) ===
    ws?: Reference<WebSocket>;
    
    // === NODE-ONLY (optional) ===
    pid?: number;
    capabilities?: ServerCapability[];
}
```

- ✅ `primaryPeer` → `primaryPeerUuid: Reference<string>` (UUID to scenario)
