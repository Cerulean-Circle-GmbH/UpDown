```typescript
// CPA.1.1-1.4: UcpComponent.ts — CLI at BASE level
import type { Scenario } from '../layer3/Scenario.interface.js';
import { DefaultCLI } from './DefaultCLI.js';

export abstract class UcpComponent<TModel extends Model = Model> implements Component<TModel> {
  // CLI is ALWAYS initialized — not nullable (P3: Field suffix, not underscore)
  private cliField: DefaultCLI;

  constructor() {
    // CLI is path authority — ALWAYS create it at base level
    this.cliField = new DefaultCLI().init() as DefaultCLI;
  }

  // P16: TypeScript accessors
  get cli(): DefaultCLI {
    return this.cliField;
  }
  
  // Setter allows external override (e.g., ONCECLI passes its own CLI)
  set cli(value: DefaultCLI) {
    this.cliField = value;
  }

  // Path accessors — delegate to CLI (single source of truth)
  get projectRoot(): string {
    return this.cli.model.projectRoot;
  }
  
  get componentsDirectory(): string {
    return this.cli.model.componentsDirectory;
  }

  // ✅ STANDARD init() signature — Scenario<TModel>
  init(scenario?: Scenario<TModel>): this {
    // ... existing init logic ...
    return this;
  }
}

// All components inherit CLI automatically:
// - DefaultWeb4TSComponent → inherits cli, projectRoot, etc.
// - NodeJsOnce → inherits cli, projectRoot, etc.
// - BrowserOnce → inherits cli, projectRoot, etc.

// Usage when ONCECLI needs to pass its own CLI:
const component = new NodeJsOnce();
component.cli = onceCli;  // Override before init()
component.init(scenario);
```
