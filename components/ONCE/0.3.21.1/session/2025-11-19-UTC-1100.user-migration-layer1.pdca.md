**File 4**: `layer1/NodeOSInfrastructure.ts` (Separate file - one class only)
```typescript
import { EnvironmentModel } from '../layer3/EnvironmentModel.interface.js';
import { EnvironmentScenario } from '../layer3/EnvironmentScenario.interface.js';
import { DefaultEnvironmentModel } from './DefaultEnvironmentModel.js';

/**
 * NodeOSInfrastructure - TRUE Radical OOP Infrastructure
 * Web4 EAM Layer 1 - ONLY place where 'os' module is imported
 */
export class NodeOSInfrastructure {
  private model: EnvironmentModel;
  
  constructor() {
    this.model = new DefaultEnvironmentModel();
  }
  
  public getModel(): EnvironmentModel {
    return this.model;
  }
  
  public setModel(model: EnvironmentModel): void {
    this.model = model;
  }
  
  public async detectEnvironment(): Promise<EnvironmentModel> {
    const os = await import('os');  // ✅ ONLY import here
    
    const fqdn: string = os.hostname();
    const hostname: string = this.extractHostname(fqdn);
    const primaryIP: string = this.detectPrimaryIP(os);
    const domain: string = this.extractDomain(fqdn);
    
    this.model.setHostname(hostname);
    this.model.setFqdn(fqdn);
    this.model.setPrimaryIP(primaryIP);
    this.model.setDomain(domain);
    
    return this.model;
  }
  
  public init(scenario?: EnvironmentScenario): this {
    if (scenario) {
      this.model.init(scenario);
    }
    return this;
  }
  
  public toScenario(): EnvironmentScenario {
    return this.model.toScenario();
  }
  
  private extractHostname(fqdn: string): string {
    if (fqdn.includes('.')) {
      const parts: string[] = fqdn.split('.');
      return parts[0];
    }
    return fqdn;
  }
  
  private extractDomain(fqdn: string): string {
    if (fqdn === 'localhost') {
      return 'local.once';
    }
    if (fqdn.includes('.')) {
      const parts: string[] = fqdn.split('.');
      parts.shift(); // Remove hostname
      return parts.reverse().join('.');
    }
    return 'local.once';
  }
  
  private detectPrimaryIP(os: any): string {
    try {
      const interfaces = os.networkInterfaces();
      const interfaceNames: string[] = Object.keys(interfaces);
      
      for (const name of interfaceNames) {
        const ifaces = interfaces[name];
        if (!ifaces) continue;
        
        for (const iface of ifaces) {
          if (iface.family === 'IPv4' && !iface.internal) {
            return iface.address;
          }
        }
      }
    } catch (error) {
      console.warn('⚠️  Failed to detect primary IP:', error);
    }
    return '127.0.0.1';
  }
}
```

```typescript
import { createHash } from 'crypto';
import { User } from '../layer3/User.interface.js';
import { UserModel } from '../layer3/UserModel.interface.js';
import { Scenario } from '../layer3/Scenario.interface.js';
import { NodeOSInfrastructure, EnvironmentModel } from '../layer1/NodeOSInfrastructure.js';

export class DefaultUser implements User {
  public readonly uuid: string;
  public readonly username: string;
  public readonly hostname: string;
  
  private infrastructure?: NodeOSInfrastructure;

  constructor(username?: string, infrastructure?: NodeOSInfrastructure) {
    this.infrastructure = infrastructure;
    this.username = username || process.env.USER || 'unknown';
    this.hostname = 'localhost'; // Temporary, set by init()
    this.uuid = this.getUserUUID(this.username);
  }
  
  async init(): Promise<this> {
    if (this.infrastructure) {
      try {
        const env: EnvironmentModel = await this.infrastructure.detectEnvironment();
        // @ts-ignore - One-time initialization of readonly field
        this.hostname = env.fqdn;
      } catch (error) {
        console.warn('⚠️  Failed to detect hostname:', error);
      }
    } else {
      // @ts-ignore - Fallback
      this.hostname = process.env.HOSTNAME || 'localhost';
    }
    return this;
  }

  getUserUUID(username: string): string {
    const hash = createHash('sha256').update(`user:${username}`).digest('hex');
    const uuid = [
      hash.substring(0, 8),
      hash.substring(8, 12),
      '4' + hash.substring(13, 16),
      ((parseInt(hash.substring(16, 17), 16) & 0x3) | 0x8).toString(16) + hash.substring(17, 20),
      hash.substring(20, 32)
    ].join('-');
    return uuid;
  }

  getScenario(): Scenario<UserModel> {
    const utcTimestamp = new Date().toISOString();
    const ownerData = {
      user: 'system',
      hostname: 'system',
      utcTimestamp,
      uuid: this.getUserUUID('system')
    };
    
    return {
      ior: {
        uuid: this.uuid,
        component: 'User',
        version: '0.3.21.1'
      },
      owner: Buffer.from(JSON.stringify(ownerData)).toString('base64'),
      model: {
        uuid: this.uuid,
        username: this.username,
        hostname: this.hostname
      }
    };
  }
  
  static async getOwnerBase64(username?: string, infrastructure?: NodeOSInfrastructure): Promise<string> {
    const user = new DefaultUser(username, infrastructure);
    await user.init();
    const scenario = user.getScenario();
    return scenario.owner;
  }
}
```

### **Architecture Improvements Delivered**

**From Functional → TRUE Radical OOP:**

**Before (Functional):**
```typescript
import os from 'os';  // ❌ Direct dependency in Layer 2

private detectHostname(): string {
  return os.hostname();  // ❌ Cannot mock, cannot test in isolation
}
```

**After (TRUE Radical OOP - JavaBean):**
```typescript
import { NodeOSInfrastructure } from '../layer1/NodeOSInfrastructure.js';

private infrastructure: NodeOSInfrastructure;

constructor() {
  this.infrastructure = new NodeOSInfrastructure();
}

public async detectAndSetEnvironment(): Promise<void> {
  const env: EnvironmentModel = await this.infrastructure.detectEnvironment();
  this.serverModel.setHostname(env.getHostname());
  // ✅ Can inject mock infrastructure for testing
}
```

**Impact:**
- ✅ **Testability**: Infrastructure can be mocked/injected
- ✅ **Maintainability**: OS calls centralized in ONE place (Layer 1)
- ✅ **Reusability**: `NodeOSInfrastructure` used by both ONCE and User
- ✅ **Clarity**: Clear layer separation (3 = contracts, 2 = logic, 1 = infrastructure)
