### **Web4 Principle 12: IOR-based Method Invocation**

**IOR URL Structure (with Protocol Layers):**
```
ior:scenario:REST:https://host:port/Component/version/uuid/method?param1=value1&param2=value2
│   │        │    │      │         │    │         │       │    │      └─ Query params → scenario properties
│   │        │    │      │         │    │         │       │    └─ Method name
│   │        │    │      │         │    │         │       └─ Instance UUID (FULL UUIDv4)
│   │        │    │      │         │    │         └─ Component version
│   │        │    │      │         │    └─ Component name
│   │        │    │      │         └─ Port
│   │        │    │      └─ Host
│   │        │    └─ Transport protocol (https, wss, etc.)
│   │        └─ Application layer (REST)
│   └─ Data format (scenario = JSON scenario format)
└─ IOR marker
```

**Protocol Stack Concept:**
- **`ior:`** - IOR marker (always first)
- **`scenario:`** - Data format layer (scenario = Web4 JSON format)
- **`REST:`** - Application protocol layer (REST = HTTP verbs)
- **`https:`** - Transport protocol layer (https = HTTP over TLS)

**Loaders:**
- **ScenarioLoader** - Handles `scenario:` protocol (JSON ↔ Scenario)
- **RESTLoader** - Handles `REST:` protocol (HTTP verbs, method mapping)
- **HTTPSLoader** - Handles `https:` protocol (TLS, TCP/IP)

**IOR Profiles (CORBA 2.3+ Failover Pattern):**

Web4 IORs support **multiple host:port profiles** for **failover** and **load balancing**:

```
ior:scenario:REST:https://primary.host:8080,failover1.host:8081,failover2.host:8082/Component/version/uuid/method
│                      │                   │                      │
│                      │                   │                      └─ Profile 3 (fallback)
│                      │                   └─ Profile 2 (failover)
│                      └─ Profile 1 (primary)
```

**Profile Resolution Strategy:**

When an IOR contains multiple profiles, loaders attempt them in order:

1. **Primary** (first profile) - Preferred endpoint
2. **Failover 1** (second profile) - If primary fails
3. **Failover 2** (third profile) - If failover 1 fails
4. **Continue** until successful or all profiles exhausted

```typescript
async function resolveIorProfiles(ior: string): Promise<Scenario<any>> {
  const profiles = parseIorProfiles(ior);  // Extract all host:port pairs
  
  for (const profile of profiles) {
    try {
      const result = await loadFromProfile(profile);
      return result;  // ✅ Success - return immediately
    } catch (error) {
      console.warn(`Profile ${profile.host}:${profile.port} failed, trying next...`);
      continue;  // ⚠️ Failed - try next profile
    }
  }
  
  throw new Error(`All IOR profiles exhausted: ${ior}`);
}
```

```
HTTP Request
  ↓
HTTPRouter.route()
  ↓
Pattern Match: /{component}/{version}/{uuid}/{method}
  ↓
IORMethodRouter.parseIOR()
  ↓

  ↓
Load via Loaders:
  1. HTTPSLoader → fetches over HTTPS
  2. RESTLoader → handles HTTP verbs (GET/POST)
  3. ScenarioLoader → parses JSON to Scenario
  ↓

  ↓

  ↓
Save via Loaders (reverse order):
  1. ScenarioLoader → serialize Scenario to JSON
  2. RESTLoader → wrap in HTTP response
  3. HTTPSLoader → send over HTTPS
```

```typescript
// layer3/Loader.interface.ts
export interface Loader extends Component<LoaderModel> {
  protocol: string;  // 'scenario', 'REST', 'https', 'wss', etc.
  
  load(ior: string): Promise<Scenario<any>>;
  save(scenario: Scenario<any>, ior: string): Promise<void>;
}

// layer2/ScenarioLoader.ts
export class ScenarioLoader implements Loader {
  protocol = 'scenario';
  
  async load(ior: string): Promise<Scenario<any>> {
    // Parse scenario:REST:https://... format
    // Delegate to next loader (RESTLoader)
    const nextProtocol = this.extractNextProtocol(ior); // 'REST'
    const nextLoader = ONCE.getLoader(nextProtocol);
    const json = await nextLoader.load(ior);
    return JSON.parse(json) as Scenario<any>;
  }
  
  async save(scenario: Scenario<any>, ior: string): Promise<void> {
    const json = JSON.stringify(scenario);
    const nextProtocol = this.extractNextProtocol(ior);
    const nextLoader = ONCE.getLoader(nextProtocol);
    await nextLoader.save(json, ior);
  }
}

// layer2/RESTLoader.ts
export class RESTLoader implements Loader {
  protocol = 'REST';
  
  async load(ior: string): Promise<string> {
    // Parse REST:https://... format
    // Delegate to next loader (HTTPSLoader)
    const method = this.extractHTTPMethod(ior); // GET, POST, etc.
    const nextProtocol = this.extractNextProtocol(ior); // 'https'
    const nextLoader = ONCE.getLoader(nextProtocol);
    return await nextLoader.load(ior, { method });
  }
}

// layer2/HTTPSLoader.ts
export class HTTPSLoader implements Loader {
  protocol = 'https';
  
  async load(ior: string, options?: any): Promise<string> {
    // Parse https://host:port/path format
    // Support IOR profiles (multiple host:port pairs)
    const profiles = this.extractProfiles(ior); // [{ host, port }, ...]
    
    // Try each profile in order (failover)
    for (const profile of profiles) {
      try {
        const url = this.buildURL(profile, ior);
        const response = await fetch(url, options);
        
        if (!response.ok) {
          throw new Error(`HTTP ${response.status}: ${response.statusText}`);
        }
        
        return await response.text(); // ✅ Success
      } catch (error) {
        console.warn(`Profile ${profile.host}:${profile.port} failed:`, error);
        // Continue to next profile
      }
    }
    
    throw new Error(`All IOR profiles exhausted: ${ior}`);
  }
  
  private extractProfiles(ior: string): Array<{ host: string; port: number }> {
    // Parse: https://host1:port1,host2:port2,host3:port3/path
    const match = ior.match(/https:\/\/([^\/]+)\//);
    if (!match) return [];
    
    const hostPortString = match[1];
    return hostPortString.split(',').map(hp => {
      const [host, portStr] = hp.split(':');
      return { host, port: parseInt(portStr, 10) };
    });
  }
  
  private buildURL(profile: { host: string; port: number }, ior: string): string {
    // Replace first host:port with this profile
    const path = ior.split('/').slice(3).join('/'); // After host:port
    return `https://${profile.host}:${profile.port}/${path}`;
  }
}
```

```typescript
export interface IOR {
  uuid: string;           // FULL UUIDv4 (e.g., "a1b2c3d4-e5f6-4789-ab12-cd34ef567890")
  component: string;      // Component name (e.g., "ONCE")
  version: string;        // Component version (e.g., "0.3.21.7")
  
  // Optional network location (for IOR URLs)
  protocol?: string;      // 'https', 'wss', etc.
  host?: string;          // Primary host (e.g., "localhost")
  port?: number;          // Primary port (e.g., 42777)
  path?: string;          // URL path (e.g., "/ONCE/0.3.21.7/uuid/method")
  
  // IOR Profiles (CORBA 2.3+ failover pattern)
  profiles?: Array<{
    host: string;         // Failover host
    port: number;         // Failover port
    priority?: number;    // Optional priority (lower = higher priority)
  }>;
  
  // Precomputed IOR string (includes profiles)
  iorString?: string;     // Full IOR URL with all profiles
}

// Helper functions
export function computeIorString(ior: IOR): string {
  if (!ior.protocol || !ior.host || !ior.port) {
    return `ior:${ior.component}/${ior.version}/${ior.uuid}`;
  }
  
  // Build host:port list (primary + profiles)
  const hostPorts = [`${ior.host}:${ior.port}`];
  if (ior.profiles && ior.profiles.length > 0) {
    ior.profiles
      .sort((a, b) => (a.priority || 999) - (b.priority || 999))
      .forEach(p => hostPorts.push(`${p.host}:${p.port}`));
  }
  
  const hosts = hostPorts.join(',');
  const path = ior.path || `/${ior.component}/${ior.version}/${ior.uuid}`;
  
  return `ior:scenario:REST:${ior.protocol}://${hosts}${path}`;
}

export function parseIorString(iorString: string): IOR {
  // Parse: ior:scenario:REST:https://host1:port1,host2:port2/Component/version/uuid/method
  const match = iorString.match(
    /^ior:(?:scenario:)?(?:REST:)?([^:]+):\/\/([^\/]+)(\/.*)?$/
  );
  
  if (!match) {
    throw new Error(`Invalid IOR string: ${iorString}`);
  }
  
  const [, protocol, hostsString, path] = match;
  const hostPorts = hostsString.split(',');
  const [primaryHost, primaryPortStr] = hostPorts[0].split(':');
  
  // Extract profiles (failover hosts)
  const profiles = hostPorts.slice(1).map((hp, idx) => {
    const [host, portStr] = hp.split(':');
    return {
      host,
      port: parseInt(portStr, 10),
      priority: idx + 1
    };
  });
  
  // Extract component, version, uuid from path
  const pathParts = (path || '').split('/').filter(p => p);
  const [component, version, uuid] = pathParts;
  
  return {
    uuid,
    component,
    version,
    protocol,
    host: primaryHost,
    port: parseInt(primaryPortStr, 10),
    path,
    profiles: profiles.length > 0 ? profiles : undefined,
    iorString
  };
}
```

```typescript
export class HTTPRouter implements Router {
    private model: RouterModel;
    
    constructor() {
        this.model = {
            uuid: crypto.randomUUID(),
            routes: [],
            statistics: { /* ... */ }
        };
    }
    
    public addRoute(route: RouteModel): void {
        this.model.routes.push(route);
        this.model.routes.sort((a, b) => b.priority - a.priority);
    }
    
    public async route(req: IncomingMessage, res: ServerResponse): Promise<void> {
        const pathname = new URL(req.url!, `http://${req.headers.host}`).pathname;
        
        for (const route of this.model.routes) {
            if (this.matchRoute(route, pathname, req.method!)) {
                await route.handler(req, res);
                return;
            }
        }
        
        // 404 Not Found
        await this.handle404(req, res);
    }
    
    private matchRoute(route: RouteModel, pathname: string, method: string): boolean {
        // Pattern matching logic (supports wildcards, params)
        return this.patternMatches(route.pattern, pathname) && 
               route.method === method;
    }
}
```

```typescript
// layer2/IORMethodRouter.ts
export class IORMethodRouter {
    private componentRegistry: Map<string, any>; // Component instances
    
    constructor() {
        this.componentRegistry = new Map();
    }
    
    public registerComponent(name: string, instance: any): void {
        this.componentRegistry.set(name, instance);
    }
    
    public async invokeMethod(ior: IOR, params: Record<string, any>): Promise<Scenario<any>> {
        const { component, version, uuid, method } = this.parseIOR(ior);
        
        // Get component instance
        const instance = this.componentRegistry.get(component);
        if (!instance) {
            throw new Error(`Component not found: ${component}`);
        }
        
        // Check if method exists
        if (typeof instance[method] !== 'function') {
            throw new Error(`Method not found: ${component}.${method}`);
        }
        
        // Set params on model (query params → scenario properties)
        Object.assign(instance.model, params);
        
        // Invoke method
        const result = await instance[method]();
        
        // Return result as scenario
        return result instanceof Object && 'ior' in result 
            ? result 
            : instance.toScenario();
    }
    
    private parseIOR(iorString: string): IOR {
        // Parse: ior:https://host:port/Component/version/uuid/method
        const match = iorString.match(/^ior:https?:\/\/[^\/]+\/([^\/]+)\/([^\/]+)\/([^\/]+)\/([^\/?\s]+)/);
        if (!match) throw new Error(`Invalid IOR: ${iorString}`);
        
        return {
            protocol: 'https',
            component: match[1],
            version: match[2],
            uuid: match[3],
            method: match[4]
        };
    }
}
```

```typescript
// Helper function for HTTPRouter
function createIORRouteHandler(iorRouter: IORMethodRouter): RouteHandler {
    return async (req: IncomingMessage, res: ServerResponse) => {
        try {
            const url = new URL(req.url!, `http://${req.headers.host}`);
            const iorString = `ior:https://${req.headers.host}${url.pathname}`;
            
            // Extract query params
            const params: Record<string, any> = {};
            url.searchParams.forEach((value, key) => {
                params[key] = value;
            });
            
            // Invoke method via IOR
            const result = await iorRouter.invokeMethod(iorString, params);
            
            // Return scenario as JSON
            res.writeHead(200, { 'Content-Type': 'application/json' });
            res.end(JSON.stringify(result));
        } catch (error: any) {
            res.writeHead(500, { 'Content-Type': 'application/json' });
            res.end(JSON.stringify({ error: error.message }));
        }
    };
}
```
