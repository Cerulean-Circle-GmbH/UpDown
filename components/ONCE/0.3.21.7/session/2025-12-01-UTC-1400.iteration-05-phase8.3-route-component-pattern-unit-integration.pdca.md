**3.2 Generic startComponent() Implementation**:
```typescript
// NodeJsOnce.ts
export class NodeJsOnce {
    private loadedComponents: Map<string, any> = new Map();  // Cache loaded components
    
    /**
     * Start a component (LITERALLY runs npm start)
     * After this, you can: const unit = new Unit().init()
     */
    async startComponent(componentName: string, version: string): Promise<any> {
        const componentKey = `${componentName}/${version}`;
        
        // Check cache first
        if (this.loadedComponents.has(componentKey)) {
            return this.loadedComponents.get(componentKey);
        }
        
        // 1. Build component path
        const componentPath = path.join(
            this.model.projectRoot,
            'components',
            componentName,
            version
        );
        
        // 2. Check if component exists
        if (!fs.existsSync(componentPath)) {
            throw new Error(`Component ${componentName}/${version} not found at ${componentPath}`);
        }
        
        // 3. Load component descriptor (component.json → scenario)
        const descriptorPath = path.join(componentPath, 'component.json');
        let implementationClassName = componentName;  // Default fallback
        
        if (fs.existsSync(descriptorPath)) {
            // component.json is a symlink to scenarios/{Component}/{version}/*.scenario.json
            const descriptorContent = fs.readFileSync(descriptorPath, 'utf-8');
            const descriptor = JSON.parse(descriptorContent);
            
            // Use implementationClassName from scenario model
            if (descriptor.model?.implementationClassName) {
                implementationClassName = descriptor.model.implementationClassName;
                console.log(`📋 Using implementationClassName: ${implementationClassName} (from descriptor)`);
            } else {
                console.log(`⚠️  No implementationClassName in descriptor, using fallback: ${componentName}`);
            }
        } else {
            console.log(`⚠️  No component.json found, using fallback className: ${componentName}`);
        }
        
        // 4. Ensure component is built (DRY: use npm start)
        // Every Web4 component has npm start that handles building automatically
        const distPath = path.join(componentPath, 'dist');
        if (!fs.existsSync(distPath)) {
            console.log(`🔨 Building ${componentName}/${version}...`);
            // ✅ DRY: Use component's own npm start (handles deps, build, everything)
            await this.runNpmStart(componentPath);
        }
        
        // 5. Dynamic import (loads compiled JS)
        // Try layer2 first (most common), then layer1, then layer3
        const possiblePaths = [
            path.join(componentPath, 'dist/ts/layer2', `${implementationClassName}.js`),
            path.join(componentPath, 'dist/ts/layer1', `${implementationClassName}.js`),
            path.join(componentPath, 'dist/ts/layer3', `${implementationClassName}.js`)
        ];
        
        let mainFile: string | null = null;
        for (const filePath of possiblePaths) {
            if (fs.existsSync(filePath)) {
                mainFile = filePath;
                break;
            }
        }
        
        if (!mainFile) {
            throw new Error(
                `Implementation class file not found for ${implementationClassName}. ` +
                `Tried: ${possiblePaths.join(', ')}`
            );
        }
        
        const componentModule = await import(mainFile);
        const ComponentClass = componentModule[implementationClassName];
        
        if (!ComponentClass) {
            throw new Error(
                `Class ${implementationClassName} not exported from ${mainFile}. ` +
                `Available exports: ${Object.keys(componentModule).join(', ')}`
            );
        }
        
        // 6. Cache and return
        this.loadedComponents.set(componentKey, ComponentClass);
        console.log(`✅ Component ${componentName}/${version} (${implementationClassName}) loaded and ready`);
        
        return ComponentClass;
    }
    
    /**
     * Run npm start in component directory (DRY: every component has this)
     * This is the standard Web4 pattern - npm start handles everything:
     * - Checks if build is needed
     * - Installs dependencies if missing
     * - Compiles TypeScript
     * - Creates symlinks
     */
    private async runNpmStart(componentPath: string): Promise<void> {
        return new Promise((resolve, reject) => {
            // ✅ Use npm start (standard Web4 component pattern)
            // This delegates to the component's own package.json scripts
            exec(`cd ${componentPath} && npm start`, (error, stdout, stderr) => {
                if (error) {
                    console.error(`❌ Build failed for ${componentPath}:`, stderr);
                    reject(new Error(`Build failed: ${stderr}`));
                } else {
                    console.log(`✅ Build complete for ${componentPath}`);
                    resolve();
                }
            });
        });
    }
}
```

1. ❌ **DRY Violation**: ONCE implements own build/load logic (duplicates web4tscomponent)

export class UnitRoute extends Route {
    private unitComponentPath: string;
    private unitInstances: Map<string, any> = new Map();
    
    static start(config: {
        pattern: string,
        method: HttpMethod,
        unitComponentPath: string,
        priority?: number,
        idProvider: IDProvider
    }): UnitRoute {
        const route = new UnitRoute();
        route.unitComponentPath = config.unitComponentPath;
        
        const scenario: Scenario<RouteModel> = {
            ior: {
                uuid: config.idProvider.create(),
                component: 'UnitRoute',
                version: '0.3.21.7'
            },
            owner: 'system',
            model: {
                uuid: config.idProvider.create(),
                name: `${config.method} ${config.pattern}`,
                pattern: config.pattern,
                method: config.method,
                priority: config.priority || 30,
                statistics: new DefaultStatistics(`UnitRoute-${config.pattern}`)
            }
        };
        route.init(scenario);
        return route;
    }
    
    async handle(req: IncomingMessage, res: ServerResponse): Promise<void> {
        // Parse UUID from URL: /unit/{uuid}
        const url = new URL(req.url || '/', `http://${req.headers.host}`);
        const uuid = url.pathname.split('/')[2];
        
        if (!uuid) {
            res.writeHead(400, { 'Content-Type': 'application/json' });
            res.end(JSON.stringify({ error: 'UUID required' }));
            return;
        }
        
        // Load or create Unit instance
        let unitInstance = this.unitInstances.get(uuid);
        if (!unitInstance) {
            // ✅ LITERALLY runs: cd components/Unit/0.3.19.1 && npm start
            // Then: const { Unit } = await import('components/Unit/0.3.19.1/dist/...')
            unitInstance = await this.startComponentUnit(uuid);
            this.unitInstances.set(uuid, unitInstance);
        }
        
        // Return Unit's scenario
        const scenario = await unitInstance.toScenario();
        res.writeHead(200, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify(scenario));
        this.model.statistics.recordHit();
    }
    
    private async startComponentUnit(uuid: string): Promise<any> {
        // For now, simplified (full implementation in future iteration)
        const { Unit } = await import(`${this.unitComponentPath}/dist/ts/layer2/Unit.js`);
        const unit = new Unit();
        await unit.init({ uuid });
        return unit;
    }
}
