**3.3 UnitRoute - Real Component Integration**:
```typescript
// UnitRoute.ts - NEW FILE!
export class UnitRoute extends Route {
    private unitComponentPath: string;
    private unitInstances: Map<string, any> = new Map();
    
    static start(config: {
        pattern: string,
        method: HttpMethod,
        unitComponentPath: string,
        priority?: number,
        idProvider: IDProvider
    }): UnitRoute {
        const route = new UnitRoute();
        route.unitComponentPath = config.unitComponentPath;
        
        const scenario: Scenario<RouteModel> = {
            ior: {
                uuid: config.idProvider.create(),
                component: 'UnitRoute',
                version: '0.3.21.7'
            },
            owner: 'system',
            model: {
                uuid: config.idProvider.create(),
                name: `${config.method} ${config.pattern}`,
                pattern: config.pattern,
                method: config.method,
                priority: config.priority || 30,
                statistics: new DefaultStatistics(`UnitRoute-${config.pattern}`)
            }
        };
        route.init(scenario);
        return route;
    }
    
    async handle(req: IncomingMessage, res: ServerResponse): Promise<void> {
        // Parse UUID from URL: /unit/{uuid}
        const url = new URL(req.url || '/', `http://${req.headers.host}`);
        const uuid = url.pathname.split('/')[2];
        
        if (!uuid) {
            res.writeHead(400, { 'Content-Type': 'application/json' });
            res.end(JSON.stringify({ error: 'UUID required' }));
            return;
        }
        
        // Load or create Unit instance
        let unitInstance = this.unitInstances.get(uuid);
        if (!unitInstance) {
            // ✅ LITERALLY runs: cd components/Unit/0.3.19.1 && npm start
            // Then: const { Unit } = await import('components/Unit/0.3.19.1/dist/...')
            unitInstance = await this.startComponentUnit(uuid);
            this.unitInstances.set(uuid, unitInstance);
        }
        
        // Return Unit's scenario
        const scenario = await unitInstance.toScenario();
        res.writeHead(200, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify(scenario));
        this.model.statistics.recordHit();
    }
    
    private async startComponentUnit(uuid: string): Promise<any> {
        // For now, simplified (full implementation in future iteration)
        const { Unit } = await import(`${this.unitComponentPath}/dist/ts/layer2/Unit.js`);
        const unit = new Unit();
        await unit.init({ uuid });
        return unit;
    }
}
```
