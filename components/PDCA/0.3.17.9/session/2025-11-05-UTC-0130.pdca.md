### **Architecture Plan**

```typescript
// TrainingModule.interface.ts
interface TrainingModule {
  model: TrainingModuleModel;
  init(scenario: Scenario): this;
  listTopics(): this;
  showTopic(topic: string): this;
}

interface TrainingModuleModel {
  topics: TrainingTopic[];  // Single source of truth
  selectedTopic?: TrainingTopic;
  output: string[];  // Model-driven output
}

interface TrainingTopic {
  id: string;
  number: number;
  emoji: string;
  title: string;
  description: string;
  requiredReading: string[];
  keyLessons: string[];
  verificationChecklist: string[];
}

// DefaultPDCA.ts
async trainAI(topic: string = ''): Promise<this> {
  const trainingModule = new TrainingModule().init({});
  
  if (!topic) {
    return trainingModule.listTopics();
  }
  
  return trainingModule.showTopic(topic);
}
```

**Key Principles:**
- ✅ Topics defined ONCE in model
- ✅ No duplication anywhere
- ✅ this.model everywhere
- ✅ Zero parameters in helpers
- ✅ Output built in model, displayed at end
