```typescript
/**
 * Write component.json scenario file after component creation
 * Web4 Principle: Configuration IS scenario - must be persisted
 * @pdca 2025-11-27-UTC-1415.iteration-01.18-dynamic-layer2-implementation-name.pdca.md
 */
private async writeComponentScenario(componentName: string, version: string): Promise<void> {
  const componentRoot = path.join(this.model.componentsDirectory, componentName, version);
  const scenarioPath = path.join(componentRoot, `${componentName}.component.json`);
  
  // Create a minimal component instance to get its scenario
  // This reads the actual implementation class name via constructor.name
  const modulePath = path.join(componentRoot, 'dist', 'ts', 'layer2', `Default${componentName}.js`);
  
  // Check if component is built
  if (!existsSync(modulePath)) {
    console.log(`   ⚠️  Component not built yet, using default scenario`);
    
    // Write minimal scenario with default implementation name
    const minimalScenario = {
      ior: {
        uuid: crypto.randomUUID(),
        component: componentName,
        version: version
      },
      owner: await this.getOwnerData(),
      model: {
        uuid: crypto.randomUUID(),
        component: componentName,
        implementationClassName: `Default${componentName}`,  // Default
        version: version,
        componentRoot: componentRoot
      }
    };
    
    const fs = await import('fs/promises');
    await fs.writeFile(scenarioPath, JSON.stringify(minimalScenario, null, 2));
    console.log(`   ✅ Created ${componentName}.component.json (default implementation)`);
    return;
  }
  
  // Component is built - load it and get real scenario
  try {
    const module = await import(modulePath);
    const ComponentClass = module[`Default${componentName}`];
    
    if (ComponentClass) {
      const instance = new ComponentClass();
      const scenario = await instance.toScenario();
      
      const fs = await import('fs/promises');
      await fs.writeFile(scenarioPath, JSON.stringify(scenario, null, 2));
      console.log(`   ✅ Created ${componentName}.component.json (from toScenario)`);
    }
  } catch (error) {
    console.warn(`   ⚠️  Could not load component for scenario generation:`, error.message);
  }
}
```

### 2.3 Components Set Their Implementation Name (UNCHANGED)
**Pattern:** Each component sets it in `init()`

```typescript
// Example: NodeJsOnce.ts
export class NodeJsOnce extends DefaultOnceKernel {
  init(scenario?: Scenario<ONCEModel>): this {
    super.init(scenario);
    
    // ✅ Set implementation class name from constructor
    if (!this.model.implementationClassName) {
      this.model.implementationClassName = this.constructor.name; // "NodeJsOnce"
    }
    
    return this;
  }
}
```

### 2.4 CLI Reads from Scenario (WITH FALLBACK)
**File:** `DefaultCLI.ts:loadComponent()`

```typescript
private async loadComponent(
  componentName: string, 
  version: string,
  componentPath: string
): Promise<Component> {
  // ✅ WEB4 PRINCIPLE: Read implementation name from scenario
  let implementationClassName = `Default${componentName}`;  // Fallback
  
  const scenarioPath = join(componentPath, `${componentName}.component.json`);
  
  if (existsSync(scenarioPath)) {
    try {
      const scenarioContent = await fs.readFile(scenarioPath, 'utf-8');
      const scenario = JSON.parse(scenarioContent);
      
      if (scenario.model?.implementationClassName) {
        implementationClassName = scenario.model.implementationClassName;
        console.log(`   ℹ️  Using ${implementationClassName} from scenario`);
      }
    } catch (error) {
      console.log(`   ℹ️  Could not read scenario, using Default${componentName}`);
    }
  } else {
    console.log(`   ℹ️  No scenario file found, using Default${componentName}`);
  }
  
  // ✅ Load the specified implementation
  const modulePath = join(componentPath, 'dist', 'ts', 'layer2', `${implementationClassName}.js`);
  
  if (!existsSync(modulePath)) {
    throw new Error(
      `Implementation not found: ${implementationClassName}\n` +
      `Expected: ${modulePath}\n` +
      `Check scenario.model.implementationClassName in: ${scenarioPath}`
    );
  }
  
  const module = await import(modulePath);
  const ComponentClass = module[implementationClassName];
  
  if (!ComponentClass) {
    throw new Error(
      `Class not exported: ${implementationClassName}\n` +
      `Available: ${Object.keys(module).join(', ')}`
    );
  }
  
  // ... rest unchanged
}
```
