  const path = await import('path');
  const componentsDir = path.dirname(path.dirname(this.model.componentRoot));
  const web4tsPath = path.join(componentsDir, 'Web4TSComponent', 'latest');
  
  // ✅ Create CLI (calculates projectRoot via calculateProjectRootInternal)
  const { Web4TSComponentCLI } = await import(
    `${web4tsPath}/dist/ts/layer5/Web4TSComponentCLI.js`
  );
  
  const cli = new Web4TSComponentCLI();
  
  // ✅ Set delegation context
  cli.component.model.context = this;
  
  // ✅ Get paths from CLI (Path Authority)
  this.model.projectRoot = cli.model.projectRoot;
  this.model.isTestIsolation = cli.model.projectRoot.includes('/test/data');
  this.model.targetDirectory = cli.model.projectRoot;
  this.model.targetComponentRoot = this.model.componentRoot;
  
  this.web4ts = cli.component;
  return this.web4ts;
}
